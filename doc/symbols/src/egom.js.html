 28 /**
 29  * Creates a new line object for an EgomModel.
 30  * @class Represents a line connecting two vertices in a model.
 31  * @param a The index of the starting vertex of the line.
 32  * @param b The index of the end vertex of the line.
 33  * @param red The red component of the line's color.
 34  * @param green The green component of the line's color.
 35  * @param blue The blue component of the line's color.
 36  * @param luminosity The luminosity of the line.
 37  * @param nx The X coordinate of the normal vector associated with the line.
 38  * @param ny The Y coordinate of the normal vector associated with the line.
 39  * @param nz The Z coordinate of the normal vector associated with the line.
 40  */
 41 function Line(a,b,red,green,blue,luminosity,nx,ny,nz) {
 42 	this.a=a;
 43 	this.b=b;
 44 	this.red=red;
 45 	this.green=green;
 46 	this.blue=blue;
 47 	this.luminosity=luminosity;
 48 	this.nx=nx;
 49 	this.ny=ny;
 50 	this.nz=nz;
 51 }

 81 /**
 82  * Creates a new 3D model object, and if a file name was given, loads the data
 83  * from the file. Only Egom XML format is supported.
 84  * @class Represents a 3D polygonal model. The Egom data stucture is optimized
 85  * for generating and editing models from the program code.
 86  * @param {string} filename The name of the file to load the model from. (optional)
 87  */
 88 function EgomModel(filename) {
 89 	this.vertices = new Array();
 90 	this.lines = new Array();
 91 	this.triangles = new Array();
 92 	this.size = 0;
 93         
 94         this.nOpaqueTriangles = 0;
 95         this.nTransparentTriangles = 0;
 96         
 97         this.filename=filename;
 98 	if(filename!==undefined) {
 99 		this.loadFromFile(filename);
100 	}
101 	
102 	this.bufferStart = 0;
103         this.bufferStartTransparent = 0;
104 	this.bufferStartLines = 0;
105 }

123 EgomModel.prototype.loadFromFile = function(filename) {
124 	var request = new XMLHttpRequest();
125 	request.open('GET', filename+"?12345", false); //timestamp added to URL to bypass cache
126 	request.send(null);
127 	modelSource = request.responseXML;
128 	
129 	var nVertices = parseInt(modelSource.getElementsByTagName("vertices")[0].getAttribute("count"));
130 	this.vertices = new Array();
131 	vertexTags = modelSource.getElementsByTagName("vertex");
132 	for(var i=0; i<nVertices; i++) {
133 		var index=vertexTags[i].getAttribute("i");
134 		if (this.vertices.length<index) {
135 			for(var j=this.vertices.length;j<index;j++) {
136 				this.vertices[j]=[0.0,0.0,0.0];
137 			}
138 		}
139 		this.vertices[index]=[
140 			vertexTags[i].getAttribute("x")/10000.0,
141 			vertexTags[i].getAttribute("y")/-10000.0,
142 			vertexTags[i].getAttribute("z")/-10000.0];
143 		if(Math.abs(this.vertices[index][0]*2)>this.size) {
144 			this.size=Math.abs(this.vertices[index][0]*2);
145 		}
146 		if(Math.abs(this.vertices[index][1]*2)>this.size) {
147 			this.size=Math.abs(this.vertices[index][1]*2);
148 		}
149 		if(Math.abs(this.vertices[index][2]*2)>this.size) {
150 			this.size=Math.abs(this.vertices[index][2]*2);
151 		}
152 	}
153 	
154 	var nLines = parseInt(modelSource.getElementsByTagName("lines")[0].getAttribute("count"));
155 	this.lines = new Array(nLines);
156 	lineTags = modelSource.getElementsByTagName("line");
157 	for(var i=0; i<nLines; i++) {
158 		this.lines[i]=new Line(
159 			lineTags[i].getAttribute("a"),
160 			lineTags[i].getAttribute("b"),
161 			lineTags[i].getAttribute("red")/255.0,
162 			lineTags[i].getAttribute("green")/255.0,
163 			lineTags[i].getAttribute("blue")/255.0,
164 			lineTags[i].getAttribute("luminosity")/255.0,
165 			lineTags[i].getAttribute("nx"),
166 			-lineTags[i].getAttribute("ny"),
167 			-lineTags[i].getAttribute("nz")
168 			);
169 	}
170 	
171 	var nTriangles = parseInt(modelSource.getElementsByTagName("triangles")[0].getAttribute("count"));
172 	this.triangles = new Array(nTriangles);
173 	triangleTags = modelSource.getElementsByTagName("triangle");
174 	for(var i=0; i<nTriangles; i++) {
175 		this.triangles[i]=new Triangle(
176 			triangleTags[i].getAttribute("a"),
177 			triangleTags[i].getAttribute("b"),
178 			triangleTags[i].getAttribute("c"),
179 			triangleTags[i].getAttribute("red")/255.0,
180 			triangleTags[i].getAttribute("green")/255.0,
181 			triangleTags[i].getAttribute("blue")/255.0,
182 			(255-triangleTags[i].getAttribute("alpha"))/255.0,
183 			triangleTags[i].getAttribute("luminosity")/255.0,
184 			triangleTags[i].getAttribute("shininess"),
185 			triangleTags[i].getAttribute("tax"),
186 			triangleTags[i].getAttribute("tay"),
187 			triangleTags[i].getAttribute("tbx"),
188 			triangleTags[i].getAttribute("tby"),
189 			triangleTags[i].getAttribute("tcx"),
190 			triangleTags[i].getAttribute("tcy"),
191 			triangleTags[i].getAttribute("nax"),
192 			-triangleTags[i].getAttribute("nay"),
193 			-triangleTags[i].getAttribute("naz"),
194 			triangleTags[i].getAttribute("nbx"),
195 			-triangleTags[i].getAttribute("nby"),
196 			-triangleTags[i].getAttribute("nbz"),
197 			triangleTags[i].getAttribute("ncx"),
198 			-triangleTags[i].getAttribute("ncy"),
199 			-triangleTags[i].getAttribute("ncz"),
200                         (triangleTags[i].hasAttribute("group")?triangleTags[i].getAttribute("group"):0)
201 			);
202                 if ((this.triangles[i].alpha<1.0)&&(this.nTransparentTriangles===0)) {
203                     this.nTransparentTriangles=nTriangles-i;
204                 }
205 	}
206         this.nOpaqueTriangles=nTriangles-this.nTransparentTriangles;
207 };

209 EgomModel.prototype.getBuffers = function(lineMode) {
210 	if(lineMode===true) {
211 		var vertexData = new Float32Array(this.lines.length*6);
212 		for(var i=0;i<this.lines.length;i++) {
213 			vertexData[i*6]=this.vertices[this.lines[i].a][0];
214 			vertexData[i*6+1]=this.vertices[this.lines[i].a][1];
215 			vertexData[i*6+2]=this.vertices[this.lines[i].a][2];
216 			vertexData[i*6+3]=this.vertices[this.lines[i].b][0];
217 			vertexData[i*6+4]=this.vertices[this.lines[i].b][1];
218 			vertexData[i*6+5]=this.vertices[this.lines[i].b][2];
219 		}
220 		var texCoordData = new Float32Array(this.lines.length*4);
221 		for(var i=0;i<this.lines.length;i++) {
222 			texCoordData[i*4]=0.0;
223 			texCoordData[i*4+1]=1.0;
224 			texCoordData[i*4+2]=1.0;
225 			texCoordData[i*4+3]=1.0;
226 		}
227 		var normalData = new Float32Array(this.lines.length*6);
228 		for(var i=0;i<this.lines.length;i++) {
229 			normalData[i*6]=this.lines[i].nx;
230 			normalData[i*6+1]=this.lines[i].ny;
231 			normalData[i*6+2]=this.lines[i].nz;
232 			normalData[i*6+3]=this.lines[i].nx;
233 			normalData[i*6+4]=this.lines[i].ny;
234 			normalData[i*6+5]=this.lines[i].nz;
235 		}
236 		var colorData = new Float32Array(this.lines.length*8);
237 		for(var i=0;i<this.lines.length;i++) {
238 			colorData[i*8]=this.lines[i].red;
239 			colorData[i*8+1]=this.lines[i].green;
240 			colorData[i*8+2]=this.lines[i].blue;
241 			colorData[i*8+3]=1.0;
242 			colorData[i*8+4]=this.lines[i].red;
243 			colorData[i*8+5]=this.lines[i].green;
244 			colorData[i*8+6]=this.lines[i].blue;
245 			colorData[i*8+7]=1.0;
246 		}
247 		var luminosityData = new Float32Array(this.lines.length*2);
248 		for(var i=0;i<this.lines.length;i++) {
249 			luminosityData[i*2]=this.lines[i].luminosity;
250 			luminosityData[i*2+1]=this.lines[i].luminosity;
251 		}
252 		var shininessData = new Float32Array(this.lines.length*2);
253 		for(var i=0;i<this.lines.length;i++) {
254 			shininessData[i*2]=0;
255 			shininessData[i*2+1]=0;
256 		}
257                 var groupIndexData = new Float32Array(this.lines.length*2);
258 		for(var i=0;i<this.lines.length;i++) {
259 			groupIndexData[i*2]=0;
260 			groupIndexData[i*2+1]=0;
261 		}
262 	} else {
263 		var vertexData = new Float32Array(this.triangles.length*9);
264 		for(var i=0;i<this.triangles.length;i++) {
265 			vertexData[i*9]=this.vertices[this.triangles[i].a][0];
266 			vertexData[i*9+1]=this.vertices[this.triangles[i].a][1];
267 			vertexData[i*9+2]=this.vertices[this.triangles[i].a][2];
268 			vertexData[i*9+3]=this.vertices[this.triangles[i].b][0];
269 			vertexData[i*9+4]=this.vertices[this.triangles[i].b][1];
270 			vertexData[i*9+5]=this.vertices[this.triangles[i].b][2];
271 			vertexData[i*9+6]=this.vertices[this.triangles[i].c][0];
272 			vertexData[i*9+7]=this.vertices[this.triangles[i].c][1];
273 			vertexData[i*9+8]=this.vertices[this.triangles[i].c][2];
274 		}
275 		var texCoordData = new Float32Array(this.triangles.length*6);
276 		for(var i=0;i<this.triangles.length;i++) {
277 			texCoordData[i*6]=this.triangles[i].tax;
278 			texCoordData[i*6+1]=this.triangles[i].tay;
279 			texCoordData[i*6+2]=this.triangles[i].tbx;
280 			texCoordData[i*6+3]=this.triangles[i].tby;
281 			texCoordData[i*6+4]=this.triangles[i].tcx;
282 			texCoordData[i*6+5]=this.triangles[i].tcy;
283 		}
284 		var normalData = new Float32Array(this.triangles.length*9);
285 		for(var i=0;i<this.triangles.length;i++) {
286 			normalData[i*9]=this.triangles[i].nax;
287 			normalData[i*9+1]=this.triangles[i].nay;
288 			normalData[i*9+2]=this.triangles[i].naz;
289 			normalData[i*9+3]=this.triangles[i].nbx;
290 			normalData[i*9+4]=this.triangles[i].nby;
291 			normalData[i*9+5]=this.triangles[i].nbz;
292 			normalData[i*9+6]=this.triangles[i].ncx;
293 			normalData[i*9+7]=this.triangles[i].ncy;
294 			normalData[i*9+8]=this.triangles[i].ncz;
295 		}
296 		var colorData = new Float32Array(this.triangles.length*12);
297 		for(var i=0;i<this.triangles.length;i++) {
298 			colorData[i*12]=this.triangles[i].red;
299 			colorData[i*12+1]=this.triangles[i].green;
300 			colorData[i*12+2]=this.triangles[i].blue;
301 			colorData[i*12+3]=this.triangles[i].alpha;
302 			colorData[i*12+4]=this.triangles[i].red;
303 			colorData[i*12+5]=this.triangles[i].green;
304 			colorData[i*12+6]=this.triangles[i].blue;
305 			colorData[i*12+7]=this.triangles[i].alpha;
306 			colorData[i*12+8]=this.triangles[i].red;
307 			colorData[i*12+9]=this.triangles[i].green;
308 			colorData[i*12+10]=this.triangles[i].blue;
309 			colorData[i*12+11]=this.triangles[i].alpha;
310 		}
311 		var luminosityData = new Float32Array(this.triangles.length*3);
312 		for(var i=0;i<this.triangles.length;i++) {
313 			luminosityData[i*3]=this.triangles[i].luminosity;
314 			luminosityData[i*3+1]=this.triangles[i].luminosity;
315 			luminosityData[i*3+2]=this.triangles[i].luminosity;
316 		}
317 		var shininessData = new Float32Array(this.triangles.length*3);
318 		for(var i=0;i<this.triangles.length;i++) {
319 			shininessData[i*3]=this.triangles[i].shininess;
320 			shininessData[i*3+1]=this.triangles[i].shininess;
321 			shininessData[i*3+2]=this.triangles[i].shininess;
322 		}
323                 var groupIndexData = new Float32Array(this.triangles.length*3);
324 		for(var i=0;i<this.triangles.length;i++) {
325 			groupIndexData[i*3]=this.triangles[i].group;
326 			groupIndexData[i*3+1]=this.triangles[i].group;
327 			groupIndexData[i*3+2]=this.triangles[i].group;
328 		}
329 	}
330 	
331 	return {
332 		"position": vertexData,
333 		"texCoord": texCoordData,
334 		"normal": normalData,
335 		"color": colorData,
336 		"luminosity": luminosityData,
337 		"shininess": shininessData,
338                 "groupIndex": groupIndexData
339 		};
340 };

342 EgomModel.prototype.setupBuffers = function(gl,program) {
343 	var bufferData=this.getBuffers();
344 	var vertexData=bufferData[0];
345 	var texCoordData=bufferData[1];
346 	var normalData=bufferData[2];
347 	var colorData=bufferData[3];
348 	var luminosityData=bufferData[4];
349 	var shininessData=bufferData[5];
350         var groupIndexData=bufferData[6];
351 	
352 	// look up where the vertex data needs to go.
353 	var positionLocation = gl.getAttribLocation(program, "a_position");
354 	// Create a buffer and put a single clipspace rectangle in
355 	// it (2 triangles)
356 	var buffer = gl.createBuffer();
357 	gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
358 	gl.bufferData(
359 		gl.ARRAY_BUFFER, 
360 		vertexData,
361 		gl.STATIC_DRAW);
362 	gl.enableVertexAttribArray(positionLocation);
363 	gl.vertexAttribPointer(positionLocation, 3, gl.FLOAT, false, 0, 0);
364 	
365 	// look up where the texture coordinates need to go.
366 	var texCoordLocation = gl.getAttribLocation(program, "a_texCoord");
367 	// provide texture coordinates for the rectangle.
368 	var texCoordBuffer = gl.createBuffer();
369 	gl.bindBuffer(gl.ARRAY_BUFFER, texCoordBuffer);
370 	gl.bufferData(
371 		gl.ARRAY_BUFFER,
372 		texCoordData,
373 		gl.STATIC_DRAW);
374 	gl.enableVertexAttribArray(texCoordLocation);
375 	gl.vertexAttribPointer(texCoordLocation, 2, gl.FLOAT, false, 0, 0);
376 	
377 	// look up where the normal data needs to go.
378 	var normalLocation = gl.getAttribLocation(program, "a_normal");
379 	var normalBuffer = gl.createBuffer();
380 	gl.bindBuffer(gl.ARRAY_BUFFER, normalBuffer);
381 	gl.bufferData(
382 		gl.ARRAY_BUFFER, 
383 		normalData,
384 		gl.STATIC_DRAW);
385 	gl.enableVertexAttribArray(normalLocation);
386 	gl.vertexAttribPointer(normalLocation, 3, gl.FLOAT, false, 0, 0);
387 	
388 	var colorLocation = gl.getAttribLocation(program, "a_color");
389 	var colorBuffer = gl.createBuffer();
390 	gl.bindBuffer(gl.ARRAY_BUFFER, colorBuffer);
391 	gl.bufferData(
392 		gl.ARRAY_BUFFER, 
393 		colorData,
394 		gl.STATIC_DRAW);
395 	gl.enableVertexAttribArray(colorLocation);
396 	gl.vertexAttribPointer(colorLocation, 4, gl.FLOAT, false, 0, 0);
397 	
398 	var luminosityLocation = gl.getAttribLocation(program, "a_luminosity");
399 	var luminosityBuffer = gl.createBuffer();
400 	gl.bindBuffer(gl.ARRAY_BUFFER, luminosityBuffer);
401 	gl.bufferData(
402 		gl.ARRAY_BUFFER, 
403 		luminosityData,
404 		gl.STATIC_DRAW);
405 	gl.enableVertexAttribArray(luminosityLocation);
406 	gl.vertexAttribPointer(luminosityLocation, 1, gl.FLOAT, false, 0, 0);
407 	
408 	var shininessLocation = gl.getAttribLocation(program, "a_shininess");
409 	var shininessBuffer = gl.createBuffer();
410 	gl.bindBuffer(gl.ARRAY_BUFFER, shininessBuffer);
411 	gl.bufferData(
412 		gl.ARRAY_BUFFER, 
413 		shininessData,
414 		gl.STATIC_DRAW);
415 	gl.enableVertexAttribArray(shininessLocation);
416 	gl.vertexAttribPointer(shininessLocation, 1, gl.FLOAT, false, 0, 0);
417         
418         var groupIndexLocation = gl.getAttribLocation(program, "a_groupIndex");
419 	var groupIndexBuffer = gl.createBuffer();
420 	gl.bindBuffer(gl.ARRAY_BUFFER, groupIndexBuffer);
421 	gl.bufferData(
422 		gl.ARRAY_BUFFER, 
423 		groupIndexData,
424 		gl.STATIC_DRAW);
425 	gl.enableVertexAttribArray(groupIndexLocation);
426 	gl.vertexAttribPointer(groupIndexLocation, 1, gl.UINT, false, 0, 0);
427 };

429 EgomModel.prototype.render = function(gl,lineMode) {
430 	if (lineMode===true) {
431 		gl.drawArrays(gl.LINES, this.bufferStartLines, 2*this.lines.length);
432 	} else {
433 		gl.drawArrays(gl.TRIANGLES, this.bufferStart, 3*this.triangles.length);
434 	}
435 };

438 EgomModel.prototype.addCuboid = function(x,y,z,width,height,depth,color,luminosity, textureCoordinates, cullFace) {
439 	// front
440 	this.vertices.push([x-width/2,y-height/2,z+depth/2]);
441 	this.vertices.push([x+width/2,y-height/2,z+depth/2]);
442 	this.vertices.push([x+width/2,y+height/2,z+depth/2]);
443 	this.vertices.push([x-width/2,y+height/2,z+depth/2]);
444 	// back
445 	this.vertices.push([x-width/2,y+height/2,z-depth/2]);
446 	this.vertices.push([x+width/2,y+height/2,z-depth/2]);
447 	this.vertices.push([x+width/2,y-height/2,z-depth/2]);
448 	this.vertices.push([x-width/2,y-height/2,z-depth/2]);
449 	// top
450 	this.vertices.push([x+width/2,y+height/2,z-depth/2]);
451 	this.vertices.push([x-width/2,y+height/2,z-depth/2]);
452 	this.vertices.push([x-width/2,y+height/2,z+depth/2]);
453 	this.vertices.push([x+width/2,y+height/2,z+depth/2]);
454 	// bottom
455 	this.vertices.push([x-width/2,y-height/2,z-depth/2]);
456 	this.vertices.push([x+width/2,y-height/2,z-depth/2]);
457 	this.vertices.push([x+width/2,y-height/2,z+depth/2]);
458 	this.vertices.push([x-width/2,y-height/2,z+depth/2]);
459 	// right
460 	this.vertices.push([x+width/2,y-height/2,z-depth/2]);
461 	this.vertices.push([x+width/2,y+height/2,z-depth/2]);
462 	this.vertices.push([x+width/2,y+height/2,z+depth/2]);
463 	this.vertices.push([x+width/2,y-height/2,z+depth/2]);
464 	// left
465 	this.vertices.push([x-width/2,y+height/2,z-depth/2]);
466 	this.vertices.push([x-width/2,y-height/2,z-depth/2]);
467 	this.vertices.push([x-width/2,y-height/2,z+depth/2]);
468 	this.vertices.push([x-width/2,y+height/2,z+depth/2]);
469 		
470 	var normals = [[0,0,1],[0,0,-1],[0,1,0],[0,-1,0],[1,0,0],[-1,0,0]];
471 	
472 	var i0 = this.vertices.length-24;
473 	
474 	for(var i=0;i<6;i++) {

492 /**
493  * Adds vertices and triangles that form a sphere (UV sphere).
494  * @param {number} x The X coordinate of the center.
495  * @param {number} y The Y coordinate of the center.
496  * @param {number} z The Z coordinate of the center.
497  * @param {number} radius The radius of the sphere.
498  * @param {number} angles The number of angles one circle should have within the sphere.
499  * @param {number[]} color The RGBA components of the desired color of the triangles.
500  * @param {number} luminosity The luminosity of the triangles.
501  * @param {number} shininess The shininess exponent of the triangles.
502  * @param {number[][]} textureCoordinates The coordinate pairs of the texture to map to the sphere: from bottom left counter-clockwise
503  * @param {boolean} cullFace Whether to omit the triangles from the inside of the sphere.
504  */
505 EgomModel.prototype.addSphere = function(x,y,z,radius,angles,color,luminosity,shininess,textureCoordinates, cullFace) {    
506     // circles with vertices indexed starting from the top, starting from XY and spinned around axis Y
507     for (var i=0;i<angles;i++) {
508         for (var j=0;j<angles;j++) {
509             this.vertices.push([x+radius*Math.sin(j*2*3.1415/angles)*Math.cos(i*2*3.1415/angles),y+radius*Math.cos(j*2*3.1415/angles),z+radius*Math.sin(i*2*3.1415/angles)*Math.sin(j*2*3.1415/angles)]);
510         }
511     }
512     
513     var i0 = this.vertices.length-angles*angles;
514     
515     if((angles%2)===1) {
516         this.vertices[i0+(angles+1)/2][0]=x;
517         this.vertices[i0+(angles+1)/2][2]=z;
518     }
519     
520     for(var i=0;i<angles;i++) {
521         // adding triangles connected to the top
522         // the vertex indices:
523         var v = [i0+(i*angles),i0+(((i+1)%angles)*angles)+1,i0+(i*angles)+1];
524         // the UV texture coordinates:
525         var uv1 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],1/(2*angles)+(i/angles),1);
526         var uv2 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],(i+1)/angles,1-2/angles);
527         var uv3 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],i/angles,1-2/angles);
528         // the normal vectors:
529         var n1 = [(this.vertices[v[0]][0]-x)/radius,(this.vertices[v[0]][1]-y)/radius,(this.vertices[v[0]][2]-z)/radius];
530         var n2 = [(this.vertices[v[1]][0]-x)/radius,(this.vertices[v[1]][1]-y)/radius,(this.vertices[v[1]][2]-z)/radius];
531         var n3 = [(this.vertices[v[2]][0]-x)/radius,(this.vertices[v[2]][1]-y)/radius,(this.vertices[v[2]][2]-z)/radius];
532         this.triangles.push(new Triangle(v[0],v[1],v[2],color[0],color[1],color[2],color[3],luminosity,shininess,
533             uv1[0],uv1[1],uv2[0],uv2[1],uv3[0],uv3[1],
534             n1[0],n1[1],n1[2],n2[0],n2[1],n2[2],n3[0],n3[1],n3[2]));
535         if(cullFace!==true) {
536             this.triangles.push(new Triangle(v[0],v[2],v[1],color[0],color[1],color[2],color[3],luminosity,shininess,
537                 uv1[0],uv1[1],uv3[0],uv3[1],uv2[0],uv2[1],
538                 -n1[0],-n1[1],-n1[2],-n3[0],-n3[1],-n3[2],-n2[0],-n2[1],-n2[2]));
539         }
540         // triangles connected to the bottom
541         if ((angles%2)===0) {
542             v = [i0+(i*angles)+angles/2,i0+(i*angles)+angles/2-1,i0+(((i+1)%angles)*angles)+angles/2-1];
543         } else {
544             v= [i0+(angles+1)/2,i0+(i*angles)+(angles-1)/2,i0+(((i+1)%angles)*angles)+(angles-1)/2];
545         }
546         uv1 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],1/(2*angles)+(i/angles),0);
547         uv2 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],i/angles,((angles%2)===0)?2/angles:1/angles);
548         uv3 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],(i+1)/angles,((angles%2)===0)?2/angles:1/angles);
549         n1 = [(this.vertices[v[0]][0]-x)/radius,(this.vertices[v[0]][1]-y)/radius,(this.vertices[v[0]][2]-z)/radius];
550         n2 = [(this.vertices[v[1]][0]-x)/radius,(this.vertices[v[1]][1]-y)/radius,(this.vertices[v[1]][2]-z)/radius];
551         n3 = [(this.vertices[v[2]][0]-x)/radius,(this.vertices[v[2]][1]-y)/radius,(this.vertices[v[2]][2]-z)/radius];
552         this.triangles.push(new Triangle(v[0],v[1],v[2],color[0],color[1],color[2],color[3],luminosity,shininess,
553             uv1[0],uv1[1],uv2[0],uv2[1],uv3[0],uv3[1],
554             n1[0],n1[1],n1[2],n2[0],n2[1],n2[2],n3[0],n3[1],n3[2]));
555         if(cullFace!==true) {
556             this.triangles.push(new Triangle(v[0],v[2],v[1],color[0],color[1],color[2],color[3],luminosity,shininess,
557                 uv1[0],uv1[1],uv3[0],uv3[1],uv2[0],uv2[1],
558                 -n1[0],-n1[1],-n1[2],-n3[0],-n3[1],-n3[2],-n2[0],-n2[1],-n2[2]));
559         }
560         // quads between the two subsequent circles
561         for (var j=1;j<angles/2-1;j++) {
562             v=[i0+(i*angles)+j,i0+(((i+1)%angles)*angles)+j+1,i0+(i*angles)+j+1];
563             uv1 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],i/angles,1-2*j/angles);
564             uv2 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],(i+1)/angles,1-2*(j+1)/angles);
565             uv3 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],i/angles,1-2*(j+1)/angles);
566             n1 = [(this.vertices[v[0]][0]-x)/radius,(this.vertices[v[0]][1]-y)/radius,(this.vertices[v[0]][2]-z)/radius];
567             n2 = [(this.vertices[v[1]][0]-x)/radius,(this.vertices[v[1]][1]-y)/radius,(this.vertices[v[1]][2]-z)/radius];
568             n3 = [(this.vertices[v[2]][0]-x)/radius,(this.vertices[v[2]][1]-y)/radius,(this.vertices[v[2]][2]-z)/radius];
569             this.triangles.push(new Triangle(v[0],v[1],v[2],color[0],color[1],color[2],color[3],luminosity,shininess,
570                 uv1[0],uv1[1],uv2[0],uv2[1],uv3[0],uv3[1],
571                 n1[0],n1[1],n1[2],n2[0],n2[1],n2[2],n3[0],n3[1],n3[2]));
572             if(cullFace!==true) {
573                 this.triangles.push(new Triangle(v[0],v[2],v[1],color[0],color[1],color[2],color[3],luminosity,shininess,
574                     uv1[0],uv1[1],uv3[0],uv3[1],uv2[0],uv2[1],
575                     -n1[0],-n1[1],-n1[2],-n3[0],-n3[1],-n3[2],-n2[0],-n2[1],-n2[2]));
576             }
577             v=[i0+(((i+1)%angles)*angles)+j+1,i0+(i*angles)+j,i0+(((i+1)%angles)*angles)+j];
578             uv1 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],(i+1)/angles,1-2*(j+1)/angles);
579             uv2 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],i/angles,1-2*j/angles);
580             uv3 = uvCoordsOnTexture(textureCoordinates[0][0],textureCoordinates[0][1],textureCoordinates[2][0],textureCoordinates[2][1],(i+1)/angles,1-2*j/angles);
581             n1 = [(this.vertices[v[0]][0]-x)/radius,(this.vertices[v[0]][1]-y)/radius,(this.vertices[v[0]][2]-z)/radius];
582             n2 = [(this.vertices[v[1]][0]-x)/radius,(this.vertices[v[1]][1]-y)/radius,(this.vertices[v[1]][2]-z)/radius];
583             n3 = [(this.vertices[v[2]][0]-x)/radius,(this.vertices[v[2]][1]-y)/radius,(this.vertices[v[2]][2]-z)/radius];
584             this.triangles.push(new Triangle(v[0],v[1],v[2],color[0],color[1],color[2],color[3],luminosity,shininess,
585                 uv1[0],uv1[1],uv2[0],uv2[1],uv3[0],uv3[1],
586                 n1[0],n1[1],n1[2],n2[0],n2[1],n2[2],n3[0],n3[1],n3[2]));
587             if(cullFace!==true) {
588                 this.triangles.push(new Triangle(v[0],v[2],v[1],color[0],color[1],color[2],color[3],luminosity,shininess,
589                     uv1[0],uv1[1],uv3[0],uv3[1],uv2[0],uv2[1],
590                     -n1[0],-n1[1],-n1[2],-n3[0],-n3[1],-n3[2],-n2[0],-n2[1],-n2[2]));
591             }    
592         }
593     }
594 };

668 /**

670  * with the given properties.
671  * @param {number} width Size of the box along the X axis.
672  * @param {number} height Size of the box along the Y axis.

689 	result.vertices.push([-width/2,-height/2,-depth/2]);
690 	// top
691 	result.vertices.push([width/2,height/2,-depth/2]);
692 	result.vertices.push([-width/2,height/2,-depth/2]);
693 	result.vertices.push([-width/2,height/2,depth/2]);

708 	result.vertices.push([-width/2,-height/2,-depth/2]);

710 	
711 	var normals = [[0,0,1],[0,0,-1],[0,1,0],[0,-1,0],[1,0,0],[-1,0,0]];

716 		result.triangles.push(new Triangle((i*4)+2,(i*4)+1,(i*4)+0,color[0],color[1],color[2],color[3],128,0, 0,1, 1,1, 1,0, -normals[i][0], -normals[i][1], -normals[i][2], -normals[i][0], -normals[i][1], -normals[i][2], -normals[i][0], -normals[i][1], -normals[i][2])); 
717 		result.triangles.push(new Triangle((i*4)+0,(i*4)+3,(i*4)+2,color[0],color[1],color[2],color[3],128,0, 1,0, 0,0, 0,1, -normals[i][0], -normals[i][1], -normals[i][2], -normals[i][0], -normals[i][1], -normals[i][2], -normals[i][0], -normals[i][1], -normals[i][2])); 
718 	}
719 		
720 	return result;
721 }
