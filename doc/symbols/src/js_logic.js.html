298 function DustParticle(graphicsContext,shader,positionMatrix) {
299     this.visualModel = new PointParticle(
300         graphicsContext.resourceCenter.addModel(dustModel([0.5,0.5,0.5]),"dust"),
301         shader,
302         [0.5,0.5,0.5],
303         positionMatrix
304         );
305     graphicsContext.scene.addObject(this.visualModel);
306 	
307     this.toBeDeleted = false;
308 }
309 
310 DustParticle.prototype.simulate = function(camera) {
311     this.visualModel.shift=[-camera.velocityVector[0],-camera.velocityVector[1],-camera.velocityVector[2]];
312     if(this.visualModel.positionMatrix[12]>-camera.positionMatrix[12]+25.0) {
313         this.visualModel.positionMatrix[12]-=50.0;
314     } else if(this.visualModel.positionMatrix[12]<-camera.positionMatrix[12]-25.0) {
315         this.visualModel.positionMatrix[12]+=50.0;
316     }
317     if(this.visualModel.positionMatrix[13]>-camera.positionMatrix[13]+25.0) {
318         this.visualModel.positionMatrix[13]-=50.0;
319     } else if(this.visualModel.positionMatrix[13]<-camera.positionMatrix[13]-25.0) {
320         this.visualModel.positionMatrix[13]+=50.0;
321     }
322     if(this.visualModel.positionMatrix[14]>-camera.positionMatrix[14]+25.0) {
323         this.visualModel.positionMatrix[14]-=50.0;
324     } else if(this.visualModel.positionMatrix[14]<-camera.positionMatrix[14]-25.0) {
325         this.visualModel.positionMatrix[14]+=50.0;
326     }
327     this.visualModel.matrix=this.visualModel.positionMatrix;
328 };

330 /**
331  * Defines a dust cloud.
332  * @class Represents a dust cloud instantiated for a certain level.
333  * @param {GraphicsContext} graphicsContext
334  * @param {DustCloudClass} dustCloudClass
335  * @returns {DustCloud}

358 function Projectile(resourceCenter,scene,projectileClass,positionMatrix,orientationMatrix,muzzleFlashPositionMatrix,spacecraft,weapon) {
359 	this.class=projectileClass;
360 	this.visualModel = new Billboard(
361 		resourceCenter.addModel(projectileModel(this.class.intersections),"projectileModel-"+this.class.name),
362 		resourceCenter.getShader(projectileClass.shaderName),
363 		resourceCenter.getTexture(projectileClass.textureFileName),
364 		projectileClass.size,
365 		positionMatrix,
366 		orientationMatrix
367 		);
368 	var muzzleFlash = new DynamicParticle(
369 		resourceCenter.addModel(squareModel(),"squareModel"),
370 		resourceCenter.getShader(projectileClass.muzzleFlashShaderName),
371 		resourceCenter.getTexture(projectileClass.muzzleFlashTextureFilename),
372 		projectileClass.muzzleFlashColor,
373 		projectileClass.size,
374 		muzzleFlashPositionMatrix,
375 		500
376 		);
377 	this.physicalModel=new PhysicalObject(projectileClass.mass,projectileClass.size,positionMatrix,orientationMatrix,spacecraft.physicalModel.velocityMatrix,[]);
378 	
379 	this.timeLeft=projectileClass.duration;
380 	
381 	this.origin=spacecraft;
382 	
383 	scene.objects.push(this.visualModel);
384 	weapon.visualModel.addSubnode(muzzleFlash);
385 	
386 	this.toBeDeleted = false;
387 }
388 
389 Projectile.prototype.simulate = function(dt,hitObjects) {
390 	this.timeLeft-=dt;
391 	if(this.timeLeft<=0) {
392 		this.toBeDeleted = true;
393 		this.visualModel.toBeDeleted=true;
394 		delete this.visualModel;
395 		delete this.physicalModel;
396 		delete this;
397 	} else {
398 		this.physicalModel.simulate(dt);
399 		this.visualModel.positionMatrix=this.physicalModel.positionMatrix;
400 		this.visualModel.orientationMatrix=this.physicalModel.orientationMatrix;
401 		var positionVector=getPositionVector4(this.physicalModel.positionMatrix);
402 		for(var i=0;i<hitObjects.length;i++) {
403 			if ((hitObjects[i]!==this.origin)&&(hitObjects[i].physicalModel.checkHit(positionVector,[],0))) {
404 				this.timeLeft=0;
405 			}
406 		}
407 	}
408 };

418 Weapon.prototype.fire = function(resourceCenter,scene,projectiles,positionMatrix,orientationMatrix,scalingMatrix) {
419 	// check cooldown
420         var curTime = new Date();
421         if ((curTime-this.lastFireTime)>this.class.cooldown) {
422             this.lastFireTime=curTime;
423             var weaponSlotPosVector = vector4Matrix4Product(getPositionVector4(this.slot.positionMatrix),mul(scalingMatrix,orientationMatrix));
424             var projectilePosMatrix = mul(positionMatrix,translationMatrixv(weaponSlotPosVector));
425             var projectileOriMatrix = mul(this.slot.orientationMatrix,orientationMatrix);
426             for(var i=0;i<this.class.barrels.length;i++) {
427                     var barrelPosVector = vector3Matrix3Product(this.class.barrels[i].positionVector,matrix3from4(mul(this.slot.orientationMatrix,mul(scalingMatrix,orientationMatrix))));
428                     var muzzleFlashPosMatrix = translationMatrixv(this.class.barrels[i].positionVector);
429                     var p = new Projectile(
430                             resourceCenter,
431                             scene,
432                             this.class.barrels[i].projectileClass,
433                             mul(projectilePosMatrix,translationMatrixv(barrelPosVector)),
434                             projectileOriMatrix,
435                             muzzleFlashPosMatrix,
436                             this.spacecraft,
437                             this);
438                     projectiles.push(p);
439                     p.physicalModel.forces.push(new Force("",this.class.barrels[i].force,[projectileOriMatrix[4],projectileOriMatrix[5],projectileOriMatrix[6]],TIME_UNIT));
440             }
441         }
442         
443 };

450 function Propulsion(propulsionClass,drivenPhysicalObject) {
451 	this.class=propulsionClass;
452 	this.drivenPhysicalObject=drivenPhysicalObject;
453 	this.thrusterBurn={
454 		"forward":    0,
455 		"reverse":    0,
456 		"slideLeft":  0,
457 		"slideRight": 0,
458 		"raise":      0,
459 		"lower":      0,
460 		
461 		"yawLeft":   0,
462 		"yawRight":  0,
463 		"pitchUp":   0,
464 		"pitchDown": 0,
465 		"rollLeft":  0,
466 		"rollRight": 0
467 	};
468 }
469 
470 Propulsion.prototype.simulate = function(dt) {
471 	var directionVector = [this.drivenPhysicalObject.orientationMatrix[4],this.drivenPhysicalObject.orientationMatrix[5],this.drivenPhysicalObject.orientationMatrix[6]];
472 	var yawAxis = [this.drivenPhysicalObject.orientationMatrix[8],this.drivenPhysicalObject.orientationMatrix[9],this.drivenPhysicalObject.orientationMatrix[10]];
473 	var pitchAxis = [this.drivenPhysicalObject.orientationMatrix[0],this.drivenPhysicalObject.orientationMatrix[1],this.drivenPhysicalObject.orientationMatrix[2]];
474         
475 	if(this.thrusterBurn["forward"]>0.001) {
476 		this.drivenPhysicalObject.addOrRenewForce("forwardThrust",2*this.class.thrust*this.thrusterBurn["forward"],directionVector,TIME_UNIT);
477 	}
478 	if(this.thrusterBurn["reverse"]>0.001) {
479 		this.drivenPhysicalObject.addOrRenewForce("reverseThrust",-2*this.class.thrust*this.thrusterBurn["reverse"],directionVector,TIME_UNIT);
480 	}
481 	if(this.thrusterBurn["slideRight"]>0.001) {
482 		this.drivenPhysicalObject.addOrRenewForce("slideRightThrust",2*this.class.thrust*this.thrusterBurn["slideRight"],pitchAxis,TIME_UNIT);
483 	}
484 	if(this.thrusterBurn["slideLeft"]>0.001) {
485 		this.drivenPhysicalObject.addOrRenewForce("slideLeftThrust",-2*this.class.thrust*this.thrusterBurn["slideLeft"],pitchAxis,TIME_UNIT);
486 	}
487 	if(this.thrusterBurn["raise"]>0.001) {
488 		this.drivenPhysicalObject.addOrRenewForce("raiseThrust",2*this.class.thrust*this.thrusterBurn["raise"],yawAxis,TIME_UNIT);
489 	}
490 	if(this.thrusterBurn["lower"]>0.001) {
491 		this.drivenPhysicalObject.addOrRenewForce("lowerThrust",-2*this.class.thrust*this.thrusterBurn["lower"],yawAxis,TIME_UNIT);
492 	}
493 	if(this.thrusterBurn["yawRight"]>0.001) {
494 		this.drivenPhysicalObject.addOrRenewTorque("yawRightThrust",2*this.class.angularThrust*this.thrusterBurn["yawRight"],yawAxis,TIME_UNIT);
495 	}
496 	if(this.thrusterBurn["yawLeft"]>0.001) {
497 		this.drivenPhysicalObject.addOrRenewTorque("yawLeftThrust",-2*this.class.angularThrust*this.thrusterBurn["yawLeft"],yawAxis,TIME_UNIT);
498 	}
499 	if(this.thrusterBurn["pitchUp"]>0.001) {
500 		this.drivenPhysicalObject.addOrRenewTorque("pitchUpThrust",-2*this.class.angularThrust*this.thrusterBurn["pitchUp"],pitchAxis,TIME_UNIT);
501 	}
502 	if(this.thrusterBurn["pitchDown"]>0.001) {
503 		this.drivenPhysicalObject.addOrRenewTorque("pitchDownThrust",2*this.class.angularThrust*this.thrusterBurn["pitchDown"],pitchAxis,TIME_UNIT);
504 	}
505 	if(this.thrusterBurn["rollRight"]>0.001) {
506 		this.drivenPhysicalObject.addOrRenewTorque("rollRightThrust",-2*this.class.angularThrust*this.thrusterBurn["rollRight"],directionVector,TIME_UNIT);
507 	}
508 	if(this.thrusterBurn["rollLeft"]>0.001) {
509 		this.drivenPhysicalObject.addOrRenewTorque("rollLeftThrust",2*this.class.angularThrust*this.thrusterBurn["rollLeft"],directionVector,TIME_UNIT);
510 	}
511 };

540 /**
541  * Assigns the controller property and makes sure that the controller's 
542  * controlled entity is updated as well to this one.
543  * @param {Controller} newController The new value of controller.
544  */
545 ControllableEntity.prototype.setController = function(newController) {
546     if ((this.controller!==newController)&&(newController!==undefined)) {
547         if ((newController!==null)&&(newController.getControlledEntity()!==this)) {
548             newController.setControlledEntityWithoutChecks(this);
549         }
550         if((this.controller!==null)&&(this.controller!==undefined)&&(this.controller.getControlledEntity()===this)) {
551             this.controller.setControlledEntityWithoutChecks(null);

760 Spacecraft.prototype.addDirectionalThrusterBurn = function(directionVector,value) {
761 	if(value<0) {
762 		value=-value;
763 		directionVector[0]=-directionVector[0];
764 		directionVector[1]=-directionVector[1];
765 		directionVector[2]=-directionVector[2];
766 	}
767 	var relativeDirection = vector3Matrix3Product(directionVector,matrix3from4(this.physicalModel.modelMatrixInverse));
768 	if(relativeDirection[0]>0.0001) {
769 		this.addThrusterBurn("slideRight",relativeDirection[0]*value);
770 	}
771 	if(relativeDirection[0]<-0.0001) {
772 		this.addThrusterBurn("slideLeft",-relativeDirection[0]*value);
773 	}
774 	if(relativeDirection[1]>0.0001) {
775 		this.addThrusterBurn("forward",relativeDirection[1]*value);
776 	}
777 	if(relativeDirection[1]<-0.0001) {
778 		this.addThrusterBurn("reverse",-relativeDirection[1]*value);
779 	}
780 	if(relativeDirection[2]>0.0001) {
781 		this.addThrusterBurn("raise",relativeDirection[2]*value);
782 	}
783 	if(relativeDirection[2]<-0.0001) {
784 		this.addThrusterBurn("lower",-relativeDirection[2]*value);
785 	}
786 };

1015 /**
1016  * Constructs and returns a rotation matrix described by a series of rotations
1017  * stored in the XML tags.
1018  * @param {XMLElement[]} tags The tags describing rotations.
1019  * @returns {Float32Array} The costructed rotation matrix.
1020  */
1021 function getRotationMatrixFromXMLTags(tags) {
1022     var result = identityMatrix4();
1023     for(var i=0;i<tags.length;i++) {
1024         var axis=[0,0,0];
1025         if (tags[i].getAttribute("axis")==="x") {
1026                 axis=[1,0,0];
1027         } else
1028         if (tags[i].getAttribute("axis")==="y") {
1029                 axis=[0,1,0];
1030         } else
1031         if (tags[i].getAttribute("axis")==="z") {
1032                 axis=[0,0,1];
1033         }
1034         result=
1035                 mul(
1036                         result,
1037                         rotationMatrix4(
1038                                 axis,
1039                                 parseFloat(tags[i].getAttribute("degree"))/180*3.1415
1040                                 )
1041                         );
1042     }
1043     return result;
1044 }

1340 			this.spacecrafts[i].simulate(dt);
1341 		}
1342 	}

1349 		}
1350 	}
