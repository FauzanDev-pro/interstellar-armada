 35 function KeyboardControlContext() {
 36     var currentlyPressedKeys = new Array(256);
 37     var i;
 38     //var keyPressEvents = new Array();
 39     var keyCodeTable = {
 40         "backspace": 8,
 41         "tab": 9,
 42         "enter": 13,
 43         "shift": 16,
 44         "ctrl": 17,
 45         "alt": 18,
 46         "pause": 19,
 47         "caps lock": 20,
 48         "escape": 27,
 49         "space": 32,
 50         "page up": 33,
 51         "page down": 34,
 52         "end": 35,
 53         "home": 36,
 54         "left": 37,
 55         "up": 38,
 56         "right": 39,
 57         "down": 40,
 58         "insert": 45,
 59         "delete": 46,
 60         "0": 48, "1": 49, "2": 50, "3": 51, "4": 52, "5": 53, "6": 54, "7": 55, "8": 56, "9": 57,
 61         "a": 65, "b": 66, "c": 67, "d": 68, "e": 69, "f": 70, "g": 71, "h": 72, "i": 73, "j": 74,
 62         "k": 75, "l": 76, "m": 77, "n": 78, "o": 79, "p": 80, "q": 81, "r": 82, "s": 83, "t": 84,
 63         "u": 85, "v": 86, "w": 87, "x": 88, "y": 89, "z": 90,
 64         "left window": 91, "right window": 92, "select": 93,
 65         "numpad 0": 96, "numpad 1": 97, "numpad 2": 98, "numpad 3": 99, "numpad 4": 100,
 66         "numpad 5": 101, "numpad 6": 102, "numpad 7": 103, "numpad 8": 104, "numpad 9": 105,
 67         "f1": 112, "f2": 113, "f3": 114, "f4": 115, "f5": 116, "f6": 117, "f7": 118, "f8": 119, "f9": 120,
 68         "f10": 121, "f11": 122, "f12": 123
 69     };
 70      var keyCommands={};
 71     
 72     for(i=0;i<currentlyPressedKeys.length;i++) { currentlyPressedKeys[i]=false; }
 73 
 74     this.handleKeyDown = function(event) {
 75 	currentlyPressedKeys[event.keyCode] = true;
 76     };  
 77 
 78     this.handleKeyUp = function(event) {
 79 	currentlyPressedKeys[event.keyCode] = false;
 80     };
 81     
 82     /**
 83      * Creates a new keyboard command object.
 84      * @class Represents a keypress - command association.
 85      * @param {string} name Name of the command. The different controllers identify the command association by this name.
 86      * @param {string} key The string representation of the pressed key (see keyCodeTable).
 87      * @param {boolean} shiftState Whether the shift key should be held pressed while activating this command.
 88      * @param {boolean} ctrlState Whether the control key should be held pressed while activating this command.
 89      * @param {boolean} altState Whether the alt key should be held pressed while activating this command.
 90      */
 91     this.KeyCommand = function(name,key,shiftState,ctrlState,altState) {
 92         this.name=name;
 93         this.keyCode=keyCodeTable[key];
 94         this.shiftState=shiftState;
 95         this.ctrlState=ctrlState;
 96         this.altState=altState;
 97         var oneShotExecuted=false;
 98         
 99         /**
100          * A function that can be executed with CheckAndExecute automatically
101          * upon every new press (following a previous release) of the key(s).
102          */
103         this.executeOneShotCommand=function() {};
104         /**
105          * A function that can be executed with CheckAndExecute automatically
106          * continuously while the key(s) are pressed.
107          */
108         this.executeContinuousCommand=function() {};
109         /**
110          * Checks if the key(s) are pressed currently, which means the continuous
111          * commands should be executed.
112          * @returns {boolean} Whether the continuous command should be executed now.
113          */
114         this.checkContinuous=function () {
115             return (currentlyPressedKeys[this.keyCode] &&
116                     (currentlyPressedKeys[16]===this.shiftState) &&
117                     (currentlyPressedKeys[17]===this.ctrlState) &&
118                     (currentlyPressedKeys[18]===this.altState));
119         };
120         /**
121          * Checks if the one shot command should be executed and updates the
122          * state as if it was, but without actually executing it (allowing an
123          * external code block to be used for this purpose, making it possible
124          * to use local variables of an external block.
125          * @returns {boolean} Whether the one shot command should be executed now.
126          */
127         this.checkAndSetOneShot=function () {
128             if (currentlyPressedKeys[this.keyCode] &&
129                     (currentlyPressedKeys[16]===this.shiftState) &&
130                     (currentlyPressedKeys[17]===this.ctrlState) &&
131                     (currentlyPressedKeys[18]===this.altState)) {
132                 if (!oneShotExecuted) {
133                     oneShotExecuted=true;
134                     return true;
135                 }
136             } else {
137                 oneShotExecuted=false;
138             }
139             return false;
140         };
141         /**
142          * Checks the current state of keys and executes both the one shot and
143          * continuous commands accordingly as well as updates the state for further
144          * checks.
145          */
146         this.checkAndExecute=function () {
147             if (currentlyPressedKeys[this.keyCode] &&
148                     (currentlyPressedKeys[16]===this.shiftState) &&
149                     (currentlyPressedKeys[17]===this.ctrlState) &&
150                     (currentlyPressedKeys[18]===this.altState)) {
151                 if (!oneShotExecuted) {
152                     this.executeOneShotCommand();
153                     oneShotExecuted=true;
154                 }
155                 this.executeContinuousCommand();
156             } else {
157                 oneShotExecuted=false;
158             }
159         };
160     };
161     
162     this.addKeyCommand = function(keyCommand) {
163         keyCommands[keyCommand.name]=keyCommand;
164     };
165     
166     this.setOneShotActionForCommand = function(commandName,action) {
167         if(keyCommands[commandName]!==undefined) {
168             keyCommands[commandName].executeOneShotCommand=action;
169         }
170         return keyCommands[commandName];
171     };
172     
173     this.setContinuousActionForCommand = function(commandName,action) {
174         if(keyCommands[commandName]!==undefined) {
175             keyCommands[commandName].executeContinuousCommand=action;
176         }
177         return keyCommands[commandName];
178     };
179 }

181 /**
182  * Creates a new controller object.
183  * @class The parent class for all controller objects that control a certain
184  * entity from the logic module.
185  * @param {ControllableEntity} controlledEntity The entity which is controlled by the controller.
186  * @param {GraphicsContext} graphicsContext The graphics context where the visual model of the controlled entity resides.
187  * @param {LogicContext} logicContext The logic context of the controlled entity.
188  * @param {KeyboardControlContext} controlContext The control context supplying the input data (currently only keyboard).
189  */
190 function Controller(controlledEntity,graphicsContext,logicContext,controlContext) {
191     this.graphicsContext=graphicsContext;
192     this.logicContext=logicContext;
193     this.controlContext=controlContext;
194     this.setControlledEntity(controlledEntity);
195 }

210 /**
211  * Assigns the controlledEntity property and makes sure the entity's controller
212  * is also updated.
213  * @param {ControllableEntity} newControlledEntity The new value of controlledEntity.
214  */
215 Controller.prototype.setControlledEntity = function(newControlledEntity) {
216     if ((this.controlledEntity!==newControlledEntity)&&(newControlledEntity!==undefined)) {
217         if ((this.controlledEntity!==null)&&(this.controlledEntity!==undefined)) {
218             this.controlledEntity.setControllerWithoutChecks(null);
219         }
220         this.controlledEntity=newControlledEntity;
221         if (newControlledEntity!==null) {
222             newControlledEntity.setControllerWithoutChecks(this);
223         }
224     }

249 CameraController.prototype.control = function() {
250     var camera=this.controlledEntity;
251     var inverseOrientationMatrix;
252     var translationVector;
253     var rotationMatrix;
254 
255     if(camera.controllableDirection) {
256 	if (this.turnLeftCommand.checkContinuous()) {
257                 if (camera.angularVelocityVector[1]<camera.maxTurn) {
258                                 camera.angularVelocityVector[1]+=camera.angularAcceleration;
259                 }
260         } else {
261                 if (camera.angularVelocityVector[1]>0) {
262                         camera.angularVelocityVector[1]-=
263                                 Math.min(camera.angularAcceleration,camera.angularVelocityVector[1]);
264                 }
265         }
266 	if (this.turnRightCommand.checkContinuous()) {
267                 if (camera.angularVelocityVector[1]>-camera.maxTurn) {
268                         camera.angularVelocityVector[1]-=camera.angularAcceleration;
269                 }
270         } else {
271                 if (camera.angularVelocityVector[1]<0) {
272                         camera.angularVelocityVector[1]+=
273                                 Math.min(camera.angularAcceleration,-camera.angularVelocityVector[1]);
274                 }
275         }
276 	if (this.turnUpCommand.checkContinuous()) {
277                 if (camera.angularVelocityVector[0]<camera.maxTurn) {
278                                 camera.angularVelocityVector[0]+=camera.angularAcceleration;
279                 }
280         } else {
281                 if (camera.angularVelocityVector[0]>0) {
282                         camera.angularVelocityVector[0]-=
283                                 Math.min(camera.angularAcceleration,camera.angularVelocityVector[0]);
284                 }
285         }
286 	if (this.turnDownCommand.checkContinuous()) {
287                 if (camera.angularVelocityVector[0]>-camera.maxTurn) {
288                         camera.angularVelocityVector[0]-=camera.angularAcceleration;
289                 }
290         } else {
291                 if (camera.angularVelocityVector[0]<0) {
292                         camera.angularVelocityVector[0]+=
293                                 Math.min(camera.angularAcceleration,-camera.angularVelocityVector[0]);
294                 }
295         }
296     }
297     if(camera.controllablePosition) {
298 	if (this.moveLeftCommand.checkContinuous()) {
299                 if (camera.velocityVector[0]<camera.maxSpeed) {
300                         camera.velocityVector[0]+=camera.acceleration;
301                 }
302         } else {
303                 if (camera.velocityVector[0]>0) {
304                         camera.velocityVector[0]-=
305                                 Math.min(camera.acceleration,camera.velocityVector[0]);
306                 }
307         }
308         if (this.moveRightCommand.checkContinuous()) {
309                 if (camera.velocityVector[0]>-camera.maxSpeed) {
310                         camera.velocityVector[0]-=camera.acceleration;
311                 }
312         } else {
313                 if (camera.velocityVector[0]<0) {
314                         camera.velocityVector[0]+=
315                                 Math.min(camera.acceleration,-camera.velocityVector[0]);
316                 }
317         }
318         if (this.moveDownCommand.checkContinuous()) {
319                 if (camera.velocityVector[1]<camera.maxSpeed) {
320                         camera.velocityVector[1]+=camera.acceleration;
321                 }
322         } else {
323                 if (camera.velocityVector[1]>0) {
324                         camera.velocityVector[1]-=
325                                 Math.min(camera.acceleration,camera.velocityVector[1]);
326                 }
327         }
328         if (this.moveUpCommand.checkContinuous()) {
329                 if (camera.velocityVector[1]>-camera.maxSpeed) {
330                         camera.velocityVector[1]-=camera.acceleration;
331                 }
332         } else {
333                 if (camera.velocityVector[1]<0) {
334                         camera.velocityVector[1]+=
335                                 Math.min(camera.acceleration,-camera.velocityVector[1]);
336                 }
337         }
338         if (this.moveForwardCommand.checkContinuous()) {
339                 if (camera.velocityVector[2]<camera.maxSpeed) {
340                         camera.velocityVector[2]+=camera.acceleration;
341                 }
342         } else {
343                 if (camera.velocityVector[2]>0) {
344                         camera.velocityVector[2]-=
345                                 Math.min(camera.acceleration,camera.velocityVector[2]);
346                 }
347         }
348         if (this.moveBackwardCommand.checkContinuous()) {
349                 if (camera.velocityVector[2]>-camera.maxSpeed) {
350                         camera.velocityVector[2]-=camera.acceleration;
351                 }
352         } else {
353                 if (camera.velocityVector[2]<0) {
354                         camera.velocityVector[2]+=
355                                 Math.min(camera.acceleration,-camera.velocityVector[2]);
356                 }
357         }
358     }
359 
360     if (camera.controllablePosition) {
361         inverseOrientationMatrix=transposed3(inverse3(matrix3from4(camera.orientationMatrix)));
362         translationVector = matrix3Vector3Product(
363             camera.velocityVector,
364             inverseOrientationMatrix
365             );
366         if(camera.followedObject===undefined) {
367             camera.positionMatrix=
368                     mul(
369                             camera.positionMatrix,
370                             translationMatrixv(translationVector)
371                             );
372         } else {
373             camera.followPositionMatrix=
374                     mul(
375                             camera.followPositionMatrix,
376                             translationMatrixv(translationVector)
377                             );
378         }
379     }
380     if (camera.controllableDirection) {
381         if(camera.followedObject===undefined) {
382             rotationMatrix=
383                 mul(
384                     rotationMatrix4(
385                                 [0,1,0],
386                                 camera.angularVelocityVector[1]
387                                 ),
388                     rotationMatrix4(
389                                 [1,0,0],
390                                 camera.angularVelocityVector[0]
391                                 )    
392                     );
393             camera.orientationMatrix=mul(camera.orientationMatrix,rotationMatrix);
394         } else {
395             rotationMatrix=
396                 mul(
397                     rotationMatrix4(
398                                 [0,0,1],
399                                 camera.angularVelocityVector[1]
400                                 ),
401                     rotationMatrix4(
402                                 [1,0,0],
403                                 camera.angularVelocityVector[0]
404                                 )    
405                     );
406             camera.followOrientationMatrix=mul(camera.followOrientationMatrix,rotationMatrix);
407         }
408     }
409             
410     if (this.decreaseFOVCommand.checkContinuous()) {
411             camera.setFOV(camera.fov-1);
412     }
413     if (this.increaseFOVCommand.checkContinuous()) {
414             camera.setFOV(camera.fov+1);
415     }
416     
417     if (camera.followedObject!==undefined) {
418         // look in direction y instead of z:
419         var newOrientationMatrix =
420                 mul(
421                         mul(
422                                 inverseRotationMatrix(camera.followedObject.getOrientationMatrix()),
423                                 camera.followOrientationMatrix
424                                 ),
425                         rotationMatrix4([1,0,0],3.1415/2)        
426                         );
427         camera.orientationMatrix=newOrientationMatrix;
428         var camPositionMatrix = 
429                 mul(
430                         mul(
431                                 camera.rotationCenterIsObject?
432                                     translationMatrixv(getPositionVector(mul(
433                                         camera.followPositionMatrix,
434                                             inverseRotationMatrix(camera.followOrientationMatrix)
435                                         )))
436                                     :
437                                     camera.followPositionMatrix,
438                                 camera.followedObject.getOrientationMatrix()
439                                 ),
440                         camera.followedObject.getPositionMatrix()
441                         );
442         var newPositionMatrix=
443                 translationMatrix(
444                         -camPositionMatrix[12],
445                         -camPositionMatrix[13],
446                         -camPositionMatrix[14]
447                         );
448         var velocityMatrix = mul(translationMatrix(
449                 newPositionMatrix[12]-camera.positionMatrix[12],
450                 newPositionMatrix[13]-camera.positionMatrix[13],
451                 newPositionMatrix[14]-camera.positionMatrix[14]),camera.orientationMatrix);
452         camera.velocityVector = [velocityMatrix[12],velocityMatrix[13],velocityMatrix[14]];
453         camera.positionMatrix=newPositionMatrix;
454     }
455 };

457 /**
458  * Defines a fighter controller instance.
459  * @class Interprets the user input for controlling a fighter and translates it
460  * to actions such as firing weapons or thrusters.
461  * @param {ControllableEntity} controlledEntity
462  * @param {GraphicsContext} graphicsContext
463  * @param {LogicContext} logicContext
464  * @param {ControlContext} controlContext
465  * @returns {FighterController}
466  */
467 function FighterController(controlledEntity,graphicsContext,logicContext,controlContext) {
468 	Controller.call(this,controlledEntity,graphicsContext,logicContext,controlContext);
469         
470         var self=this;
471         	
472 	this.FM_INERTIAL    = 0;
473 	this.FM_COMPENSATED = 1;
474 	//this.FM_RESTRICTED  = 2;
475 	
476 	this.NUM_FLIGHTMODES = 2;
477 	
478 	this.reset();
479 	
480 	this.TURNING_LIMIT = this.controlledEntity.propulsion.class.angularThrust/this.controlledEntity.physicalModel.mass*200;
481         this.TURN_TOLERANCE=0.00001; // the minimum rotation wich is compensated
482                                     // automatically by the thrusters to bring
483                                     // the craft to a halt
484         
485         this.fireCommand=controlContext.setContinuousActionForCommand("fire",function(){
486             self.controlledEntity.fire(self.graphicsContext.resourceCenter,self.graphicsContext.scene,self.logicContext.level.projectiles);
487         });
488         this.changeFlightModeCommand=controlContext.setOneShotActionForCommand("changeFlightMode",function(){
489             self.flightMode=(self.flightMode+1)%self.NUM_FLIGHTMODES;
490         });
491         this.forwardCommand=controlContext.setContinuousActionForCommand("forward",function(){
492             switch(self.flightMode) {
493                 case self.FM_INERTIAL:
494                         self.controlledEntity.addThrusterBurn("forward",0.5);
495                         break;
496                 case self.FM_COMPENSATED:
497                         self.intendedSpeed+=1;
498                         break;
499             }
500         });
501         this.reverseCommand=controlContext.setContinuousActionForCommand("reverse",function(){
502             switch(self.flightMode) {
503                 case self.FM_INERTIAL:
504                         self.controlledEntity.addThrusterBurn("reverse",0.5);
505                         break;
506                 case self.FM_COMPENSATED:
507                         self.intendedSpeed-=1;
508                         if(self.intendedSpeed<0) {
509                                 self.intendedSpeed=0;
510                         }
511                         break;
512             }
513         });
514         this.resetSpeedCommand=controlContext.setOneShotActionForCommand("resetSpeed",function(){
515             switch(self.flightMode) {
516                 case self.FM_COMPENSATED:
517                         self.intendedSpeed=0;
518                         break;
519             }
520         });
521         this.yawLeftCommand=controlContext.setContinuousActionForCommand("yawLeft",function(){});
522         this.yawRightCommand=controlContext.setContinuousActionForCommand("yawRight",function(){});
523         this.pitchDownCommand=controlContext.setContinuousActionForCommand("pitchDown",function(){});
524         this.pitchUpCommand=controlContext.setContinuousActionForCommand("pitchUp",function(){});
525         this.rollRightCommand=controlContext.setContinuousActionForCommand("rollRight",function(){});
526         this.rollLeftCommand=controlContext.setContinuousActionForCommand("rollLeft",function(){});
527 }

541 FighterController.prototype.control = function() {
542         document.getElementById('ui').innerHTML="";
543     
544 	this.fireCommand.checkAndExecute();
545 	
546 	var physicalModel = this.controlledEntity.physicalModel;
547 	
548 	var relativeVelocityMatrix = mul(
549 		physicalModel.velocityMatrix,
550 		matrix4from3(matrix3from4(physicalModel.modelMatrixInverse)));
551 	
552 	var turningMatrix = mul(
553 		mul(
554 			physicalModel.orientationMatrix,
555 			physicalModel.angularVelocityMatrix
556 			),
557 		matrix4from3(matrix3from4(physicalModel.modelMatrixInverse)));
558 	
559 	this.changeFlightModeCommand.checkAndExecute();
560         
561 	this.controlledEntity.resetThrusterBurn();
562         
563 	this.forwardCommand.checkAndExecute();
564 	this.reverseCommand.checkAndExecute();
565 	this.resetSpeedCommand.checkAndExecute();
566         
567         // compansating for drift in compensated flight mode by firing side
568         // thrusters + correcting to reach intended speed
569 	if(this.flightMode===this.FM_COMPENSATED) {
570                 document.getElementById('ui').innerHTML+="COMPENSATED flight<br/>";
571                 // checking X relative velocity component for side drift
572 		if(relativeVelocityMatrix[12]<-0.0001) {
573 			this.controlledEntity.addThrusterBurnCapped("slideRight",0.5,this.controlledEntity.getNeededBurnForSpeedChange(-relativeVelocityMatrix[12]));
574 		} else if(relativeVelocityMatrix[12]>0.0001) {
575 			this.controlledEntity.addThrusterBurnCapped("slideLeft",0.5,this.controlledEntity.getNeededBurnForSpeedChange(relativeVelocityMatrix[12]));
576 		}
577                 // checking Z relative velocity component for side drift
578 		if(relativeVelocityMatrix[14]<-0.0001) {
579 			this.controlledEntity.addThrusterBurnCapped("raise",0.5,this.controlledEntity.getNeededBurnForSpeedChange(-relativeVelocityMatrix[14]));
580 		} else if(relativeVelocityMatrix[14]>0.0001) {
581 			this.controlledEntity.addThrusterBurnCapped("lower",0.5,this.controlledEntity.getNeededBurnForSpeedChange(relativeVelocityMatrix[14]));
582 		}
583                 // correcting to reach intended speed
584 		if(relativeVelocityMatrix[13]<this.intendedSpeed-0.0001) {
585 			this.controlledEntity.addThrusterBurnCapped("forward",0.5,this.controlledEntity.getNeededBurnForSpeedChange(this.intendedSpeed-relativeVelocityMatrix[13]));
586                         document.getElementById('ui').innerHTML+="[forward] ";
587 		} else if(relativeVelocityMatrix[13]>this.intendedSpeed+0.0001) {
588 			this.controlledEntity.addThrusterBurnCapped("reverse",0.5,this.controlledEntity.getNeededBurnForSpeedChange(relativeVelocityMatrix[13]-this.intendedSpeed));
589                         document.getElementById('ui').innerHTML+="[reverse] ";
590 		}
591 	} else
592         {
593             document.getElementById('ui').innerHTML+="FREE flight<br/>";
594         }
595         
596         // controlling yaw
597         // if yaw left command is given, fire thrusters until turning speed
598         // limit is reached
599 	if (this.yawLeftCommand.checkContinuous()) {
600 		if(turningMatrix[4]>-this.TURNING_LIMIT) {
601 			this.controlledEntity.addThrusterBurn("yawLeft",0.5);
602 		}
603 	} else
604         // if yaw right command is given, fire thrusters until turning speed
605         // limit is reached
606 	if (this.yawRightCommand.checkContinuous()) {
607 		if(turningMatrix[4]<this.TURNING_LIMIT) {
608 			this.controlledEntity.addThrusterBurn("yawRight",0.5);
609 		}
610         // if there is no yaw command given, but the craft is yawing to the
611         // left, then fire yaw right thrusters to stop the yaw
612 	} else if(turningMatrix[4]<-this.TURN_TOLERANCE) {
613                 // we need to calculate how strong burn is needed to bring the
614                 // rotation to a stop (if it is less then the capacity of the
615                 // thrusters, we don't want to overshoot)
616                 // the trick: divided by 10 in the end, because the torque will
617                 // be in effect for 50 ms, and the angular velocity matrix

619                 this.controlledEntity.addThrusterBurn("yawRight",
620                         Math.min(
621                             0.5,
622                             angleDifferenceOfUnitVectors2D(

629                         Math.min(
630                             0.5,
631                             angleDifferenceOfUnitVectors2D(

633 					normalizeVector2D([turningMatrix[4],turningMatrix[5]])

635         }

637 	if (this.pitchDownCommand.checkContinuous()) {
638 		if(turningMatrix[6]>-this.TURNING_LIMIT) {
639 			this.controlledEntity.addThrusterBurn("pitchDown",0.5);
640 		}
641 	} else
642 	if (this.pitchUpCommand.checkContinuous()) {
643 		if(turningMatrix[6]<this.TURNING_LIMIT) {
644 			this.controlledEntity.addThrusterBurn("pitchUp",0.5);

646 	} else if(turningMatrix[6]<-this.TURN_TOLERANCE) {
647                 // trick: with the pitch, we are comparing to [1,0] unit vector,
648                 // because in case of a still object, the [5] component of the
649                 // turning matrix is 1 and the [6] is 0
650 		this.controlledEntity.addThrusterBurn("pitchUp",
651                         Math.min(
652                             0.5,

665 	}

670 		}
671 	} else

678                         Math.min(
679                             0.5,
680                             angleDifferenceOfUnitVectors2D(
681 					[1,0],
682 					normalizeVector2D([turningMatrix[0],turningMatrix[2]])
683 					)*physicalModel.mass/2/this.controlledEntity.propulsion.class.angularThrust/10));

685 		this.controlledEntity.addThrusterBurn("rollRight",

690 					normalizeVector2D([turningMatrix[0],turningMatrix[2]])

692 	}

703                 "mass: "+physicalModel.mass+" kg";

946 /**
947  * Initializes the global keyboard commands by looking for the appropriate associations
948  * in the given control context and setting their execution actions.
949  * @param {GraphicsContext} graphicsContext The graphics context within which to set the commands.
950  * @param {LogicContext} logicContext The logic context within which to set the commands.
951  * @param {KeyboardControlContext} controlContext The context that contains the key (combination) - action associations.
952  * @returns {KeyboardControlContext.KeyCommand[]} The global keyboard commands organized in an array.
953  */
954 function initGlobalCommands(graphicsContext,logicContext,controlContext) {
955     
956     var globalCommands=new Array();
957     var i,j;
958     
959     globalCommands.push(controlContext.setOneShotActionForCommand("pause",function(){
960         alert("Game paused.");
961     }));
962     globalCommands.push(controlContext.setOneShotActionForCommand("changeView",function(){
963         if ((graphicsContext.scene.activeCamera.followedCamera!==undefined) && 
964                 (graphicsContext.scene.activeCamera.followedCamera.nextView!==null)) {
965             graphicsContext.scene.activeCamera.followCamera(graphicsContext.scene.activeCamera.followedCamera.nextView,500);
966         }
967     }));
968     globalCommands.push(controlContext.setOneShotActionForCommand("followNext",function(){
969         // if we are currently following a camera, we have to look for the first subsequent
970         // camera that follows a different object than the current
971         if (graphicsContext.scene.activeCamera.followedCamera!==undefined) {
972             // first find the index of the first camera following the current object, iterating
973             // through the cameras from the beginning
974             i=0;
975             while ((i<graphicsContext.scene.cameras.length)&&
976                     (graphicsContext.scene.cameras[i].followedObject!==graphicsContext.scene.activeCamera.followedCamera.followedObject)) {
977                 i++;
978             }
979             // then find the first camera which has a different followed object
980             if(i<graphicsContext.scene.cameras.length) {
981                 while(
982                         (i<graphicsContext.scene.cameras.length)&&
983                         (graphicsContext.scene.cameras[i].followedObject===graphicsContext.scene.activeCamera.followedCamera.followedObject)) {
984                     i++;
985                 }
986                 // if we found such a camera, start following it
987                 if(i<graphicsContext.scene.cameras.length) {
988                     graphicsContext.scene.activeCamera.followCamera(graphicsContext.scene.cameras[i],4000);
989                 // if we didn't find such a camera, go back to free camera mode
990                 } else {
991                     graphicsContext.scene.activeCamera.followedCamera=undefined;
992                 }
993             }
994         // if we are currently not following any cameras, just start following the first one
995         } else {
996             if (graphicsContext.scene.cameras.length>0) {
997                 graphicsContext.scene.activeCamera.followCamera(graphicsContext.scene.cameras[0],4000);
998             }
999         }
1000         // cancel spacecraft control
1001         if (manualController!==undefined) {
1002             manualController.setControlledEntity(null);
1003         }
1004     }));
1005     globalCommands.push(controlContext.setOneShotActionForCommand("followPrevious",function(){
1006         // if we are currently following a camera, we have to look for the last preceding
1007         // camera that follows a different object than the current
1008         if (graphicsContext.scene.activeCamera.followedCamera!==undefined) {
1009             // first find the index of the first camera following the current object, iterating
1010             // through the cameras from the beginning
1011             i=0;
1012             while ((i<graphicsContext.scene.cameras.length)&&
1013                     (graphicsContext.scene.cameras[i].followedObject!==graphicsContext.scene.activeCamera.followedCamera.followedObject)) {
1014                 i++;
1015             }
1016             // then find the first camera backwards which has a different followed object
1017             if(i<graphicsContext.scene.cameras.length) {
1018                 while(
1019                         (i>=0)&&
1020                         (graphicsContext.scene.cameras[i].followedObject===graphicsContext.scene.activeCamera.followedCamera.followedObject)) {
1021                     i--;
1022                 }
1023                 // and then go back more until we find the last camera still following the previous object
1024                 j=i-1;
1025                 while(
1026                         (j>=0)&&
1027                         (graphicsContext.scene.cameras[j].followedObject===graphicsContext.scene.cameras[i].followedObject)) {
1028                     j--;
1029                 }
1030                 // if we found such a camera (i), start following the last good one (j+1)
1031                 if(i>=0) {
1032                     graphicsContext.scene.activeCamera.followCamera(graphicsContext.scene.cameras[j+1],4000);
1033                 // if we didn't find such a camera, go back to free camera mode
1034                 } else {
1035                     graphicsContext.scene.activeCamera.followedCamera=undefined;
1036                 }
1037             }
1038         // if we are currently not following any cameras, just start following the first one
1039         // wich follows the same object as the last one
1040         } else {
1041             if (graphicsContext.scene.cameras.length>0) {
1042                 i=graphicsContext.scene.cameras.length-1;
1043                 j=i-1;
1044                 while(
1045                         (j>=0)&&
1046                         (graphicsContext.scene.cameras[j].followedObject===graphicsContext.scene.cameras[i].followedObject)) {
1047                     j--;
1048                 }
1049                 graphicsContext.scene.activeCamera.followCamera(graphicsContext.scene.cameras[j+1],4000);
1050             }
1051         }
1052         // cancel spacecraft control
1053         if (manualController!==undefined) {
1054             manualController.setControlledEntity(null);
1055         }
1056     }));
1057     // assuming manual control of a spacecraft
1058     globalCommands.push(controlContext.setOneShotActionForCommand("setManualControl",function(){
1059         // we only assume control if a spacecraft is being followed by a camera
1060         var followedSpacecraft = graphicsContext.scene.activeCamera.getFollowedSpacecraft(logicContext);
1061         if (followedSpacecraft!==null) {
1062                 // if the controller does not exist yet, create it
1063                 if (manualController===undefined) {
1064                     manualController = new FighterController(followedSpacecraft,graphicsContext,logicContext,controlContext);                    
1065                 // if it exist, reassign it to the new spacecraft
1066                 } else {
1067                     manualController.setControlledEntity(followedSpacecraft);
1068                     manualController.reset();
1069                 }
