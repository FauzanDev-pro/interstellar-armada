 44 PhysicalObject.prototype.calculateBodySize = function() {
 45 	this.bodySize=0;
 46 	for(var i=0;i<this.bodies.length;i++) {
 47 		var bodyPos = getPositionVector(this.bodies[i].position);
 48 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[this.bodies[i].halfWidth,this.bodies[i].halfHeight,this.bodies[i].halfDepth])));
 49 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[this.bodies[i].halfWidth,this.bodies[i].halfHeight,-this.bodies[i].halfDepth])));
 50 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[this.bodies[i].halfWidth,-this.bodies[i].halfHeight,this.bodies[i].halfDepth])));
 51 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[this.bodies[i].halfWidth,-this.bodies[i].halfHeight,-this.bodies[i].halfDepth])));
 52 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[-this.bodies[i].halfWidth,this.bodies[i].halfHeight,this.bodies[i].halfDepth])));
 53 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[-this.bodies[i].halfWidth,this.bodies[i].halfHeight,-this.bodies[i].halfDepth])));
 54 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[-this.bodies[i].halfWidth,-this.bodies[i].halfHeight,this.bodies[i].halfDepth])));
 55 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[-this.bodies[i].halfWidth,-this.bodies[i].halfHeight,-this.bodies[i].halfDepth])));
 56 	}
 57 	this.bodySize*=1/this.size;

 60 PhysicalObject.prototype.checkHit = function(position,direction,range) {
 61 	if ((Math.abs(position[0]-this.position[12])<this.bodySize)&&
 62 		(Math.abs(position[1]-this.position[13])<this.bodySize)&&
 63 		(Math.abs(position[2]-this.position[14])<this.bodySize)) {
 64 			
 65 		var relativePos = vector4Matrix4Product(position,this.modelMatrixInverse);
 66 		var result=false;
 67 		for(var i=0;(result===false)&&(i<this.bodies.length);i++) {
 68 			var posRelativeToBody = vector4Matrix4Product(relativePos,this.bodies[i].modelMatrixInverse);
 69 			result = 
 70 				(posRelativeToBody[0]>=-this.bodies[i].halfWidth)&&(posRelativeToBody[0]<=this.bodies[i].halfWidth)&&
 71 				(posRelativeToBody[1]>=-this.bodies[i].halfHeight)&&(posRelativeToBody[1]<=this.bodies[i].halfHeight)&&
 72 				(posRelativeToBody[2]>=-this.bodies[i].halfDepth)&&(posRelativeToBody[2]<=this.bodies[i].halfDepth);
 73 		}
 74 	}
 75 	return result;
 76 };

104 PhysicalObject.prototype.simulate = function(dt) {
105 	// calculate acceleration from forces
106 	var acceleration=identityMatrix4();
107 	for(var i=0;i<this.forces.length;i++) {
108 		var factor = this.forces[i].strength*Math.min(dt,this.forces[i].duration)/this.mass;
109 		acceleration = mul(
110 			acceleration,
111 			translationMatrix(
112 				factor*this.forces[i].direction[0],
113 				factor*this.forces[i].direction[1],
114 				factor*this.forces[i].direction[2])
115 			);
116 		this.forces[i].duration=Math.max(this.forces[i].duration-dt,0.0);
117 	}
118 	// remove past forces
119 	while((this.forces.length>0)&&(this.forces[0].duration<=0)) {
120 		this.forces.shift();
121 	}
122 	
123 	// calculate angular acceleration from torques
124 	var angularAcc=identityMatrix4();
125 	for(var i=0;i<this.torques.length;i++) {
126 		var factor = this.torques[i].strength*Math.min(dt,this.torques[i].duration)/this.mass;
127 		angularAcc = mul(
128 			angularAcc,
129 			rotationMatrix4(this.torques[i].axis,factor));
130 		this.torques[i].duration=Math.max(this.torques[i].duration-dt,0.0);
131 	}
132 	// remove past torques
133 	while((this.torques.length>0)&&(this.torques[0].duration<=0)) {
134 		this.torques.shift();
135 	}
136 	this.velocity=mul(this.velocity,acceleration);
137 	this.angularVelocity=mul(this.angularVelocity,angularAcc);
138 	
139 	this.position=mul(this.position,this.velocity);
140 	this.orientation=mul(this.orientation,this.angularVelocity);
141 	this.modelMatrixInverse=inverse4(mul(this.orientation,this.position));
142 	
143 	this.timeSinceLastMatrixCorrection+=dt;
144 	if(this.timeSinceLastMatrixCorrection>=10000) {
145 		this.correctMatrices();
146 	}
147 };
