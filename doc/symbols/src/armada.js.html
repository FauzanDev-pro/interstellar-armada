 27 /** 
 28  * Function to load additional JavaScript code from a list of source files.
 29  * Loads the scripts in the specified order, then executes a callback function.
 30  * @param {String[]} urls The list of JavaScript sourse files to load
 31  * @param {function} callback The function to call after loading the scripts
 32  */
 33 function loadScripts(urls,callback) {
 34     // We add a new script tag inside the head of the document
 35     var head = document.getElementsByTagName('head')[0];
 36     var script = document.createElement('script');
 37     script.type = 'text/javascript';
 38     script.src = urls.shift()+"?123";
 39 
 40     // Then bind the event to the callback function.
 41     // There are several events for cross browser compatibility.
 42     if(urls.length>0) {
 43 		script.onload = function() {loadScripts(urls,callback);};
 44 		script.onreadystatechange = function() {loadScripts(urls,callback);};
 45 	} else {
 46 		script.onload = callback;
 47 		script.onreadystatechange = callback;
 48 	}
 49 
 50     // Fire the loading
 51     head.appendChild(script);
 52 }

 82 function resizeCanvas() {
 83    // only change the size of the canvas if the size it's being displayed
 84    // has changed.
 85    var canvas = document.getElementById("canvas");
 86    var width = canvas.clientWidth;
 87    var height = canvas.clientHeight;
 88    if (canvas.width !== width ||
 89        canvas.height !== height) {
 90      // Change the size of the canvas to match the size it's being displayed
 91      canvas.width = width;
 92      canvas.height = height;
 93    }
 94    if (mainScene!==undefined) {
 95        mainScene.resizeViewport(width,height);
 96    } 
 97 }
