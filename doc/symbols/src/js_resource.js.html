 26 /**
 27  * Defines a resource object.
 28  * @class Ancestor class for all classes representing resources that need to be 
 29  * prepared (e.g. loaded from external source) before they can be used. As the
 30  * loading happends asynchronously in many cases, this class provides a safe way
 31  * to interact with the objects at any time, queuing the actions if the resource
 32  * is not ready yet to use.
 33  * @returns {Resource}
 34  */
 35 function Resource() {
 36     /**
 37      * Whether the resource is ready to be used (its members can be accessed) or
 38      * not
 39      * @name Resource#_readyToUse
 40      * @type Boolean
 41      * @default false
 42      */
 43     this._readyToUse = false;
 44     
 45     /**
 46      * The queue of actions that will be performed when the resource will next
 47      * be set ready
 48      * @name Resource#_onReadyQueue
 49      * @type Function[]
 50      * @default []
 51      */
 52     this._onReadyQueue = new Array();
 53 }
 54 
 55 /**
 56  * Adds the given function to the queue to be executed ones the resource gets
 57  * ready.
 58  * @param {Function} onReadyFunction
 59  */
 60 Resource.prototype.addOnReadyFunction = function(onReadyFunction) {
 61     this._onReadyQueue.push(onReadyFunction);
 62 };
 63 
 64 /**
 65  * Returns if the resource is ready to be used at the moment. (i.e. properties
 66  * are initialized)
 67  * @returns {Boolean}
 68  */
 69 Resource.prototype.isReadyToUse = function() {
 70     return this._readyToUse;
 71 };
 72 
 73 /**
 74  * Resets the state of the resource to be not ready, resetting the queued 
 75  * actions as well.
 76  */
 77 Resource.prototype.reset = function() {
 78     this._readyToUse = false;
 79     this._onReadyQueue = new Array();
 80 };
 81 
 82 /**
 83  * Sets the ready state of the resource and executes the queued actions that
 84  * were requested in advance. Also erases the queue.
 85  */
 86 Resource.prototype.setToReady = function() {
 87     this._readyToUse = true;
 88     for(var i=0;i<this._onReadyQueue.length;i++) {
 89         this._onReadyQueue[i]();
 90     }
 91     this._onReadyQueue=new Array();
 92 };
 93 
 94 /**
 95  * Executes the first given function if the resourse is ready, otherwise queues
 96  * it to be executed when it gets ready. Optionally takes a second function to
 97  * be executed right now in case the resource is not ready yet to execute the
 98  * first one (such as notifying the user).
 99  * @param {Function} functionToExecute The function to execute when the resource
100  * is ready (now or later).
101  * @param {Function} [functionToExecuteIfNotReady] The function to be executed if
102  * the resource is not ready yet.
103  * @returns {Boolean} True if the first function got executed, false if it got
104  * queued.
105  */
106 Resource.prototype.executeWhenReady = function(functionToExecute,functionToExecuteIfNotReady) {
107     if(this._readyToUse) {
108         functionToExecute();
109         return true;
110     } else {
111         this.addOnReadyFunction(functionToExecute);
112         if (functionToExecuteIfNotReady) {
113             functionToExecuteIfNotReady();
114         }
115         return false;
116     }
117 };
