 26 // event.type must be keypress
 27 function getChar(event) {
 28   if (event.which === null) {
 29     return String.fromCharCode(event.keyCode); // IE
 30   } else if (event.which!==0 && event.charCode!==0) {
 31     return String.fromCharCode(event.which);   // the rest
 32   } else
 33   return null; // special key
 34 }

 49 function FighterController(controlledEntity,graphicsContext,logicContext) {
 50 	Controller.call(this,controlledEntity,graphicsContext,logicContext);
 51

 53 	this.FM_COMPENSATED = 1;

 57 	
 58 	this.flightMode = this.FM_INERTIAL;
 59 	this.intendedSpeed = 0;
 60 	
 61 	this.TURNING_LIMIT = 0.1;

 67 FighterController.prototype.control = function() {
 68 	if (currentlyPressedKeys[70]) { // f
 69 		this.controlledEntity.fire(this.graphicsContext.resourceCenter,this.graphicsContext.scene,this.logicContext.level.projectiles);
 70 	}
 71 	
 72 	var physicalModel = this.controlledEntity.physicalModel;
 73 	
 74 	var relativeVelocityMatrix = mul(
 75 		physicalModel.velocityMatrix,
 76 		matrix4from3(matrix3from4(physicalModel.modelMatrixInverse)));
 77 	
 78 	var turningMatrix = mul(
 79 		mul(
 80 			physicalModel.orientationMatrix,
 81 			physicalModel.angularVelocityMatrix
 82 			),
 83 		matrix4from3(matrix3from4(physicalModel.modelMatrixInverse)));
 84 	
 85 	for(var i=0;i<keyPressEvents.length;i++) {
 86 		event = keyPressEvents[i];
 87 		if (getChar(event) === 'o') {
 88 			this.flightMode=(this.flightMode+1)%this.NUM_FLIGHTMODES;
 89 		}
 90 	}
 91 	this.controlledEntity.resetThrusterBurn();
 92 	if (currentlyPressedKeys[87]) { // W
 93 		switch(this.flightMode) {
 94 			case this.FM_INERTIAL:
 95 				this.controlledEntity.addThrusterBurn("forward",0.5);
 96 				break;
 97 			case this.FM_COMPENSATED:
 98 				this.intendedSpeed+=this.controlledEntity.propulsion.class.thrust/physicalModel.mass;
 99 				break;
100 		}
101 	}
102 	if (currentlyPressedKeys[83]) { // S
103 		switch(this.flightMode) {
104 			case this.FM_INERTIAL:
105 				this.controlledEntity.addThrusterBurn("reverse",0.5);
106 				break;
107 			case this.FM_COMPENSATED:
108 				this.intendedSpeed-=this.controlledEntity.propulsion.class.thrust/physicalModel.mass;
109 				if(this.intendedSpeed<0) {
110 					this.intendedSpeed=0;
111 				}
112 				break;
113 		}
114 	}
115 	if (currentlyPressedKeys[8]) { // backspace
116 		switch(this.flightMode) {
117 			case this.FM_COMPENSATED:
118 				this.intendedSpeed=0;
119 				break;
120 		}
121 	}
122 	if(this.flightMode===this.FM_COMPENSATED) {
123 		if(relativeVelocityMatrix[12]<-0.0001) {
124 			this.controlledEntity.addThrusterBurnCapped("slideRight",0.5,this.controlledEntity.getNeededBurnForAcc(-relativeVelocityMatrix[12]));
125 		} else if(relativeVelocityMatrix[12]>0.0001) {
126 			this.controlledEntity.addThrusterBurnCapped("slideLeft",0.5,this.controlledEntity.getNeededBurnForAcc(relativeVelocityMatrix[12]));
127 		}
128 		if(relativeVelocityMatrix[14]<-0.0001) {
129 			this.controlledEntity.addThrusterBurnCapped("raise",0.5,this.controlledEntity.getNeededBurnForAcc(-relativeVelocityMatrix[14]));
130 		} else if(relativeVelocityMatrix[14]>0.0001) {
131 			this.controlledEntity.addThrusterBurnCapped("lower",0.5,this.controlledEntity.getNeededBurnForAcc(relativeVelocityMatrix[14]));
132 		}
133 		if(relativeVelocityMatrix[13]<this.intendedSpeed-0.0001) {
134 			this.controlledEntity.addThrusterBurnCapped("forward",0.5,this.controlledEntity.getNeededBurnForAcc(this.intendedSpeed-relativeVelocityMatrix[13]));
135 		} else if(relativeVelocityMatrix[13]>this.intendedSpeed+0.0001) {
136 			this.controlledEntity.addThrusterBurnCapped("reverse",0.5,this.controlledEntity.getNeededBurnForAcc(relativeVelocityMatrix[13]-this.intendedSpeed));
137 		}
138 	}
139 	if (currentlyPressedKeys[37]) { // left
140 		if(turningMatrix[4]>-this.TURNING_LIMIT) {
141 			this.controlledEntity.addThrusterBurn("yawLeft",0.5);
142 		}
143 	} else
144 	if (currentlyPressedKeys[39]) { // right
145 		if(turningMatrix[4]<this.TURNING_LIMIT) {
146 			this.controlledEntity.addThrusterBurn("yawRight",0.5);
147 		}
148 	} else if(turningMatrix[4]<-0.0001) {
149 		var burn =
150 			Math.min(
151 				this.controlledEntity.propulsion.class.angularThrust,
152 				angleDifferenceOfUnitVectors2D(
153 					[0,1],
154 					normalizeVector2D([turningMatrix[4],turningMatrix[5]])
155 					)*physicalModel.mass
156 				);
157 		this.controlledEntity.addThrusterBurn("yawRight",0.5*burn/this.controlledEntity.propulsion.class.angularThrust);
158 	} else if(turningMatrix[4]>0.0001) {
159 		var burn =
160 			Math.min(
161 				this.controlledEntity.propulsion.class.angularThrust,
162 				angleDifferenceOfUnitVectors2D(
163 					[0,1],
164 					normalizeVector2D([turningMatrix[4],turningMatrix[5]])
165 					)*physicalModel.mass
166 				);
167 		this.controlledEntity.addThrusterBurn("yawLeft",0.5*burn/this.controlledEntity.propulsion.class.angularThrust);
168 	} 
169 	if (currentlyPressedKeys[38]) { // up
170 		if(turningMatrix[6]>-this.TURNING_LIMIT) {
171 			this.controlledEntity.addThrusterBurn("pitchDown",0.5);
172 		}
173 	} else
174 	if (currentlyPressedKeys[40]) { // down
175 		if(turningMatrix[6]<this.TURNING_LIMIT) {
176 			this.controlledEntity.addThrusterBurn("pitchUp",0.5);
177 		}
178 	} else if(turningMatrix[6]<-0.0001) {
179 		var burn =
180 			Math.min(
181 				this.controlledEntity.propulsion.class.angularThrust,
182 				angleDifferenceOfUnitVectors2D(
183 					[1,0],
184 					normalizeVector2D([turningMatrix[5],turningMatrix[6]])
185 					)*physicalModel.mass
186 				);
187 		this.controlledEntity.addThrusterBurn("pitchUp",0.5*burn/this.controlledEntity.propulsion.class.angularThrust);
188 	} else if(turningMatrix[6]>0.0001) {
189 		var burn =
190 			Math.min(
191 				this.controlledEntity.propulsion.class.angularThrust,
192 				angleDifferenceOfUnitVectors2D(
193 					[1,0],
194 					normalizeVector2D([turningMatrix[5],turningMatrix[6]])
195 					)*physicalModel.mass
196 				);
197 		this.controlledEntity.addThrusterBurn("pitchDown",0.5*burn/this.controlledEntity.propulsion.class.angularThrust);
198 	}
199 	if (currentlyPressedKeys[34]) { // page down
200 		if(turningMatrix[2]>-this.TURNING_LIMIT) {
201 			this.controlledEntity.addThrusterBurn("rollRight",0.5);
202 		}
203 	} else
204 	if (currentlyPressedKeys[33]) { // page up
205 		if(turningMatrix[2]<this.TURNING_LIMIT) {
206 			this.controlledEntity.addThrusterBurn("rollLeft",0.5);
207 		}
208 	} else if(turningMatrix[2]<-0.0001) {
209 		var burn =
210 			Math.min(
211 				this.controlledEntity.propulsion.class.angularThrust,
212 				angleDifferenceOfUnitVectors2D(
213 					[1,0],
214 					normalizeVector2D([turningMatrix[0],turningMatrix[2]])
215 					)*physicalModel.mass
216 				);
217 		this.controlledEntity.addThrusterBurn("rollLeft",0.5*burn/this.controlledEntity.propulsion.class.angularThrust);
218 	} else if(turningMatrix[2]>0.0001) {
219 		var burn =
220 			Math.min(
221 				this.controlledEntity.propulsion.class.angularThrust,
222 				angleDifferenceOfUnitVectors2D(
223 					[1,0],
224 					normalizeVector2D([turningMatrix[0],turningMatrix[2]])
225 					)*physicalModel.mass
226 				);
227 		this.controlledEntity.addThrusterBurn("rollRight",0.5*burn/this.controlledEntity.propulsion.class.angularThrust);
228 	}
229 };

235 function AIController(controlledEntity,graphicsContext,logicContext) {
236 	Controller.call(this,controlledEntity,graphicsContext,logicContext);
237 	this.goals=new Array();

239 	this.TURNING_LIMIT = 0.1;

245 AIController.prototype.control = function() {
246 	var physicalModel = this.controlledEntity.physicalModel;
247 	
248 	var speed2=translationDistance2(physicalModel.velocityMatrix,nullMatrix4());
249 	var speed=Math.sqrt(speed2);
250 	var acc=this.controlledEntity.propulsion.class.thrust/physicalModel.mass;
251 	var turnAcc=this.controlledEntity.propulsion.class.angularThrust/physicalModel.mass;
252 	
253 	var directionVector = normalizeVector([physicalModel.orientationMatrix[4],physicalModel.orientationMatrix[5],physicalModel.orientationMatrix[6]]);
254 	var velocityVector = speed>0.0?normalizeVector([physicalModel.velocityMatrix[12],physicalModel.velocityMatrix[13],physicalModel.velocityMatrix[14]]):[0,0,0];
255 	
256 	var turningMatrix = mul(
257 		mul(
258 			physicalModel.orientationMatrix,
259 			physicalModel.angularVelocityMatrix
260 			),
261 		matrix4from3(matrix3from4(physicalModel.modelMatrixInverse)));
262 	
263 	this.controlledEntity.resetThrusterBurn();
264 	
265 	// only proceed of the craft has a goal to reach
266 	if(this.goals.length>0) {
267 		
268 		var distance2=translationDistance2(this.goals[0].positionMatrix,physicalModel.positionMatrix);
269 		var distance=Math.sqrt(distance2);
270 		var toGoal = normalizeVector([
271 			this.goals[0].positionMatrix[12]-physicalModel.positionMatrix[12],
272 			this.goals[0].positionMatrix[13]-physicalModel.positionMatrix[13],
273 			this.goals[0].positionMatrix[14]-physicalModel.positionMatrix[14]
274 			]);
275 		var speedTowardsGoal = vectorDotProduct(velocityVector,toGoal)*speed;
276 		var speedTowardsGoal2 = speedTowardsGoal*speedTowardsGoal;
277 		
278 		var angleToDesiredDirection = angleDifferenceOfUnitVectors(directionVector,toGoal);	
279 		
280 		var relativeVectorToGoal = vector3Matrix3Product(toGoal,matrix3from4(physicalModel.modelMatrixInverse));
281 		
282 		// if the craft is already at the target location, remove the goal
283 		if(distance<0.5) {
284 			this.goals.shift();
285 		// if not, apply the necessary maneuvers
286 		} else {	
287 			if (speedTowardsGoal<0) {
288 				this.controlledEntity.addDirectionalThrusterBurn(velocityVector,-0.5);
289 			} else if (speed*0.999>speedTowardsGoal) {
290 				var burn = -Math.min(this.controlledEntity.propulsion.class.thrust*0.25,(speed-speedTowardsGoal)/physicalModel.mass);
291 				this.controlledEntity.addDirectionalThrusterBurn(velocityVector,0.5*burn/this.controlledEntity.propulsion.class.thrust);
292 				if ((2*distance*acc>speedTowardsGoal2)&&(angleToDesiredDirection<0.1)) {
293 					this.controlledEntity.addThrusterBurn("forward",0.375);
294 				} else {
295 					burn = -Math.min(this.controlledEntity.propulsion.class.thrust*0.75,(speed-speedTowardsGoal)/physicalModel.mass);
296 					this.controlledEntity.addDirectionalThrusterBurn(velocityVector,0.5*burn/this.controlledEntity.propulsion.class.thrust);
297 				}
298 			} else {
299 				if (speed2>2*distance*acc) {
300 					this.controlledEntity.addDirectionalThrusterBurn(velocityVector,-0.5);
301 				} else{
302 					this.controlledEntity.addThrusterBurn("forward",0.5);
303 				}
304 			}
305 			
306 			var relativeToGoalXY = normalizeVector2D([relativeVectorToGoal[0],relativeVectorToGoal[1]]);
307 			var yawAngleDifference = angleDifferenceOfUnitVectors2D([0,1],relativeToGoalXY);
308 			var turningVectorXY = normalizeVector2D([turningMatrix[4],turningMatrix[5]]);
309 			var yawAngularVelocity = angleDifferenceOfUnitVectors2D([0,1],turningVectorXY);	
310 			
311 			if ((yawAngleDifference>0.01)||(Math.abs(turningMatrix[4])>0.0001)) {
312 				if(relativeVectorToGoal[0]>0.0001) {
313 					if(
314 						(turningMatrix[4]<this.TURNING_LIMIT)&&
315 						(yawAngularVelocity*yawAngularVelocity<(2*yawAngleDifference*turnAcc))
316 						) {
317 						this.controlledEntity.addThrusterBurn("yawRight",0.5);
318 					} else if(turningMatrix[4]>0.0001) {
319 						this.controlledEntity.addThrusterBurnCapped("yawLeft",0.5,this.controlledEntity.getNeededBurnForAngularAcc(yawAngularVelocity));
320 					}
321 				} else if(relativeVectorToGoal[0]<-0.0001) {
322 					if(
323 						(turningMatrix[4]>-this.TURNING_LIMIT)&&
324 						(yawAngularVelocity*yawAngularVelocity<(2*yawAngleDifference*turnAcc))
325 						) {
326 						this.controlledEntity.addThrusterBurn("yawLeft",0.5);
327 					} else if(turningMatrix[4]<-0.0001) {
328 						this.controlledEntity.addThrusterBurnCapped("yawRight",0.5,this.controlledEntity.getNeededBurnForAngularAcc(yawAngularVelocity));
329 					}
330 				}
331 			}
332 			
333 			var relativeToGoalYZ = normalizeVector2D([relativeVectorToGoal[1],relativeVectorToGoal[2]]);
334 			var pitchAngleDifference = angleDifferenceOfUnitVectors2D([1,0],relativeToGoalYZ);
335 			var turningVectorYZ = normalizeVector2D([turningMatrix[5],turningMatrix[6]]);
336 			var pitchAngularVelocity = angleDifferenceOfUnitVectors2D([1,0],turningVectorYZ);	
337 			
338 			if ((pitchAngleDifference>0.01)||(Math.abs(turningMatrix[6])>0.0001)) {
339 				if(relativeVectorToGoal[2]>0.0001) {
340 					if(
341 						(turningMatrix[6]<this.TURNING_LIMIT)&&
342 						(pitchAngularVelocity*pitchAngularVelocity<(2*pitchAngleDifference*turnAcc))
343 						) {
344 						this.controlledEntity.addThrusterBurn("pitchUp",0.5);
345 					} else if(turningMatrix[6]>0.0001) {
346 						this.controlledEntity.addThrusterBurnCapped("pitchDown",0.5,this.controlledEntity.getNeededBurnForAngularAcc(pitchAngularVelocity));
347 					}
348 				} else if(relativeVectorToGoal[2]<-0.0001) {
349 					if(
350 						(turningMatrix[6]>-this.TURNING_LIMIT)&&
351 						(yawAngularVelocity*yawAngularVelocity<(2*yawAngleDifference*turnAcc))
352 						) {
353 						this.controlledEntity.addThrusterBurn("pitchDown",0.5);
354 					} else if(turningMatrix[6]<-0.0001) {
355 						this.controlledEntity.addThrusterBurnCapped("pitchUp",0.5,this.controlledEntity.getNeededBurnForAngularAcc(pitchAngularVelocity));
356 					}
357 				}
358 			}
359 		}
360 	} else {
361 		if (speed>0) {
362 			var burn = -Math.min(this.controlledEntity.propulsion.class.thrust,speed*physicalModel.mass);
363 			this.controlledEntity.addDirectionalThrusterBurn(velocityVector,0.5*burn/this.controlledEntity.propulsion.class.thrust);
364 		}
365 		var turningVectorXY = normalizeVector2D([turningMatrix[4],turningMatrix[5]]);
366 		var yawAngularVelocity = angleDifferenceOfUnitVectors2D([0,1],turningVectorXY);	
367 		if(turningMatrix[4]>0.0001) {
368 			this.controlledEntity.addThrusterBurnCapped("yawLeft",0.5,this.controlledEntity.getNeededBurnForAngularAcc(yawAngularVelocity));
369 		} else if(turningMatrix[4]<-0.0001) {
370 			this.controlledEntity.addThrusterBurnCapped("yawRight",0.5,this.controlledEntity.getNeededBurnForAngularAcc(yawAngularVelocity));
371 		}
372 		var turningVectorYZ = normalizeVector2D([turningMatrix[5],turningMatrix[6]]);
373 		var pitchAngularVelocity = angleDifferenceOfUnitVectors2D([1,0],turningVectorYZ);	
374 		if(turningMatrix[6]>0.0001) {
375 			this.controlledEntity.addThrusterBurnCapped("pitchDown",0.5,this.controlledEntity.getNeededBurnForAngularAcc(pitchAngularVelocity));
376 		} else if(turningMatrix[6]<-0.0001) {
377 			this.controlledEntity.addThrusterBurnCapped("pitchUp",0.5,this.controlledEntity.getNeededBurnForAngularAcc(pitchAngularVelocity));
378 		}
379 	}
380 	
381 	var turningVectorXZ = normalizeVector2D([turningMatrix[0],turningMatrix[2]]);
382 	var rollAngularVelocity = angleDifferenceOfUnitVectors2D([1,0],turningVectorXZ);
383 	
384 	if(turningMatrix[2]>0.0001) {
385 		this.controlledEntity.addThrusterBurnCapped("rollRight",0.5,this.controlledEntity.getNeededBurnForAngularAcc(rollAngularVelocity));
386 	} else if(turningMatrix[2]<-0.0001) {
387 		this.controlledEntity.addThrusterBurnCapped("rollLeft",0.5,this.controlledEntity.getNeededBurnForAngularAcc(rollAngularVelocity));
388 	}
389 };

391 function controlCamera(camera) {
392     if(camera.controllableDirection) {
393         if (currentlyPressedKeys[17]&&currentlyPressedKeys[37]) {
394                 //ctrl left
395                 if (camera.angularVelocityVector[1]<camera.maxTurn) {
396                                 camera.angularVelocityVector[1]+=camera.angularAcceleration;
397                 }
398         } else {
399                 if (camera.angularVelocityVector[1]>0) {
400                         camera.angularVelocityVector[1]-=
401                                 Math.min(camera.angularAcceleration,camera.angularVelocityVector[1]);
402                 }
403         }
404         if (currentlyPressedKeys[17]&&currentlyPressedKeys[39]) {
405                 // ctrl Right
406                 if (camera.angularVelocityVector[1]>-camera.maxTurn) {
407                         camera.angularVelocityVector[1]-=camera.angularAcceleration;
408                 }
409         } else {
410                 if (camera.angularVelocityVector[1]<0) {
411                         camera.angularVelocityVector[1]+=
412                                 Math.min(camera.angularAcceleration,-camera.angularVelocityVector[1]);
413                 }
414         }
415         if (currentlyPressedKeys[17]&&currentlyPressedKeys[38]) {
416                 //ctrl up
417                 if (camera.angularVelocityVector[0]<camera.maxTurn) {
418                                 camera.angularVelocityVector[0]+=camera.angularAcceleration;
419                 }
420         } else {
421                 if (camera.angularVelocityVector[0]>0) {
422                         camera.angularVelocityVector[0]-=
423                                 Math.min(camera.angularAcceleration,camera.angularVelocityVector[0]);
424                 }
425         }
426         if (currentlyPressedKeys[17]&&currentlyPressedKeys[40]) {
427                 // ctrl down
428                 if (camera.angularVelocityVector[0]>-camera.maxTurn) {
429                         camera.angularVelocityVector[0]-=camera.angularAcceleration;
430                 }
431         } else {
432                 if (camera.angularVelocityVector[0]<0) {
433                         camera.angularVelocityVector[0]+=
434                                 Math.min(camera.angularAcceleration,-camera.angularVelocityVector[0]);
435                 }
436         }
437     }
438     if(camera.controllablePosition) {
439         if (currentlyPressedKeys[37]) {
440                 // Left
441                 if (camera.velocityVector[0]<camera.maxSpeed) {
442                         camera.velocityVector[0]+=camera.acceleration;
443                 }
444         } else {
445                 if (camera.velocityVector[0]>0) {
446                         camera.velocityVector[0]-=
447                                 Math.min(camera.acceleration,camera.velocityVector[0]);
448                 }
449         }
450         if (currentlyPressedKeys[39]) {
451                 // Right
452                 if (camera.velocityVector[0]>-camera.maxSpeed) {
453                         camera.velocityVector[0]-=camera.acceleration;
454                 }
455         } else {
456                 if (camera.velocityVector[0]<0) {
457                         camera.velocityVector[0]+=
458                                 Math.min(camera.acceleration,-camera.velocityVector[0]);
459                 }
460         }
461         if (currentlyPressedKeys[34]) {
462                 // Page down
463                 if (camera.velocityVector[1]<camera.maxSpeed) {
464                         camera.velocityVector[1]+=camera.acceleration;
465                 }
466         } else {
467                 if (camera.velocityVector[1]>0) {
468                         camera.velocityVector[1]-=
469                                 Math.min(camera.acceleration,camera.velocityVector[1]);
470                 }
471         }
472         if (currentlyPressedKeys[33]) {
473                 // Page up
474                 if (camera.velocityVector[1]>-camera.maxSpeed) {
475                         camera.velocityVector[1]-=camera.acceleration;
476                 }
477         } else {
478                 if (camera.velocityVector[1]<0) {
479                         camera.velocityVector[1]+=
480                                 Math.min(camera.acceleration,-camera.velocityVector[1]);
481                 }
482         }
483         if (currentlyPressedKeys[38]) {
484                 // Up
485                 if (camera.velocityVector[2]<camera.maxSpeed) {
486                         camera.velocityVector[2]+=camera.acceleration;
487                 }
488         } else {
489                 if (camera.velocityVector[2]>0) {
490                         camera.velocityVector[2]-=
491                                 Math.min(camera.acceleration,camera.velocityVector[2]);
492                 }
493         }
494         if (currentlyPressedKeys[40]) {
495                 // Down
496                 if (camera.velocityVector[2]>-camera.maxSpeed) {
497                         camera.velocityVector[2]-=camera.acceleration;
498                 }
499         } else {
500                 if (camera.velocityVector[2]<0) {
501                         camera.velocityVector[2]+=
502                                 Math.min(camera.acceleration,-camera.velocityVector[2]);
503                 }
504         }
505     }
506 
507     if (camera.controllablePosition) {
508         var inverseOrientationMatrix=transposed3(inverse3(matrix3from4(camera.orientationMatrix)));
509         var translationVector = matrix3Vector3Product(
510             camera.velocityVector,
511             inverseOrientationMatrix
512             );
513         if(camera.followedObject===undefined) {
514             camera.positionMatrix=
515                     mul(
516                             camera.positionMatrix,
517                             translationMatrixv(translationVector)
518                             );
519         } else {
520             camera.followPositionMatrix=
521                     mul(
522                             camera.followPositionMatrix,
523                             translationMatrixv(translationVector)
524                             );
525         }
526     }
527     if (camera.controllableDirection) {
528         var rotationMatrix=
529                 mul(
530                     rotationMatrix4(
531                                 [0,0,1],
532                                 camera.angularVelocityVector[1]
533                                 ),
534                     rotationMatrix4(
535                                 [1,0,0],
536                                 camera.angularVelocityVector[0]
537                                 )    
538                     );
539         if(camera.followedObject===undefined) {
540             camera.orientationMatrix=mul(camera.orientationMatrix,rotationMatrix);
541         } else {
542             camera.followOrientationMatrix=mul(camera.followOrientationMatrix,rotationMatrix);
543         }
544     }
545             
546     if (currentlyPressedKeys[90]) { // z
547             camera.setFOV(camera.fov-1);
548     }
549     if (currentlyPressedKeys[85]) { // u
550             camera.setFOV(camera.fov+1);
551     }
552     
553     if (camera.followedObject!==undefined) {
554         // look in direction y instead of z:
555         var newOrientationMatrix =
556                 mul(
557                         mul(
558                                 inverseRotationMatrix(camera.followedObject.getOrientationMatrix()),
559                                 camera.followOrientationMatrix
560                                 ),
561                         rotationMatrix4([1,0,0],3.1415/2)        
562                         );
563         camera.orientationMatrix=newOrientationMatrix;
564         var camPositionMatrix = 
565                 mul(
566                         mul(
567                                 camera.rotationCenterIsObject?
568                                     translationMatrixv(getPositionVector(mul(
569                                         camera.followPositionMatrix,
570                                             inverseRotationMatrix(camera.followOrientationMatrix)
571                                         )))
572                                     :
573                                     camera.followPositionMatrix,
574                                 camera.followedObject.getOrientationMatrix()
575                                 ),
576                         camera.followedObject.getPositionMatrix()
577                         );
578         var newPositionMatrix=
579                 translationMatrix(
580                         -camPositionMatrix[12],
581                         -camPositionMatrix[13],
582                         -camPositionMatrix[14]
583                         );
584         var velocityMatrix = mul(translationMatrix(
585                 newPositionMatrix[12]-camera.positionMatrix[12],
586                 newPositionMatrix[13]-camera.positionMatrix[13],
587                 newPositionMatrix[14]-camera.positionMatrix[14]),camera.orientationMatrix);
588         camera.velocityVector = [velocityMatrix[12],velocityMatrix[13],velocityMatrix[14]];
589         camera.positionMatrix=newPositionMatrix;
590     }
591 }

593 function control(resourceCenter,scene,level) {
594 	if (scene.activeCamera.followedCamera===undefined) {
595             controlCamera(scene.activeCamera);
596 	} else {
597             controlCamera(scene.activeCamera.followedCamera);
598             scene.activeCamera.velocityVector=scene.activeCamera.followedCamera.velocityVector;
599             scene.activeCamera.angularVelocityVector=scene.activeCamera.followedCamera.angularVelocityVector;
600         }
601         // pause game
602 	if (currentlyPressedKeys[80]) { // p
603 		alert("paused");
604 	}
605         
606 	for(var i=0;i<keyPressEvents.length;i++) {
607 		event = keyPressEvents[i];
608                 // follow next camera
609 		if (getChar(event) === 'c') {	
610 			activeCameraIndex=(activeCameraIndex+1)%scene.cameras.length;
611 			scene.activeCamera.followCamera(scene.cameras[activeCameraIndex]);
612 		}
613                 // follow previous camera
614 		if (getChar(event) === 'x') {
615 			activeCameraIndex=(activeCameraIndex-1)%scene.cameras.length;
616                         scene.activeCamera.followCamera(scene.cameras[activeCameraIndex]);
617 		}
618                 // set control to manual
619                 if (getChar(event) === 'm') {
620                     if(activeCameraIndex>0) {
621                         level.spacecrafts[activeCameraIndex-1].controller=new FighterController(level.spacecrafts[activeCameraIndex-1],new GraphicsContext(resourceCenter,scene),new LogicContext(level));
622                     }
623                 }
624                 // set control to AI
625                 if (getChar(event) === 'n') {
626                     if(activeCameraIndex>0) {
627                         level.spacecrafts[activeCameraIndex-1].controller=new AIController(level.spacecrafts[activeCameraIndex-1],new GraphicsContext(resourceCenter,scene),new LogicContext(level));
628                         for(var j=0;j<10;j++) {
629                             level.spacecrafts[activeCameraIndex-1].controller.goals.push(new Goal(translationMatrix(Math.random()*mapSize-mapSize/2,Math.random()*mapSize-mapSize/2,Math.random()*mapSize-mapSize/2)));
630                         }
631                     }
632                 }
633                 // stop all units
634 		if (getChar(event) === '0') {
635 			for(var j=0;j<level.spacecrafts.length;j++) {
636                                 if(level.spacecrafts[j].controller instanceof AIController) {
637                                     level.spacecrafts[j].controller.goals=new Array();
638                                 }
639 			}
640 		}
641                 // toggle rotation of directional lightsource
642 		if (getChar(event) === 'l') {
643 			lightTurn=!lightTurn;
644 		}
645                 // toggle visibility of hitboxes
646 		if (getChar(event) === 'h') {
647 			for(var j=0;j<level.spacecrafts.length;j++) {
648 				for(var k=0;k<level.spacecrafts[j].visualModel.subnodes.length;k++) {
649 					if(level.spacecrafts[j].visualModel.subnodes[k].texture.filename==="textures/white.png") {
650 						level.spacecrafts[j].visualModel.subnodes[k].visible=!level.spacecrafts[j].visualModel.subnodes[k].visible;
651 					}
652 				}
653 			}
654 		}
655                 keyPressEvents.splice(i,1);
656                 i-=1;
657 	}
658         scene.activeCamera.update();
659 	scene.activeCamera.matrix =
660 		mul(
661 			scene.activeCamera.positionMatrix,
662 			scene.activeCamera.orientationMatrix
663 		);
664 }
