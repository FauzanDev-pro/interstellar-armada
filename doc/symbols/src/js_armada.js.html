 38  * - game should be the only global variable
 39  * - they don't belong to the game object (the game object is initialized after
 40  * all source files are loaded)
 41  * - they might contain calculations/functions later, which is more appropriate
 42  * to put inside a function then a global variable definition
 43  * - the declarations are moved above all definitions, and this way the 
 44  * operations to get the names happen when the function is called, not when the
 45  * definition is executed (as is the case with variables)

118 /** 
119  * Function to load additional JavaScript code from a list of source files.
120  * Loads the scripts in the specified order, then executes a callback function.
121  * @param {String} folder The folder where the scripts reside
122  * @param {String[]} filenames The list of JavaScript sourse files to load
123  * @param {Function} callback The function to call after loading the scripts
124  * @param {Boolean} bypassCaching If true, the script files will be forcefully
125  * downloaded again, even if they are in the cache already.
126  */
127 function loadScripts(folder,filenames,bypassCaching,callback) {
128     // NOTE: recursive function - if the arguments are changed, change the calls
129     // as well
130     // We add a new script tag inside the head of the document
131     var head = document.getElementsByTagName('head')[0];
132     var script = document.createElement('script');
133     script.type = 'text/javascript';
134     script.src = folder + (bypassCaching?
135                     filenames.shift()+"?123":
136                     filenames.shift());        
137 
138     // Then bind the event to the callback function.
139     // There are several events for cross browser compatibility.
140     if(filenames.length>0) {
141 		script.onload = function() {loadScripts(folder,filenames,bypassCaching,callback);};
142 		script.onreadystatechange = function() {loadScripts(folder,filenames,bypassCaching,callback);};
143 	} else if (callback!==undefined) {
144 		script.onload = callback;
145 		script.onreadystatechange = callback;
146 	}
147 
148     // Fire the loading
149     head.appendChild(script);
150 }

152 /**
153  * Defines a Game object.
154  * @class Holds the general properties of the game (the current context for the
155  * different modules)
156  * @returns {Game}
157  */
158 function Game() {
159     /**
160      * The game's available screens stored in an associative array, with the 
161      * keys being the names of the screens.
162      * @name Game#_screens
163      * @type Object
164      * @default {}
165      */
166     this._screens = new Object();
167     /**
168      * A reference to the currently active (displayed) screen of the game.
169      * @name Game#_currentScreen
170      * @type GameScreen
171      * @default null
172      */
173     this._currentScreen = null;
174     
175     /**
176      * The graphics context of the game, that can be used to access and 
177      * manipulate graphical resources.
178      * @name Game#graphicsContext
179      * @type GraphicsContext
180      */
181     this.graphicsContext = new GraphicsContext(new ResourceCenter(),null);
182     /**
183      * The control context of the game, that can be used to bind input controls
184      * to in-game actions.
185      * @name Game#controlContext
186      * @type ControlContext
187      * @default null
188      */
189     this.controlContext = null;
190     
191     this.requestSettingsLoad();
192 }

202 /**
203  * Sends an asynchronous request to get the XML file describing the game
204  * settings and sets the callback function to set them.
205  */
206 Game.prototype.requestSettingsLoad = function () {
207     var request = new XMLHttpRequest();
208     request.open('GET', getXMLFolder()+"settings.xml?123", true);
209     var self = this;
210     request.onreadystatechange = function () {
211         if (request.readyState === 4) {
212             var settingsXML = this.responseXML;
213             self.controlContext = new KeyboardControlContext();
214             self.controlContext.loadFromXML(settingsXML.getElementsByTagName("control")[0]);
215         }
216     };
217     request.send(null);
218 };

220 /**
221  * Adds a new screen to the list that can be set as current later.
222  * @param {GameScreen} screen The new game screen to be added.
223  * @param {Boolean} [isDefaultScreen=false] If true, this screen will be taken 
224  * as the default, starting screen. (and will be set, but not reloaded)
225  */
226 Game.prototype.addScreen = function(screen,isDefaultScreen) {
227     this._screens[screen.getName()]=screen;
228     if(isDefaultScreen===true) {
229         this._currentScreen=screen;
230     }
231 };

242 /**
243  * Sets the current game screen to the one with the specified name (from the
244  * list of available screens), including refreshing the HTML body.
245  * @param {String} screenName
246  */
247 Game.prototype.setCurrentScreen = function(screenName) {
248     this._currentScreen.closePage();
249     var screen = this.getScreen(screenName);
250     screen.buildPage();
251     this._currentScreen = screen;
252 };
