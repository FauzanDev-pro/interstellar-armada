  7 /** 
  8  * Function to load additional JavaScript code from a list of source files.
  9  * Loads the scripts in the specified order, then executes a callback function.
 10  * @param {String[]} urls The list of JavaScript sourse files to load
 11  * @param {function} callback The function to call after loading the scripts
 12  */
 13 function loadScripts(urls,callback) {
 14     // We add a new script tag inside the head of the document
 15     var head = document.getElementsByTagName('head')[0];
 16     var script = document.createElement('script');
 17     script.type = 'text/javascript';
 18     script.src = urls.shift()+"?123";
 19 
 20     // Then bind the event to the callback function.
 21     // There are several events for cross browser compatibility.
 22     if(urls.length>0) {
 23 		script.onload = function() {loadScripts(urls,callback);};
 24 		script.onreadystatechange = function() {loadScripts(urls,callback);};
 25 	} else {
 26 		script.onload = callback;
 27 		script.onreadystatechange = callback;
 28 	}
 29 
 30     // Fire the loading
 31     head.appendChild(script);
 32 }
