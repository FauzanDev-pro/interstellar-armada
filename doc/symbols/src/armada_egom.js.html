 59 function EgomModel(filename) {
 60 	this.vertices = new Array();
 61 	this.lines = new Array();
 62 	this.triangles = new Array();
 63 	this.size = 0;
 64 	
 65 	this.filename=filename;
 66 	if(filename!==undefined) {
 67 		this.loadFromFile(filename);
 68 	}
 69 	
 70 	this.bufferStart = 0;
 71 	this.bufferStartLines = 0;
 72 }

 86 EgomModel.prototype.loadFromFile = function(filename) {
 87 	var request = new XMLHttpRequest();
 88 	request.open('GET', filename+"?12345", false); //timestamp added to URL to bypass cache
 89 	request.send(null);
 90 	modelSource = request.responseXML;
 91 	
 92 	var nVertices = parseInt(modelSource.getElementsByTagName("vertices")[0].getAttribute("count"));
 93 	this.vertices = new Array();
 94 	vertexTags = modelSource.getElementsByTagName("vertex");
 95 	for(var i=0; i<nVertices; i++) {
 96 		var index=vertexTags[i].getAttribute("i");
 97 		if (this.vertices.length<index) {
 98 			for(var j=this.vertices.length;j<index;j++) {
 99 				this.vertices[j]=[0.0,0.0,0.0];
100 			}
101 		}
102 		this.vertices[index]=[
103 			vertexTags[i].getAttribute("x")/10000.0,
104 			vertexTags[i].getAttribute("y")/-10000.0,
105 			vertexTags[i].getAttribute("z")/-10000.0];
106 		if(Math.abs(this.vertices[index][0]*2)>this.size) {
107 			this.size=Math.abs(this.vertices[index][0]*2);
108 		}
109 		if(Math.abs(this.vertices[index][1]*2)>this.size) {
110 			this.size=Math.abs(this.vertices[index][1]*2);
111 		}
112 		if(Math.abs(this.vertices[index][2]*2)>this.size) {
113 			this.size=Math.abs(this.vertices[index][2]*2);
114 		}
115 	}
116 	
117 	var nLines = parseInt(modelSource.getElementsByTagName("lines")[0].getAttribute("count"));
118 	this.lines = new Array(nLines);
119 	lineTags = modelSource.getElementsByTagName("line");
120 	for(var i=0; i<nLines; i++) {
121 		this.lines[i]=new Line(
122 			lineTags[i].getAttribute("a"),
123 			lineTags[i].getAttribute("b"),
124 			lineTags[i].getAttribute("red")/255.0,
125 			lineTags[i].getAttribute("green")/255.0,
126 			lineTags[i].getAttribute("blue")/255.0,
127 			lineTags[i].getAttribute("luminosity")/255.0,
128 			lineTags[i].getAttribute("nx"),
129 			-lineTags[i].getAttribute("ny"),
130 			-lineTags[i].getAttribute("nz")
131 			);
132 	}
133 	
134 	var nTriangles = parseInt(modelSource.getElementsByTagName("triangles")[0].getAttribute("count"));
135 	this.triangles = new Array(nTriangles);
136 	triangleTags = modelSource.getElementsByTagName("triangle");
137 	for(var i=0; i<nTriangles; i++) {
138 		this.triangles[i]=new Triangle(
139 			triangleTags[i].getAttribute("a"),
140 			triangleTags[i].getAttribute("b"),
141 			triangleTags[i].getAttribute("c"),
142 			triangleTags[i].getAttribute("red")/255.0,
143 			triangleTags[i].getAttribute("green")/255.0,
144 			triangleTags[i].getAttribute("blue")/255.0,
145 			(255-triangleTags[i].getAttribute("alpha"))/255.0,
146 			triangleTags[i].getAttribute("luminosity")/255.0,
147 			triangleTags[i].getAttribute("shininess"),
148 			triangleTags[i].getAttribute("tax"),
149 			triangleTags[i].getAttribute("tay"),
150 			triangleTags[i].getAttribute("tbx"),
151 			triangleTags[i].getAttribute("tby"),
152 			triangleTags[i].getAttribute("tcx"),
153 			triangleTags[i].getAttribute("tcy"),
154 			triangleTags[i].getAttribute("nax"),
155 			-triangleTags[i].getAttribute("nay"),
156 			-triangleTags[i].getAttribute("naz"),
157 			triangleTags[i].getAttribute("nbx"),
158 			-triangleTags[i].getAttribute("nby"),
159 			-triangleTags[i].getAttribute("nbz"),
160 			triangleTags[i].getAttribute("ncx"),
161 			-triangleTags[i].getAttribute("ncy"),
162 			-triangleTags[i].getAttribute("ncz")
163 			);
164 	}
165 	
166 	//alert("model with "+nVertices+" vertices loaded!");
167 	
168 	//xmlDoc.getElementsByTagName("to")[0].childNodes[0].nodeValue;
169 	//getAttribute('category')
170 };

172 EgomModel.prototype.getBuffers = function(lineMode) {
173 	if(lineMode===true) {
174 		var vertexData = new Float32Array(this.lines.length*6);
175 		for(var i=0;i<this.lines.length;i++) {
176 			vertexData[i*6]=this.vertices[this.lines[i].a][0];
177 			vertexData[i*6+1]=this.vertices[this.lines[i].a][1];
178 			vertexData[i*6+2]=this.vertices[this.lines[i].a][2];
179 			vertexData[i*6+3]=this.vertices[this.lines[i].b][0];
180 			vertexData[i*6+4]=this.vertices[this.lines[i].b][1];
181 			vertexData[i*6+5]=this.vertices[this.lines[i].b][2];
182 		}
183 		var texCoordData = new Float32Array(this.lines.length*4);
184 		for(var i=0;i<this.lines.length;i++) {
185 			texCoordData[i*4]=0.0;
186 			texCoordData[i*4+1]=1.0;
187 			texCoordData[i*4+2]=1.0;
188 			texCoordData[i*4+3]=1.0;
189 		}
190 		var normalData = new Float32Array(this.lines.length*6);
191 		for(var i=0;i<this.lines.length;i++) {
192 			normalData[i*6]=this.lines[i].nx;
193 			normalData[i*6+1]=this.lines[i].ny;
194 			normalData[i*6+2]=this.lines[i].nz;
195 			normalData[i*6+3]=this.lines[i].nx;
196 			normalData[i*6+4]=this.lines[i].ny;
197 			normalData[i*6+5]=this.lines[i].nz;
198 		}
199 		var colorData = new Float32Array(this.lines.length*8);
200 		for(var i=0;i<this.lines.length;i++) {
201 			colorData[i*8]=this.lines[i].red;
202 			colorData[i*8+1]=this.lines[i].green;
203 			colorData[i*8+2]=this.lines[i].blue;
204 			colorData[i*8+3]=1.0;
205 			colorData[i*8+4]=this.lines[i].red;
206 			colorData[i*8+5]=this.lines[i].green;
207 			colorData[i*8+6]=this.lines[i].blue;
208 			colorData[i*8+7]=1.0;
209 		}
210 		var luminosityData = new Float32Array(this.lines.length*2);
211 		for(var i=0;i<this.lines.length;i++) {
212 			luminosityData[i*2]=this.lines[i].luminosity;
213 			luminosityData[i*2+1]=this.lines[i].luminosity;
214 		}
215 		var shininessData = new Float32Array(this.lines.length*2);
216 		for(var i=0;i<this.lines.length;i++) {
217 			shininessData[i*2]=0;
218 			shininessData[i*2+1]=0;
219 		}
220 	} else {
221 		var vertexData = new Float32Array(this.triangles.length*9);
222 		for(var i=0;i<this.triangles.length;i++) {
223 			vertexData[i*9]=this.vertices[this.triangles[i].a][0];
224 			vertexData[i*9+1]=this.vertices[this.triangles[i].a][1];
225 			vertexData[i*9+2]=this.vertices[this.triangles[i].a][2];
226 			vertexData[i*9+3]=this.vertices[this.triangles[i].b][0];
227 			vertexData[i*9+4]=this.vertices[this.triangles[i].b][1];
228 			vertexData[i*9+5]=this.vertices[this.triangles[i].b][2];
229 			vertexData[i*9+6]=this.vertices[this.triangles[i].c][0];
230 			vertexData[i*9+7]=this.vertices[this.triangles[i].c][1];
231 			vertexData[i*9+8]=this.vertices[this.triangles[i].c][2];
232 		}
233 		var texCoordData = new Float32Array(this.triangles.length*6);
234 		for(var i=0;i<this.triangles.length;i++) {
235 			texCoordData[i*6]=this.triangles[i].tax;
236 			texCoordData[i*6+1]=this.triangles[i].tay;
237 			texCoordData[i*6+2]=this.triangles[i].tbx;
238 			texCoordData[i*6+3]=this.triangles[i].tby;
239 			texCoordData[i*6+4]=this.triangles[i].tcx;
240 			texCoordData[i*6+5]=this.triangles[i].tcy;
241 		}
242 		var normalData = new Float32Array(this.triangles.length*9);
243 		for(var i=0;i<this.triangles.length;i++) {
244 			normalData[i*9]=this.triangles[i].nax;
245 			normalData[i*9+1]=this.triangles[i].nay;
246 			normalData[i*9+2]=this.triangles[i].naz;
247 			normalData[i*9+3]=this.triangles[i].nbx;
248 			normalData[i*9+4]=this.triangles[i].nby;
249 			normalData[i*9+5]=this.triangles[i].nbz;
250 			normalData[i*9+6]=this.triangles[i].ncx;
251 			normalData[i*9+7]=this.triangles[i].ncy;

282 	
283 	return {

293 EgomModel.prototype.setupBuffers = function(gl,program) {
294 	var bufferData=this.getBuffers();
295 	var vertexData=bufferData[0];
296 	var texCoorddata=bufferData[1];
297 	var normalData=bufferData[2];
298 	var colorData=bufferData[3];
299 	var luminosityData=bufferData[4];
300 	var shininessData=bufferData[5];
301 	
302 	// look up where the vertex data needs to go.
303 	var positionLocation = gl.getAttribLocation(program, "a_position");
304 	// Create a buffer and put a single clipspace rectangle in
305 	// it (2 triangles)
306 	var buffer = gl.createBuffer();
307 	gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
308 	gl.bufferData(
309 		gl.ARRAY_BUFFER, 
310 		vertexData,
311 		gl.STATIC_DRAW);
312 	gl.enableVertexAttribArray(positionLocation);
313 	gl.vertexAttribPointer(positionLocation, 3, gl.FLOAT, false, 0, 0);
314 	
315 	// look up where the texture coordinates need to go.
316 	var texCoordLocation = gl.getAttribLocation(program, "a_texCoord");
317 	// provide texture coordinates for the rectangle.
318 	var texCoordBuffer = gl.createBuffer();
319 	gl.bindBuffer(gl.ARRAY_BUFFER, texCoordBuffer);
320 	gl.bufferData(
321 		gl.ARRAY_BUFFER,
322 		texCoordData,
323 		gl.STATIC_DRAW);
324 	gl.enableVertexAttribArray(texCoordLocation);
325 	gl.vertexAttribPointer(texCoordLocation, 2, gl.FLOAT, false, 0, 0);
326 	
327 	// look up where the normal data needs to go.
328 	var normalLocation = gl.getAttribLocation(program, "a_normal");
329 	var normalBuffer = gl.createBuffer();
330 	gl.bindBuffer(gl.ARRAY_BUFFER, normalBuffer);
331 	gl.bufferData(
332 		gl.ARRAY_BUFFER, 
333 		normalData,
334 		gl.STATIC_DRAW);
335 	gl.enableVertexAttribArray(normalLocation);
336 	gl.vertexAttribPointer(normalLocation, 3, gl.FLOAT, false, 0, 0);
337 	
338 	var colorLocation = gl.getAttribLocation(program, "a_color");
339 	var colorBuffer = gl.createBuffer();
340 	gl.bindBuffer(gl.ARRAY_BUFFER, colorBuffer);
341 	gl.bufferData(
342 		gl.ARRAY_BUFFER, 
343 		colorData,
344 		gl.STATIC_DRAW);
345 	gl.enableVertexAttribArray(colorLocation);
346 	gl.vertexAttribPointer(colorLocation, 4, gl.FLOAT, false, 0, 0);
347 	
348 	var luminosityLocation = gl.getAttribLocation(program, "a_luminosity");
349 	var luminosityBuffer = gl.createBuffer();
350 	gl.bindBuffer(gl.ARRAY_BUFFER, luminosityBuffer);
351 	gl.bufferData(
352 		gl.ARRAY_BUFFER, 
353 		luminosityData,
354 		gl.STATIC_DRAW);
355 	gl.enableVertexAttribArray(luminosityLocation);
356 	gl.vertexAttribPointer(luminosityLocation, 1, gl.FLOAT, false, 0, 0);
357 	
358 	var shininessLocation = gl.getAttribLocation(program, "a_shininess");
359 	var shininessBuffer = gl.createBuffer();
360 	gl.bindBuffer(gl.ARRAY_BUFFER, shininessBuffer);
361 	gl.bufferData(
362 		gl.ARRAY_BUFFER, 
363 		shininessData,
364 		gl.STATIC_DRAW);
365 	gl.enableVertexAttribArray(shininessLocation);
366 	gl.vertexAttribPointer(shininessLocation, 1, gl.FLOAT, false, 0, 0);
367 };

369 EgomModel.prototype.render = function(gl,lineMode) {
370 	if (lineMode===true) {
371 		gl.drawArrays(gl.LINES, this.bufferStartLines, 2*this.lines.length);
372 	} else {
373 		gl.drawArrays(gl.TRIANGLES, this.bufferStart, 3*this.triangles.length);
374 	}
375 };
