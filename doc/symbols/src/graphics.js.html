 40 /**
 41  * Initiates the loading of the texture from file and sets a callback for the
 42  * finish of the loading to initiate loading the next texture from the resource
 43  * center if such exists, otherwise to execute the callback passed in parameter.
 44  * This is required to make sure all the textures are loaded by the time the
 45  * final callback is executed, that is proceeding with any further steps already
 46  * making use of the textures.
 47  * @param {ResourceCenter} resourceCenter The resource center object where the
 48  * subsequents texture resources to be loaded are stored.
 49  * @param {number} index The index of the current texture resource in the array
 50  * of textures.
 51  * @param {function} callback The function to be executed after all textures
 52  * have been loaded.
 53  */
 54 Texture.prototype.chainLoad = function(resourceCenter,index,callback) {
 55 	var self = resourceCenter.textures[index];
 56 	if(resourceCenter.textures.length-1===index) {
 57 		self.image.onload = callback;
 58 	} else {
 59 		self.image.onload = function() {
 60                     document.getElementById("status").innerHTML="loaded texture: "+self.filename+".";
 61                     self.chainLoad(resourceCenter,index+1,callback);
 62                 };
 63 	}
 64 	self.image.src=self.filename;
 65 };

 67 /**
 68  * Sets up the webGL resource for the texture within the provided GL context.
 69  * Call only after the chainload has finished!
 70  * @param {WebGLRenderingContext} gl The webGL context.
 71  */
 72 Texture.prototype.setup = function(gl) {
 73 	this.id = gl.createTexture();
 74 	gl.bindTexture(gl.TEXTURE_2D, this.id);
 75 	
 76 	// Set the parameters so we can render any size image.
 77 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
 78 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
 79 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
 80 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
 81 
 82 	// Upload the image into the texture.
 83 	gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.image);
 84 };

104 /**
105  * Initiates the loading of one of the face textures of a cubemapped texture 
106  * object from file and sets a callback for the finish of the loading to 
107  * initiate loading the next face, the next cubemapped texture object from the
108  * resource center if such exists, otherwise to execute the callback passed in 
109  * parameter.
110  * This is required to make sure all the pictures are loaded by the time the
111  * final callback is executed, that is proceeding with any further steps already
112  * making use of the textures.
113  * @param {ResourceCenter} resourceCenter The resource center object where the
114  * subsequents cubemapped texture resources to be loaded are stored.
115  * @param {number} index The index of the current cubemap resource in the array
116  * of cubemaps.
117  * @param {number} face The index of the current face to load (0-5)
118  * @param {function} callback The function to be executed after all textures
119  * have been loaded.
120  */
121 Cubemap.prototype.chainLoad = function(resourceCenter,index,face,callback) {
122 	var self = this;
123 	if(face<5) {
124 		this.images[face].onload = function() {self.chainLoad(resourceCenter,index,face+1,callback);};
125 	} else {
126 		if(resourceCenter.cubemaps.length-1===index) {
127 			this.images[face].onload = callback;
128 		} else {
129 			this.images[face].onload = function() {resourceCenter.cubemaps[index+1].chainLoad(resourceCenter,index+1,0,callback);};
130 		}
131 	}
132 	this.images[face].src=this.imageURLs[face];
133 };

135 /**
136  * Sets up the webGL resource for the cubemapped texture within the provided GL
137  * context.
138  * Call only after the chainload has finished!
139  * @param {WebGLRenderingContext} gl The webGL context.
140  */
141 Cubemap.prototype.setup = function(gl) {
142 	// Create a texture.
143 	this.id = gl.createTexture();
144 	gl.bindTexture(gl.TEXTURE_CUBE_MAP, this.id);
145 	
146 	// Set the parameters so we can render any size image.
147 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
148 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
149 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
150 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
151 	
152 	gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false);
153 	
154 	var type = [
155 		gl.TEXTURE_CUBE_MAP_POSITIVE_X,
156 		gl.TEXTURE_CUBE_MAP_NEGATIVE_X,
157 		gl.TEXTURE_CUBE_MAP_POSITIVE_Y,
158 		gl.TEXTURE_CUBE_MAP_NEGATIVE_Y,
159 		gl.TEXTURE_CUBE_MAP_POSITIVE_Z,
160 		gl.TEXTURE_CUBE_MAP_NEGATIVE_Z];
161 
162 	// Upload the images into the texture.
163 	for(var i=0;i<6;i++) {
164 		gl.texImage2D(type[i], 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.images[i]);
165 	}
166 };

185 /**
186  * Enumeration defining the available variables types of shaders.
187  */
188 var ShaderVariableTypes = Object.freeze(
189         {
190             none: 0,
191             float: 1, 
192             mat4: 2, 
193             mat3: 3,
194             vec3: 4,
195             vec4: 5,
196             sampler2D: 6,
197             samplerCube: 7,
198             uint: 8
199         });

201 /**
202  * Determining the enumeration value of a shader variable type from the string
203  * containing the name of the variable type. 
204  * @param {String} type The name of the variable type 
205  */
206 function ShaderVariableTypeFromString(type) {
207     if (type==="float") return ShaderVariableTypes.float;
208     if (type==="mat4") return ShaderVariableTypes.mat4;
209     if (type==="mat3") return ShaderVariableTypes.mat3;
210     if (type==="vec3") return ShaderVariableTypes.vec3;
211     if (type==="vec4") return ShaderVariableTypes.vec4;
212     if (type==="sampler2D") return ShaderVariableTypes.sampler2D;
213     if (type==="samplerCube") return ShaderVariableTypes.samplerCube;
214     if (type==="uint") return ShaderVariableTypes.uint;
215     return ShaderVariableTypes.none;
216 }

218 /**
219  * Creates a new ShaderUniform.
220  * @class A class representing and wrapping a GLSL uniform 
221  * variable.
222  * @param {String} name The name of the uniform variable. Has to be the same
223  * as the name specified in the GLSL source.
224  * @param {String} type The type of the uniform variable. Only certain variable
225  * types are supported.
226  * @param {Number} arraySize If 0 or undefined, the uniform is not an array. If 
227  * one or more, then gives the size of the uniform array.
228  */
229 function ShaderUniform(name,type,arraySize) {
230 	this.name=name;
231 	this.type=ShaderVariableTypeFromString(type);
232         if(arraySize!==undefined) {

234         } else {
235             this.arraySize=0;
236         }
237 	this.location=-1;

253 /**
254  * Get the GL function that sets the uniform of the specified type at the
255  * specified location to the specified value. Used directly from the prototype,
256  * do not use private properties here!
257  * @param {WebGLRenderingContext} gl The GL context
258  * @param {number} location The location of the uniform
259  * @param {ShaderVariableTypes member} type The type of the uniform
260  * @param {object} value The value to set the uniform to (can be of different
261  * types)
262  * @param {Number} arraySize The size of the uniform array. If 0, it is not an array.
263  * @returns {function} A function that executes the webGL query of setting the
264  * uniform to the specified value.
265  */
266 ShaderUniform.prototype.getSetterFunction = function(gl,location,type,value,arraySize) {
267     if(arraySize>0) {
268         switch(type) {
269             case ShaderVariableTypes.float: return function() { gl.uniform1fv(location,value); };
270                 break;
271         }
272     } else {
273         switch(type) {
274             case ShaderVariableTypes.float: return function() { gl.uniform1f(location,value); };
275                 break;
276             case ShaderVariableTypes.mat4: return function() { gl.uniformMatrix4fv(location,false,value); };
277                 break;
278             case ShaderVariableTypes.mat3: return function() { gl.uniformMatrix3fv(location,false,value); };
279                 break;
280             case ShaderVariableTypes.vec3: return function() { gl.uniform3fv(location,value); };
281                 break;
282             case ShaderVariableTypes.vec4: return function() { gl.uniform4fv(location,value); };
283                 break;
284             case ShaderVariableTypes.sampler2D: return function() { gl.uniform1i(location,value); };
285                 break;
286             case ShaderVariableTypes.samplerCube: return function() { gl.uniform1i(location,value); };
287                 break;
288             case ShaderVariableTypes.uint: return function() { gl.uniform1i(location,value); };
289                 break;
290         }
291     }
292 };

348 /**
349  * Performs the full preparation of the webGL shader within the supplied GL
350  * context: dowloads the vertex and fragment shader files, compiles and links
351  * the code using webGL and saves the location IDs of the uniforms of the 
352  * shader.
353  * @param {WebGLRenderingContext} gl The webGL context to use for webGL operations.
354  */
355 Shader.prototype.setup = function(gl) {
356 	var shaderSource = null;
357 	
358 	var request = new XMLHttpRequest();
359 	request.open('GET', this.vertexShaderFileName+"?1.0", false);
360 	request.send(null);
361 	shaderSource = request.responseText;
362 
363 	var vertexShader = gl.createShader(gl.VERTEX_SHADER);
364 	gl.shaderSource(vertexShader, shaderSource);
365 	gl.compileShader(vertexShader);
366 	
367 	request.open('GET', this.fragmentShaderFileName+"?1.0", false);
368 	request.send(null);
369 	shaderSource = request.responseText;
370 
371 	var fragmentShader = gl.createShader(gl.FRAGMENT_SHADER);
372 	gl.shaderSource(fragmentShader, shaderSource);
373 	gl.compileShader(fragmentShader);
374 
375 	this.id = gl.createProgram();
376 	gl.attachShader(this.id, vertexShader);
377         gl.attachShader(this.id, fragmentShader);
378         gl.linkProgram(this.id);
379 	
380 	for(var i=0;i<this.uniforms.length;i++) {
381 		this.uniforms[i].getAndSetLocation(gl,this.id);
382 	}
383 };

417 	this.visibleWidth=0;
418 	this.visibleHeight=0;

571 /**
572  * Checks if the object is inside the viewing frustum of the passed camera,
573  * taking into account the parents of the object as well. Also sets the view
574  * width and height members of the object.
575  * @param {Camera} camera The camera the frustum of which is to be checked
576  * @returns {boolean} Whether the object is inside the frustum.
577  */
578 VisualObject.prototype.isInsideViewFrustum = function(camera) {
579         if (this.renderParent!==null) {
580             if (this.insideParent===undefined) {
581                 this.insideParent=(this.getPositionMatrix()[12]<this.renderParent.getSize()) &&
582                         (this.getPositionMatrix()[13]<this.renderParent.getSize()) && 
583                         (this.getPositionMatrix()[14]<this.renderParent.getSize());
584             }
585             if (this.renderParent.lastInsideFrustumState===false) {
586                 this.visibleWidth=0;
587                 this.visibleHeight=0;
588                 this.lastInsideFrustumState = false;
589                 return this.lastInsideFrustumState;
590             } else if (this.insideParent===true) {
591                 this.visibleWidth=this.renderParent.visibleWidth;
592                 this.visibleHeight=this.renderParent.visibleHeight;
593                 this.lastInsideFrustumState = true;
594                 return this.lastInsideFrustumState;
595             }
596         }
597 	var baseMatrix =
598 			translationMatrixv(getPositionVector4(
599 				mul(
600 					translationMatrixv(this.getCascadePositionVector()),
601 					camera.matrix
602 					)
603 				));
604         var fullMatrix =
605                mul(mul(this.getCascadeScalingMatrix(),baseMatrix),camera.perspectiveMatrix);
606 		
607 	var position = vector4Matrix4Product([0.0,0.0,0.0,1.0],fullMatrix);
608 	position = [position[0]/position[3],position[1]/position[3],position[2]/position[3]];
609 	var zOffsetPosition = vector4Matrix4Product([0.0,0.0,-this.getSize(),1.0],fullMatrix);
610 	var zOffset = (zOffsetPosition[2]/zOffsetPosition[3]);
611 		
612 	// frustum culling: back and front
613 	if (((zOffset>-1.0) && (zOffset<1.0)) || ((position[2]>-1.0) && (position[2]<1.0))) {
614                 
615                 // frustum culling: sides
616                 var xOffsetPosition = vector4Matrix4Product([0.0,0.0,0.0,1.0],mul(mul(baseMatrix,translationMatrix(this.getSize()*this.getCascadeScalingMatrix()[0],0.0,0.0)),camera.perspectiveMatrix));
617                 var yOffsetPosition = vector4Matrix4Product([0.0,0.0,0.0,1.0],mul(mul(baseMatrix,translationMatrix(0.0,this.getSize()*this.getCascadeScalingMatrix()[0],0.0)),camera.perspectiveMatrix));
618 		var xOffset = Math.abs(xOffsetPosition[0]/xOffsetPosition[3]-position[0]);
619                 var yOffset = Math.abs(yOffsetPosition[1]/yOffsetPosition[3]-position[1]);
620                 //alert(offset);
621 		if (
622                     !(((position[0]+xOffset<-1)&&(position[0]-xOffset<-1))||((position[0]+xOffset>1)&&(position[0]-xOffset>1)))&&
623 		    !(((position[1]+yOffset<-1)&&(position[1]-yOffset<-1))||((position[1]+yOffset>1)&&(position[1]-yOffset>1)))
624                     ) {
625 			this.visibleWidth=xOffset;
626 			this.visibleHeight=yOffset;
627 			this.lastInsideFrustumState = true;
628 		} else {
629                         this.visibleWidth=0;
630 			this.visibleHeight=0;
631 			this.lastInsideFrustumState = false;
632 		}
633 	} else {
634                 this.visibleWidth=0;
635 		this.visibleHeight=0;
636 		this.lastInsideFrustumState = false;
637 	}
638         return this.lastInsideFrustumState;

656 /**
657  * Renders the object and all its subnodes.
658  * @param {ResourceCenter} resourceCenter The resource center that holds the
659  * necessary rendering resources .
660  * @param {Scene} scene The scene within which the object is located.
661  * @param {number} screenWidth The size of the rendering viewport in pixels,
662  * to determine the actual drawn size of the object (for dynamic LOD)
663  * * @param {number} screenHeight The size of the rendering viewport in pixels,
664  * to determine the actual drawn size of the object (for dynamic LOD)
665  * @param {boolean} depthMaskPhase Whether we are drawing in the depthmask
666  * enabled or disabled phase (renders only phase matches with the type of the
667  * shader the object has)
668  * */ 
669 VisualObject.prototype.cascadeRender = function(resourceCenter,scene,screenWidth,screenHeight,depthMaskPhase) {
670     // the visible property determines visibility of all subnodes as well
671     if(this.visible) {
672         // subnodes (children) are only rendered if the parent's visible size
673         // reaches a set limit
674         if ((this.renderParent===null) || (this.smallestParentSizeWhenDrawn===undefined) ||
675                 (Math.max(this.renderParent.visibleWidth*screenWidth,this.renderParent.visibleHeight*screenHeight)>=this.smallestParentSizeWhenDrawn)) {
676             // checking if the object is rendered in this phase (depth mask on/
677             // off) and it is inside the view frustum
678             if(((this.renderedWithDepthMask===true)&&(depthMaskPhase===true))||
679                     ((this.renderedWithoutDepthMask===true)&&(depthMaskPhase===false))) {
680                 // the frustum check only needs to be calculated if this is the
681                 // first pass (depth mask on), or the object wasn't rendered
682                 // in the first pass
683                 if ((
684                         ((this.renderedWithDepthMask===false)||(depthMaskPhase===true))
685                         &&(this.isInsideViewFrustum(scene.activeCamera))
686                     )||(this.lastInsideFrustumState===true)) {
687                     if(this.needsToBeRendered(screenWidth,screenHeight,scene.lodContext,depthMaskPhase)) {
688                         resourceCenter.setCurrentShader(this.shader,scene);
689                         this.assignUniforms(resourceCenter.gl);
690                         this.render(resourceCenter,depthMaskPhase);
691                     }
692                 }
693             }
694             // recursive rendering of all subnodes
695             for(var i=0;i<this.subnodes.length;i++) {
696                 this.subnodes[i].cascadeRender(resourceCenter,scene,screenWidth,screenHeight,depthMaskPhase);
697             }
698         }
699     }

891  * with the calculated LOD.

895  * @param {Boolean} depthMask
896  * @returns {Boolean}
897  */
898 Mesh.prototype.needsToBeRendered = function(screenWidth,screenHeight,lodContext,depthMask) {
899     // choose the model of appropriate LOD
900     var visibleSize = Math.max(this.visibleWidth*screenWidth,this.visibleHeight*screenHeight);
901     var closestLOD = -1;
902     for(var i=0;i<this.modelsWithLOD.length;i++) {
903             if(
904                     (closestLOD===-1) ||
905                     (this.modelsWithLOD[i].lod<=lodContext.maxEnabledLOD) &&
906                     (
907                             (closestLOD>lodContext.maxEnabledLOD) ||
908                             ((lodContext.thresholds[closestLOD]>visibleSize) && (lodContext.thresholds[this.modelsWithLOD[i].lod]<=visibleSize)) ||
909                             ((lodContext.thresholds[closestLOD]<=visibleSize) && (lodContext.thresholds[this.modelsWithLOD[i].lod]<=visibleSize) && (this.modelsWithLOD[i].lod>closestLOD)) ||
910                             ((lodContext.thresholds[closestLOD]>visibleSize) && (lodContext.thresholds[this.modelsWithLOD[i].lod]>visibleSize) && (this.modelsWithLOD[i].lod<closestLOD))
911                     )) {
912                     closestLOD=this.modelsWithLOD[i].lod;
913                     this.model=this.modelsWithLOD[i].model;
914             }
915     }
916     
917     if (this.lineMode===true) {
918             return true;
919     } else {
920             if(depthMask===true) {
921                 if(this.model.nOpaqueTriangles>0) {
922                     return true;
923                 }
924             } else if ((depthMask===false)&&(this.model.nTransparentTriangles>0)) {
925                 return true;
926             }
927     }
928     return false;

1326 /**

1328  * new followed camera if it did not adapt to it fully yet.

1330 SceneCamera.prototype.update = function() {
1331     if(this.followedCamera!==undefined) {
1332         if(this.adaptationTimeLeft>0) {
1333             var currentTime=new Date().getTime();
1334             var adaptationProgress=Math.min(1.0,(currentTime-this.adaptationStartTime)/this.adaptationTime);
1335             this.adaptationTimeLeft=this.adaptationTime-(currentTime-this.adaptationStartTime);
1336             var trans = translationMatrix(
1337                     (this.followedCamera.positionMatrix[12]-this.adaptationStartPositionMatrix[12])*adaptationProgress,
1338                     (this.followedCamera.positionMatrix[13]-this.adaptationStartPositionMatrix[13])*adaptationProgress,
1339                     (this.followedCamera.positionMatrix[14]-this.adaptationStartPositionMatrix[14])*adaptationProgress
1340                     );
1341             var newPositionMatrix=translate(this.adaptationStartPositionMatrix,trans);
1342             var velocityMatrix = mul(translationMatrix(
1343                 newPositionMatrix[12]-this.positionMatrix[12],
1344                 newPositionMatrix[13]-this.positionMatrix[13],
1345                 newPositionMatrix[14]-this.positionMatrix[14]),this.orientationMatrix);
1346             this.velocityVector = [velocityMatrix[12],velocityMatrix[13],velocityMatrix[14]];
1347             this.positionMatrix=newPositionMatrix;
1348             this.orientationMatrix=correctOrthogonalMatrix(addMatrices4(
1349                 mulMatrix4Scalar(this.adaptationStartOrientationMatrix,1.0-adaptationProgress),
1350                 mulMatrix4Scalar(this.followedCamera.orientationMatrix,adaptationProgress)));
1351             this.matrix=mul(this.positionMatrix,this.orientationMatrix);
1352             this.setFOV(this.adaptationStartFOV+(this.followedCamera.fov-this.adaptationStartFOV)*adaptationProgress);
1353         } else {
1354             this.positionMatrix=this.followedCamera.positionMatrix;
1355             this.orientationMatrix=this.followedCamera.orientationMatrix;
1356             this.matrix=this.followedCamera.matrix;
1357             this.perspectiveMatrix=this.followedCamera.perspectiveMatrix;
1358         }

1720 	this.shaders=new Array();
1721 	
1722 	var shaderTags = shadersSource.getElementsByTagName("Shader");
1723 	for(var i=0;i<shaderTags.length;i++) {
1724 		this.shaders.push(new Shader(
1725 			shaderTags[i].getAttribute("name"),
1726 			shaderTags[i].getElementsByTagName("vertex")[0].getAttribute("filename"),
1727 			shaderTags[i].getElementsByTagName("fragment")[0].getAttribute("filename"),
1728 			shaderTags[i].getElementsByTagName("blendType")[0].getAttribute("value"),
1729 			new Array(),
1730 			new Array())
1731 			);
1732 		var attributeTags = shaderTags[i].getElementsByTagName("attribute");
1733 		for(var j=0;j<attributeTags.length;j++) {
1734 			this.shaders[i].attributes.push(new ShaderAttribute(
1735 				attributeTags[j].getAttribute("name"),
1736 				parseInt(attributeTags[j].getAttribute("size")),
1737 				attributeTags[j].getAttribute("role"))
1738 				);
1739 		}
1740 		var uniformTags = shaderTags[i].getElementsByTagName("uniform");
1741 		for(var j=0;j<uniformTags.length;j++) {
1742 			this.shaders[i].uniforms.push(new ShaderUniform(
1743 				uniformTags[j].getAttribute("name"),
1744 				uniformTags[j].getAttribute("type"),
1745                                 uniformTags[j].hasAttribute("arraySize")?uniformTags[j].getAttribute("arraySize"):0)
1746 				);
1747 		}
1748 	}

1751 /**
1752  * Loads the vertex buffer data to the graphic memory for the given shader program using webGL.
1753  * @param {Shader} shader The shader program for which the buffers will be loaded.
1754  * @param {boolean} loadLines Whether to fill the buffer for drawing lines as well next to triangles.
1755  */
1756 ResourceCenter.prototype.setupBuffers = function(shader,loadLines) {
1757 	this.gl.useProgram(shader.id);
1758 	var sumVertices=0;
1759 	for(var i=0;i<this.models.length;i++) {
1760 		sumVertices=sumVertices+
1761 			(loadLines?this.models[i].lines.length*2:0)+
1762 			this.models[i].triangles.length*3;
1763 	}
1764 	
1765 	// creating float32arrays of the appropriate size for the buffers
1766 	shader.vertexBuffers=new Array();
1767 	for(var i=0;i<shader.attributes.length;i++) {
1768 		shader.vertexBuffers.push(new VertexBuffer(-1,new Float32Array(sumVertices*shader.attributes[i].size),null,shader.attributes[i].size));
1769 	}
1770 	
1771 	// filling the buffer data arrays from model data
1772 	var bufferSize=0;
1773 	var objectBufferData=null;
1774 	for(var i=0;i<this.models.length;i++) {
1775 		var linesToLoad=loadLines;
1776 		var trianglesToLoad=true;
1777 		while(linesToLoad||trianglesToLoad) {
1778 			objectBufferData=this.models[i].getBuffers(linesToLoad);
1779 			 if (linesToLoad) {
1780 				this.models[i].bufferStartLines=bufferSize;
1781                             } else {
1782 				this.models[i].bufferStart=bufferSize;
1783                                 this.models[i].bufferStartTransparent=bufferSize+this.models[i].nOpaqueTriangles*3;
1784                             }
1785 			for(var j=0;j<shader.attributes.length;j++) {
1786 				shader.vertexBuffers[j].data.set(objectBufferData[shader.attributes[j].role],bufferSize*shader.attributes[j].size);
1787 			}
1788 			bufferSize+=Math.round(objectBufferData[shader.attributes[0].role].length/shader.attributes[0].size);
1789 			linesToLoad?
1790 				(linesToLoad=false):
1791 				(trianglesToLoad=false);
1792 		}
1793 	}
1794 	
1795 	// provide the data to webGL
1796 	for(var i=0;i<shader.attributes.length;i++) {
1797 		shader.vertexBuffers[i].location = this.gl.getAttribLocation(shader.id, shader.attributes[i].name);
1798 		shader.vertexBuffers[i].id = this.gl.createBuffer();
1799 		this.gl.bindBuffer(this.gl.ARRAY_BUFFER, shader.vertexBuffers[i].id);
1800 		this.gl.bufferData(
1801 			this.gl.ARRAY_BUFFER, 
1802 			shader.vertexBuffers[i].data,
1803 			this.gl.STATIC_DRAW);
1804 		this.gl.enableVertexAttribArray(shader.vertexBuffers[i].location);
1805                 this.gl.vertexAttribPointer(shader.vertexBuffers[i].location, shader.attributes[i].size, this.gl.FLOAT, false, 0, 0);
1806 	}
1807 };

1812  * @param {Shader} shader The shader to set as current.
1813  * @param {Scene} scene The scene we are drawing.

1818                 if (shader.blendType==="mix") {

1821                     this.gl.blendFunc(this.gl.SRC_ALPHA, this.gl.ONE);

1829 /**
1830  * Binds the given texture or cubemap resource with webGL
1831  * @param {Texture|Cubemap} texture The resource to bind for rendering.
1832  * @param {Number} place To which activeTexture place does it need to bind (for multi-texturing, starting with 0)
1833  */
1834 ResourceCenter.prototype.bindTexture = function(texture,place) {
1835     switch(place) {
1836         case 0: this.gl.activeTexture(this.gl.TEXTURE0);
1837             break;
1838         case 1: this.gl.activeTexture(this.gl.TEXTURE1);
1839             break;
1840         case 2: this.gl.activeTexture(this.gl.TEXTURE2);
1841             break;
1842         case 3: this.gl.activeTexture(this.gl.TEXTURE3);
1843             break;
1844     }
1845     if(this.boundTextures[place]!==texture) {
1846             if (texture instanceof Texture) {
1847                     this.gl.bindTexture(this.gl.TEXTURE_2D, texture.id);
1848             } else
1849             if (texture instanceof Cubemap) {
1850                     this.gl.bindTexture(this.gl.TEXTURE_CUBE_MAP, texture.id);
1851             }
1852             this.boundTextures[place]=texture;
1853     }

1856 /**
1857  * Sets up a webGL context and performs the basic configuration with it.
1858  * @param {object} canvas The HMTL5 canvas to get and configure the context for.
1859  * @returns {boolean} Indicates whether the creation of the WebGL context succeeded.
1860  */
1861 ResourceCenter.prototype.setupWebGL = function(canvas) {
1862 	try {
1863 		// Try to grab the standard context. If it fails, fallback to experimental.
1864 		this.gl = canvas.getContext("webgl",{alpha: false, antialias: true}) || canvas.getContext("experimental-webgl",{alpha: false, antialias: true});
1865 	}
1866 	catch(e) {}
1867 
1868 	// If we don't have a GL context, give up now
1869 	if (!this.gl) {
1870 		alert("Unable to initialize WebGL. Your browser might not support it.");
1871                 return false;
1872 	}
1873 	
1874 	this.gl.clearDepth(1.0);
1875 	this.gl.colorMask(true, true, true, true);
1876 	this.gl.enable(this.gl.DEPTH_TEST);
1877 	this.gl.depthFunc(this.gl.LEQUAL); 
1878 	this.gl.enable(this.gl.CULL_FACE);
1879 	this.gl.cullFace(this.gl.BACK);
1880 	this.gl.frontFace(this.gl.CCW);
1881 	this.gl.blendFunc(this.gl.SRC_ALPHA, this.gl.ONE_MINUS_SRC_ALPHA);
1882 	this.gl.enable(this.gl.BLEND);
1883 	this.gl.enable(this.gl.SCISSOR_TEST);
1884 	this.gl.activeTexture(this.gl.TEXTURE0);
1885 	
1886 	this.boundTexture=null;
1887 	
1888 	this.setupShaders();
1889 	this.setupTextures();
1890 	this.setupCubemaps();
1891 	
1892 	for(var i=0;i<this.shaders.length;i++) {
1893 		this.setupBuffers(this.shaders[i],true);
1894 	}
1895         
1896         return true;
1897 };
