 28 /**
 29  * Defines a screen component object.
 30  * @class A reusable component that consist of HTML elements (a fragment of a 
 31  * HTML document) and can be appended to game screens. Various components have
 32  * to be the descendants of this class, and implement their own various methods.
 33  * @extends Resource
 34  * @param {String} name The name of the component to be identified by.
 35  * @param {String} source The filename of the HTML document where the structure
 36  * of the component should be defined. The component will be loaded as the first
 37  * element (and all its children) inside the body tag of this file.
 38  * @returns {ScreenComponent}
 39  */
 40 function ScreenComponent(name,source) {
 41     Resource.call(this);
 42     
 43     this._name=name;
 44     this._source=source;
 45     
 46     this._model=null;
 47     
 48     this._rootElement = null;
 49     this._rootElementDefaultDisplayMode = null;
 50     
 51     // function to execute when the model is loaded
 52     this._onModelLoad = function() {};
 53     
 54     // source will be undefined when setting the prototypes for inheritance
 55     if(source!==undefined) {
 56         this.requestModelLoad();
 57     }
 58 }
 59 
 60 ScreenComponent.prototype = new Resource();
 61 ScreenComponent.prototype.constructor = ScreenComponent;
 62 
 63 /**
 64  * Initiates the asynchronous loading of the component's structure from the
 65  * external HTML file.
 66  */
 67 ScreenComponent.prototype.requestModelLoad = function() {
 68     // send an asynchronous request to grab the XML file containing the DOM of
 69     // this component
 70     var request = new XMLHttpRequest();
 71     request.open('GET', location.pathname+getComponentFolder()+this._source+"?123", true);
 72     var self = this;
 73     request.onreadystatechange = function() {
 74             if(request.readyState===4) {
 75                 self._model = document.implementation.createHTMLDocument(self._name);
 76                 self._model.documentElement.innerHTML = this.responseText;
 77                 self._onModelLoad();
 78             }
 79         };
 80     request.send(null);
 81 };
 82 
 83 /**
 84  * Appends the component's elements to the current document.
 85  */
 86 ScreenComponent.prototype.appendToPage = function() {
 87     var self = this;
 88     var appendToPageFunction = function() {
 89         self._rootElement = document.body.appendChild(document.importNode(self._model.body.firstElementChild,true));
 90         self._rootElementDefaultDisplayMode = self._rootElement.style.display;
 91         self._initializeComponents();
 92         self.setToReady();
 93     };
 94     // if we have built up the model of the screen already, then load it
 95     if(this._model!==null) {
 96         appendToPageFunction();
 97     // if not yet, set the callback function which fires when the model is 
 98     // loaded
 99     } else {
100         this._onModelLoad = appendToPageFunction;
101     }
102 };

279 /**
280  * Replaces the current HTML page's body with the sctructure of the screen.
281  */
282 GameScreen.prototype.buildPage = function() {
283     var self = this;
284     var buildPageFunction = function() {
285         document.body=self._model.body.cloneNode(true);
286         self._initializeComponents();
287     };
288     // if we have built up the model of the screen already, then load it
289     if(this._model!==null) {
290         buildPageFunction();
291     // if not yet, set the callback function which fires when the model is 
292     // loaded
293     } else {
294         this._onModelLoad = buildPageFunction;
295     }
296 };

340 /**
341  * Defines a game screen with canvases object.
342  * @class Represents a game screen that has one or more canvases where WebGL
343  * scenes can be rendered.
344  * @extends GameScreen
345  * @param {String} name The name by which this screen can be identified.
346  * @param {String} source The name of the HTML file where the structure of this
347  * screen is defined.
348  * @returns {GameScreenWithCanvases}
349  */
350 function GameScreenWithCanvases(name,source) {
351     GameScreen.call(this,name,source);
352     
353     this._fixCanvases=null;
354     this._resizeableCanvases=null;
355     
356     this._scenes=new Array();
357     
358     this._renderLoop = null;
359 };
360 
361 GameScreenWithCanvases.prototype=new GameScreen();
362 GameScreenWithCanvases.prototype.constructor=GameScreenWithCanvases;
363 
364 /**
365  * Stops the render loop and nulls out the components.
366  */
367 GameScreenWithCanvases.prototype.closePage = function() {
368     GameScreen.prototype.closePage.call(this);
369     
370     this.stopRenderLoop();
371     
372     this._fixCanvases = null;
373     this._resizeableCanvases = null;
374 };
375 
376 /**
377  * Initializes the components of the parent class, then the additional ones for
378  * this class (the canvases).
379  */
380 GameScreenWithCanvases.prototype._initializeComponents = function() {
381     GameScreen.prototype._initializeComponents.call(this);
382     
383     this._fixCanvases = document.querySelectorAll("canvas.fix");
384     this._resizeableCanvases= document.querySelectorAll("canvas.resizeable");
385     
386     var self = this;
387     this._model.body.onresize = self.resizeCanvases;
388 };

422 /**
423  * If there is only one canvas, returns is.
424  * @returns {Element}
425  */
426 GameScreenWithCanvases.prototype.getCanvas = function() {
427     if((this._fixCanvases.length===1)&&(this._resizeableCanvases.length===0)) {
428         return this._fixCanvases[0];
429     } else if((this._fixCanvases.length===0)&&(this._resizeableCanvases.length===1)) {
430         return this._resizeableCanvases[0];
431     } else {
432         game.showError("Screen '"+this._name+"' has zero or more than one canvases, cannot return one!");
433         return null;
434     }
435 };
436 
437 /**
438  * Updates all needed variables when the screen is resized (camera perspective
439  * matrices as well!)
440  */
441 GameScreenWithCanvases.prototype.resizeCanvases = function() {
442     var i;
443     // first, update the canvas width and height properties if the client width/
444     // height has changed
445     for (i = 0; i < this._resizeableCanvases.length; i++) {
446         var width = this._resizeableCanvases[i].clientWidth;
447         var height = this._resizeableCanvases[i].clientHeight;
448         if (this._resizeableCanvases[i].width !== width ||
449                 this._resizeableCanvases[i].height !== height) {
450             // Change the size of the canvas to match the size it's being displayed
451             this._resizeableCanvases[i].width = width;
452             this._resizeableCanvases[i].height = height;
453         }
454     }
455     // updated the variables in the scenes
456     for (i = 0; i < this._scenes.length; i++) {
457         this._scenes[i].resizeViewport();
458     }
459 };
