 27 /**
 28  * Defines a force class.
 29  * @class Represents a force affecting a physical object, causing it to 
 30  * accelerate in a linear pace.
 31  * @param {String} id Can be used to identify continuous forces so that their
 32  * duration can be subsequently renewed.
 33  * @param {Number} strength The strength of the force in newtons.
 34  * @param {Number[]} direction The vector describing the direction in which
 35  * the force creates the acceleration. (needs to be unit vector)
 36  * @param {number} duration The duration until which the force is still in
 37  * effect, given in milliseconds.
 38  */
 39 function Force(id,strength,direction,duration) {
 40         this.id=id;
 41 	this.strength=strength;
 42 	this.direction=direction;
 43 	this.duration=duration;
 44 }
 45 
 46 function Torque(id,strength,axis,duration) {
 47         this.id=id;
 48 	this.strength=strength;
 49 	this.axis=axis;
 50 	this.duration=duration;
 51 }

 53 function Body(positionMatrix,orientationMatrix,width,height,depth) {
 54 	this.positionMatrix=positionMatrix;
 55 	this.orientationMatrix=orientationMatrix;
 56 	this.modelMatrixInverse=mul(inverseTranslationMatrix(this.positionMatrix),inverseRotationMatrix(this.orientationMatrix));
 57 	this.width=width;
 58 	this.height=height;
 59 	this.depth=depth;
 60 	this.halfWidth=width/2;
 61 	this.halfHeight=height/2;
 62 	this.halfDepth=depth/2;
 63 }
 64 
 65 function PhysicalObject(mass,size,positionMatrix,orientationMatrix,initialVelocityMatrix,bodies) {
 66 	this.mass=mass;
 67 	this.size=size;
 68 	this.sizeSquared=size*size;
 69 	this.positionMatrix=positionMatrix;
 70 	this.orientationMatrix=orientationMatrix;
 71 	this.modelMatrixInverse=mul(inverseTranslationMatrix(this.positionMatrix),inverseRotationMatrix(this.orientationMatrix));
 72 	this.timeSinceLastMatrixCorrection=0;
 73 	
 74 	this.velocityMatrix=initialVelocityMatrix;
 75 	this.angularVelocityMatrix=identityMatrix4();
 76 	
 77 	this.forces=new Array();
 78 	this.torques=new Array();
 79 	this.bodies=bodies;
 80 	this.bodySize=-1;
 81 	this.calculateBodySize();
 82 }

 84 /**
 85  * Checks the forces for one with the given ID, if it exists, renews its
 86  * properties, if it does not, adds a new force with the given parameters. It
 87  * will renew all forces with the given ID, if more than one exists.
 88  * @param {String} forceID The ID of the force to look for
 89  * @param {number} strength The strength of the force.
 90  * @param {number[]} direction The vector describing the direction of the force.
 91  * @param {number} duration The force will either created with, or renewed to
 92  * last for this duration.
 93  */
 94 PhysicalObject.prototype.addOrRenewForce = function(forceID,strength,direction,duration) {
 95     var i;
 96     var found = false;
 97     for(i=0;i<this.forces.length;i++) {
 98         if (this.forces[i].id===forceID) {
 99             this.forces[i].strength=strength;
100             this.forces[i].direction=direction;
101             this.forces[i].duration=duration;
102             found=true;
103         }
104     }
105     if (found===false) {
106         this.forces.push(new Force(forceID,strength,direction,duration));
107     }
108 };

110 /**
111  * Checks the torques for one with the given ID, if it exists, renews its
112  * properties, if it does not, adds a new torque with the given parameters. It
113  * will renew all torques with the given ID, if more than one exists.
114  * @param {String} torqueID The ID of the torque to look for
115  * @param {number} strength The strength of the torque.
116  * @param {number[]} axis The vector describing the axis of the torque.
117  * @param {number} duration The torque will either created with, or renewed to
118  * last for this duration.
119  */
120 PhysicalObject.prototype.addOrRenewTorque = function(torqueID,strength,axis,duration) {
121     var i;
122     var found = false;
123     for(i=0;i<this.torques.length;i++) {
124         if (this.torques[i].id===torqueID) {
125             this.torques[i].strength=strength;
126             this.torques[i].axis=axis;
127             this.torques[i].duration=duration;
128             found=true;
129         }
130     }
131     if (found===false) {
132         this.torques.push(new Torque(torqueID,strength,axis,duration));
133     }
134 };

136 PhysicalObject.prototype.calculateBodySize = function() {
137 	this.bodySize=0;
138         var bodyPos;
139 	for(var i=0;i<this.bodies.length;i++) {
140 		bodyPos = getPositionVector(this.bodies[i].positionMatrix);
141 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[this.bodies[i].halfWidth,this.bodies[i].halfHeight,this.bodies[i].halfDepth])));
142 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[this.bodies[i].halfWidth,this.bodies[i].halfHeight,-this.bodies[i].halfDepth])));
143 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[this.bodies[i].halfWidth,-this.bodies[i].halfHeight,this.bodies[i].halfDepth])));
144 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[this.bodies[i].halfWidth,-this.bodies[i].halfHeight,-this.bodies[i].halfDepth])));
145 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[-this.bodies[i].halfWidth,this.bodies[i].halfHeight,this.bodies[i].halfDepth])));
146 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[-this.bodies[i].halfWidth,this.bodies[i].halfHeight,-this.bodies[i].halfDepth])));
147 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[-this.bodies[i].halfWidth,-this.bodies[i].halfHeight,this.bodies[i].halfDepth])));
148 		this.bodySize=Math.max(this.bodySize,vector3Length(vectorAdd3(bodyPos,[-this.bodies[i].halfWidth,-this.bodies[i].halfHeight,-this.bodies[i].halfDepth])));
149 	}
150 	this.bodySize*=1/this.size;
151 };

153 PhysicalObject.prototype.checkHit = function(positionVector,direction,range) {
154 	if ((Math.abs(positionVector[0]-this.positionMatrix[12])<this.bodySize)&&
155 		(Math.abs(positionVector[1]-this.positionMatrix[13])<this.bodySize)&&
156 		(Math.abs(positionVector[2]-this.positionMatrix[14])<this.bodySize)) {
157 			
158 		var relativePos = vector4Matrix4Product(positionVector,this.modelMatrixInverse);
159 		var result=false;
160                 var posRelativeToBody;
161 		for(var i=0;(result===false)&&(i<this.bodies.length);i++) {
162 			posRelativeToBody = vector4Matrix4Product(relativePos,this.bodies[i].modelMatrixInverse);
163 			result = 
164 				(posRelativeToBody[0]>=-this.bodies[i].halfWidth)&&(posRelativeToBody[0]<=this.bodies[i].halfWidth)&&
165 				(posRelativeToBody[1]>=-this.bodies[i].halfHeight)&&(posRelativeToBody[1]<=this.bodies[i].halfHeight)&&
166 				(posRelativeToBody[2]>=-this.bodies[i].halfDepth)&&(posRelativeToBody[2]<=this.bodies[i].halfDepth);
167 		}
168 	}
169 	return result;
170 };

172 PhysicalObject.prototype.correctMatrices = function() {
173         this.orientationMatrix = correctOrthogonalMatrix(this.orientationMatrix);
174         this.angularVelocityMatrix = correctOrthogonalMatrix(this.angularVelocityMatrix);	
175 	this.timeSinceLastMatrixCorrection=0;
176 };

178 /**
179  * Performs the physics calculations (movement, turn, acceleration) for the
180  * object based on the forces that are affecting it.
181  * @param {Number} dt The elapsed time since the last simulation step, in
182  * milliseconds.
183  */
184 PhysicalObject.prototype.simulate = function(dt) {
185         var i,a,t;
186         if (dt>0) {
187 	// first calculate the movement that happened in the past dt
188         // milliseconds as a result of the velocity sampled in the previous step
189         // the velocity matrix is in m/s
190         this.positionMatrix=mul(this.positionMatrix,translationMatrixv(scalarVector3Product(dt/1000,getPositionVector(this.velocityMatrix))));
191     
192         // calculate the movement that happened as a result of the acceleration
193         // the affecting forces caused since the previous step
194         // (s=1/2*a*t^2)
195         var accelerationMatrix=identityMatrix4();
196         for(i=0;(i<this.forces.length)&&(this.forces[i].duration>0);i++) {
197             a = this.forces[i].strength/this.mass;
198             t = Math.min(dt,this.forces[i].duration)/1000; // t is in seconds
199             this.positionMatrix=mul(this.positionMatrix,translationMatrixv(scalarVector3Product(1/2*a*t*t,this.forces[i].direction)));
200             // calculate the acceleration caused to update the velocity matrix
201             accelerationMatrix = mul(
202 			accelerationMatrix,
203 			translationMatrix(
204 				a*t*this.forces[i].direction[0],
205 				a*t*this.forces[i].direction[1],
206 				a*t*this.forces[i].direction[2])
207 			);
208             // register that the force's effect has been considered for dt ms
209             this.forces[i].duration=Math.max(this.forces[i].duration-dt,0.0);			
210 	}
211         // update velocity matrix
212         this.velocityMatrix=mul(this.velocityMatrix,accelerationMatrix);
213 	// remove past forces
214 	while((this.forces.length>0)&&(this.forces[0].duration<=0)) {
215 		this.forces.shift();
216 	}
217         
218         // the same process with rotation and torques
219         // the angular velocity matrix represents the rotation that happens
220         // during the course of 5 milliseconds (since rotation cannot be
221         // interpolated easily, for that quaternions should be used)
222         for(i=0;i+2<dt;i+=5) {
223             this.orientationMatrix=mul(this.orientationMatrix,this.angularVelocityMatrix);
224         }
225 	
226         // calculate the rotation that happened as a result of the angular
227         // acceleration the affecting torques caused since the previous step
228 	var angularAccMatrix=identityMatrix4();
229 	for(i=0;(i<this.torques.length)&&(this.torques[i].duration>0);i++) {
230             // in reality, the shape of the object should be taken into account,
231             // for simplicity, the mass is taken as the only coefficient
232             a = this.torques[i].strength/this.mass; 
233             t = Math.min(dt,this.torques[i].duration)/1000; // t is in seconds
234             this.orientationMatrix=mul(this.orientationMatrix,rotationMatrix4(this.torques[i].axis,1/2*a*t*t));
235             // calculate the acceleration caused to update the ang. acc. matrix
236             angularAccMatrix = mul(
237 			angularAccMatrix,
238 			rotationMatrix4(this.torques[i].axis,a*t*200));
239                         // multiplied by 200 since we need the amount or rotation
240                         // caused per 5 ms, not per 1 s represented in the 
241                         // matrix (see above)
242             // register that the torque's effect has been considered for dt ms
243             this.torques[i].duration=Math.max(this.torques[i].duration-dt,0.0);
244 	}
245         // update angular velocity matrix
246         this.angularVelocityMatrix=mul(this.angularVelocityMatrix,angularAccMatrix);
247 	// remove past torques
248 	while((this.torques.length>0)&&(this.torques[0].duration<=0)) {
249 		this.torques.shift();
250 	}
251 	
252 	// save the inverse of the model matrix so it's only calculated once
253         // every simulation step
254 	this.modelMatrixInverse=inverse4(mul(this.orientationMatrix,this.positionMatrix));
255 	
256         // correcting the matrices every 10 seconds, as due to small floating
257         // point calculation inaccuracies, they slowly lose orthogonality
258 	this.timeSinceLastMatrixCorrection+=dt;
259 	if(this.timeSinceLastMatrixCorrection>=10000) {
260 		this.correctMatrices();
261 	}
262     }
263 };
