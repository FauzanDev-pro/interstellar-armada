  9 /**
 10  * Creates a new Texture object.
 11  * @class Represents a webGL texture resource
 12  * @param {String} filename The name of file from which the texture resource 
 13  * is to be loaded. The constructor itself does not initiate the loading.
 14  */
 15 function Texture(filename) {
 16 	this.filename=filename;
 17 	this.id=-1;
 18 	this.image = new Image();
 19 }
 20 
 21 /**
 22  * Initiates the loading of the texture from file and sets a callback for the
 23  * finish of the loading to initiate loading the next texture from the resource
 24  * center if such exists, otherwise to execute the callback passed in parameter.
 25  * This is required to make sure all the textures are loaded by the time the
 26  * final callback is executed, that is proceeding with any further steps already
 27  * making use of the textures.
 28  * @param {ResourceCenter} resourceCenter The resource center object where the
 29  * subsequents texture resources to be loaded are stored.
 30  * @param {number} index The index of the current texture resource in the array
 31  * of textures.
 32  * @param {function} callback The function to be executed after all textures
 33  * have been loaded.
 34  */
 35 Texture.prototype.chainLoad = function(resourceCenter,index,callback) {
 36 	var self = resourceCenter.textures[index];
 37 	if(resourceCenter.textures.length-1===index) {
 38 		self.image.onload = callback;
 39 	} else {
 40 		self.image.onload = function() {self.chainLoad(resourceCenter,index+1,callback);};
 41 	}
 42 	self.image.src=self.filename;
 43 };

 45 /**
 46  * Sets up the webGL resource for the texture within the provided GL context.
 47  * Call only after the chainload has finished!
 48  * @param {object} gl The webGL context.
 49  */
 50 Texture.prototype.setup = function(gl) {
 51 	this.id = gl.createTexture();
 52 	gl.bindTexture(gl.TEXTURE_2D, this.id);
 53 	
 54 	// Set the parameters so we can render any size image.
 55 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
 56 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
 57 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
 58 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
 59 
 60 	// Upload the image into the texture.
 61 	gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.image);
 62 };

133 /**
134  * Sets up and returns a simple EgomModel that contains a cuboid (XYZ-box) model 
135  * with the given properties.
136  * @param {number} width Size of the box along the X axis.
137  * @param {number} height Size of the box along the Y axis.
138  * @param {number} depth Size of the box along the Z axis.
139  * @param {number[]} color A vector containing the RGBA components of the 
140  * desired box color.
141  */
142 function cuboidModel(width,height,depth,color) {
143 	var result = new EgomModel();
144 	// front
145 	result.vertices.push([-width/2,-height/2,depth/2]);
146 	result.vertices.push([width/2,-height/2,depth/2]);
147 	result.vertices.push([width/2,height/2,depth/2]);
148 	result.vertices.push([-width/2,height/2,depth/2]);
149 	// back
150 	result.vertices.push([-width/2,height/2,-depth/2]);
151 	result.vertices.push([width/2,height/2,-depth/2]);
152 	result.vertices.push([width/2,-height/2,-depth/2]);
153 	result.vertices.push([-width/2,-height/2,-depth/2]);
154 	// top
155 	result.vertices.push([width/2,height/2,-depth/2]);
156 	result.vertices.push([-width/2,height/2,-depth/2]);
157 	result.vertices.push([-width/2,height/2,depth/2]);
158 	result.vertices.push([width/2,height/2,depth/2]);
159 	// bottom
160 	result.vertices.push([-width/2,-height/2,-depth/2]);
161 	result.vertices.push([width/2,-height/2,-depth/2]);
162 	result.vertices.push([width/2,-height/2,depth/2]);
163 	result.vertices.push([-width/2,-height/2,depth/2]);
164 	// right
165 	result.vertices.push([width/2,-height/2,depth/2]);
166 	result.vertices.push([width/2,-height/2,-depth/2]);
167 	result.vertices.push([width/2,height/2,-depth/2]);
168 	result.vertices.push([width/2,height/2,depth/2]);
169 	// left
170 	result.vertices.push([-width/2,height/2,depth/2]);
171 	result.vertices.push([-width/2,height/2,-depth/2]);
172 	result.vertices.push([-width/2,-height/2,-depth/2]);
173 	result.vertices.push([-width/2,-height/2,depth/2]);
174 	
175 	var normals = [[0,0,1],[0,0,-1],[0,1,0],[0,-1,0],[1,0,0],[-1,0,0]];
176 	
177 	for(var i=0;i<6;i++) {
178 		result.triangles.push(new Triangle((i*4)+0,(i*4)+1,(i*4)+2,color[0],color[1],color[2],color[3],128,0, 0,1, 1,1, 1,0, normals[i][0], normals[i][1], normals[i][2], normals[i][0], normals[i][1], normals[i][2], normals[i][0], normals[i][1], normals[i][2])); 
179 		result.triangles.push(new Triangle((i*4)+2,(i*4)+3,(i*4)+0,color[0],color[1],color[2],color[3],128,0, 1,0, 0,0, 0,1, normals[i][0], normals[i][1], normals[i][2], normals[i][0], normals[i][1], normals[i][2], normals[i][0], normals[i][1], normals[i][2])); 
180 		result.triangles.push(new Triangle((i*4)+2,(i*4)+1,(i*4)+0,color[0],color[1],color[2],color[3],128,0, 0,1, 1,1, 1,0, -normals[i][0], -normals[i][1], -normals[i][2], -normals[i][0], -normals[i][1], -normals[i][2], -normals[i][0], -normals[i][1], -normals[i][2])); 
181 		result.triangles.push(new Triangle((i*4)+0,(i*4)+3,(i*4)+2,color[0],color[1],color[2],color[3],128,0, 1,0, 0,0, 0,1, -normals[i][0], -normals[i][1], -normals[i][2], -normals[i][0], -normals[i][1], -normals[i][2], -normals[i][0], -normals[i][1], -normals[i][2])); 
182 	}
183

187 /**
188  * Creates a new Cubemap object.
189  * @class Represents a webGL cubemapped texture resource
190  * @param {String} name The name of cubemap resource
191  * @param {String[]} imageURLs An array containing the URLs of the 6 faces of
192  * the cubemapped texture. The order of the pictures has to be X,Y,Z and within
193  * that, always positive first.
194  */
195 function Cubemap(name,imageURLs) {
196 	this.name=name;
197 	this.imageURLs=imageURLs;
198 	this.id=-1;
199 	this.images=new Array(6);
200 	for(var i=0;i<6;i++) {
201 		this.images[i] = new Image();
202 	}
203 }

205 /**
206  * Initiates the loading of one of the face textures of a cubemapped texture 
207  * object from file and sets a callback for the finish of the loading to 
208  * initiate loading the next face, the next cubemapped texture object from the
209  * resource center if such exists, otherwise to execute the callback passed in 
210  * parameter.
211  * This is required to make sure all the pictures are loaded by the time the
212  * final callback is executed, that is proceeding with any further steps already
213  * making use of the textures.
214  * @param {ResourceCenter} resourceCenter The resource center object where the
215  * subsequents cubemapped texture resources to be loaded are stored.
216  * @param {number} index The index of the current cubemap resource in the array
217  * of cubemaps.
218  * @param {number} face The index of the current face to load (0-5)
219  * @param {function} callback The function to be executed after all textures
220  * have been loaded.
221  */
222 Cubemap.prototype.chainLoad = function(resourceCenter,index,face,callback) {
223 	var self = this;
224 	if(face<5) {
225 		this.images[face].onload = function() {self.chainLoad(resourceCenter,index,face+1,callback);};
226 	} else {
227 		if(resourceCenter.cubemaps.length-1===index) {
228 			this.images[face].onload = callback;
229 		} else {
230 			this.images[face].onload = function() {resourceCenter.cubemaps[index+1].chainLoad(resourceCenter,index+1,0,callback);};
231 		}
232 	}
233 	this.images[face].src=this.imageURLs[face];
234 };

236 /**
237  * Sets up the webGL resource for the cubemapped texture within the provided GL
238  * context.
239  * Call only after the chainload has finished!
240  * @param {object} gl The webGL context.
241  */
242 Cubemap.prototype.setup = function(gl) {
243 	// Create a texture.
244 	this.id = gl.createTexture();
245 	gl.bindTexture(gl.TEXTURE_CUBE_MAP, this.id);
246 	
247 	// Set the parameters so we can render any size image.
248 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
249 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
250 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
251 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
252 	
253 	gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false);
254 	
255 	var type = [
256 		gl.TEXTURE_CUBE_MAP_POSITIVE_X,
257 		gl.TEXTURE_CUBE_MAP_NEGATIVE_X,
258 		gl.TEXTURE_CUBE_MAP_POSITIVE_Y,
259 		gl.TEXTURE_CUBE_MAP_NEGATIVE_Y,
260 		gl.TEXTURE_CUBE_MAP_POSITIVE_Z,
261 		gl.TEXTURE_CUBE_MAP_NEGATIVE_Z];
262 
263 	// Upload the images into the texture.
264 	for(var i=0;i<6;i++) {
265 		gl.texImage2D(type[i], 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.images[i]);
266 	}
267 };

286 /**
287  * Enumeration defining the available variables types of shaders.
288  */
289 var ShaderVariableTypes = Object.freeze(
290         {
291             none: 0,
292             float: 1, 
293             mat4: 2, 
294             mat3: 3,
295             vec3: 4,
296             vec4: 5,
297             sampler2D: 6,
298             samplerCube: 7
299         });
300  
301 /**
302  * Determining the enumeration value of a shader variable type from the string
303  * containing the name of the variable type. 
304  * @param {String} type The name of the variable type 
305  */
306 function ShaderVariableTypeFromString(type) {
307     if (type==="float") return ShaderVariableTypes.float;
308     if (type==="mat4") return ShaderVariableTypes.mat4;
309     if (type==="mat3") return ShaderVariableTypes.mat3;
310     if (type==="vec3") return ShaderVariableTypes.vec3;
311     if (type==="vec4") return ShaderVariableTypes.vec4;
312     if (type==="sampler2D") return ShaderVariableTypes.sampler2D;
313     if (type==="samplerCube") return ShaderVariableTypes.samplerCube;
314     return ShaderVariableTypes.none;
315 }

317 /**
318  * Creates a new ShaderUniform.
319  * @class A class representing and wrapping a GLSL uniform 
320  * variable.
321  * @param {String} name The name of the uniform variable. Has to be the same
322  * as the name specified in the GLSL source.
323  * @param {String} type The type of the uniform variable. Only certain variable
324  * types are supported.

326 function ShaderUniform(name,type) {
327 	this.name=name;
328 	this.type=ShaderVariableTypeFromString(type);
329 	this.location=-1;

332 /**
333  * Gets the location of the uniform variable in the specified GLSL program
334  * within the specified GL context, and sets the location member of the class
335  * to remember it.
336  * @param {object} gl The GL context.
337  * @param {number} programID The GLSL shader program ID.
338  * @returns {number} The location ID of the uniform in the program.
339  * */
340 ShaderUniform.prototype.getAndSetLocation = function(gl,programID) {
341 	this.location=gl.getUniformLocation(programID,this.name);
342         return this.location;
343 };

345 /**
346  * Get the GL function that sets the uniform of the specified type at the
347  * specified location to the specified value.
348  * @param {object} gl The GL context
349  * @param {number} location The location of the uniform
350  * @param {ShaderVariableTypes member} type The type of the uniform
351  * @param {object} value The value to set the uniform to (can be of different
352  * types)
353  * @returns {function} A function that executes the webGL query of setting the
354  * uniform to the specified value.
355  */
356 ShaderUniform.prototype.getSetterFunction = function(gl,location,type,value) {
357     switch(type) {
358         case ShaderVariableTypes.float: return function() { gl.uniform1f(location,value); };
359             break;
360         case ShaderVariableTypes.mat4: return function() { gl.uniformMatrix4fv(location,false,value); };
361             break;
362         case ShaderVariableTypes.mat3: return function() { gl.uniformMatrix3fv(location,false,value); };
363             break;
364         case ShaderVariableTypes.vec3: return function() { gl.uniform3fv(location,value); };
365             break;
366         case ShaderVariableTypes.vec4: return function() { gl.uniform4fv(location,value); };
367             break;
368         case ShaderVariableTypes.sampler2D: return function() { gl.uniform1i(location,value); };
369             break;
370         case ShaderVariableTypes.samplerCube: return function() { gl.uniform1i(location,value); };
371             break;
372     }
373 };
374 
375 /**
376  * Sets the value of the shader in the specified GL context to the return value
377  * of the passed value functio.
378  * @param {object} gl The GL context
379  * @param {function} valueFunction The function to calculate the uniform value
380  */
381 ShaderUniform.prototype.setValue = function(gl,valueFunction) {
382 	ShaderUniform.prototype.getSetterFunction(gl,this.location,this.type,valueFunction())();
383 };

414 /**
415  * Performs the full preparation of the webGL shader within the supplied GL
416  * context: dowloads the vertex and fragment shader files, compiles and links
417  * the code using webGL and saves the location IDs of the uniforms of the 
418  * shader.
419  * @param {object} gl The webGL context to use for webGL operations.
420  */
421 Shader.prototype.setup = function(gl) {
422 	var shaderSource = null;
423 	
424 	var request = new XMLHttpRequest();
425 	request.open('GET', this.vertexShaderFileName+"?1.0", false);
426 	request.send(null);
427 	shaderSource = request.responseText;
428 
429 	var vertexShader = gl.createShader(gl.VERTEX_SHADER);
430 	gl.shaderSource(vertexShader, shaderSource);
431 	gl.compileShader(vertexShader);
432 	
433 	request.open('GET', this.fragmentShaderFileName+"?1.0", false);
434 	request.send(null);
435 	shaderSource = request.responseText;
436 
437 	var fragmentShader = gl.createShader(gl.FRAGMENT_SHADER);
438 	gl.shaderSource(fragmentShader, shaderSource);
439 	gl.compileShader(fragmentShader);
440 
441 	this.id = gl.createProgram();
442 	gl.attachShader(this.id, vertexShader);
443         gl.attachShader(this.id, fragmentShader);
444         gl.linkProgram(this.id);
445 	
446 	for(var i=0;i<this.uniforms.length;i++) {
447 		this.uniforms[i].getAndSetLocation(gl,this.id);
448 	}
449 };

537 // if inside frustum, also sets visible width and height properties
538 VisualObject.prototype.insideViewFrustum = function(camera) {
539 	var baseMatrix =
540 		mul(mul(
541 			this.getCascadeScale(),
542 			translationMatrixv(getPositionVector4(
543 				mul(
544 					translationMatrixv(this.getCascadePositionVector()),
545 					camera.matrix
546 					)
547 				))
548 			),
549 			camera.perspective);
550 		
551 	var position = vector4Matrix4Product([0.0,0.0,0.0,1.0],baseMatrix);
552 	position = [position[0]/position[3],position[1]/position[3],position[2]/position[3]];
553 	var zOffsetPosition = vector4Matrix4Product([0.0,0.0,-this.getSize(),1.0],baseMatrix);
554 	var zOffset = (zOffsetPosition[2]/zOffsetPosition[3]);
555 		
556 	// frustum culling: back and front
557 	if (((zOffset>-1.0) && (zOffset<1.0)) || ((position[2]>-1.0) && (position[2]<1.0))) {
558 		var xyOffsetPosition = vector4Matrix4Product([this.getSize(),this.getSize(),0.0,1.0],baseMatrix);
559 		xyOffsetPosition = [xyOffsetPosition[0]/xyOffsetPosition[3],xyOffsetPosition[1]/xyOffsetPosition[3],xyOffsetPosition[2]/xyOffsetPosition[3]];
560 		var xOffset = xyOffsetPosition[0]-position[0];
561 		var yOffset = xyOffsetPosition[1]-position[1];
562 		
563 		// frustum culling: sides
564 		if ((position[0]+xOffset>-1)&&(position[0]-xOffset<1)&&
565 			(position[1]+yOffset>-1)&&(position[1]-yOffset<1)) {
566 			this.visibleWidth=xOffset;
567 			this.visibleHeight=yOffset;
568 			return true;
569 		} else {
570 			return false;
571 		}
572 	} else {
573 		return false;
574 	}
575 };

578 VisualObject.prototype.cascadeRender = function(parent,resourceCenter,scene,screenSize,lodContext,depthMask) {
579 	if(this.visible) {
580 		this.renderParent=parent;
581 		if((this.shader.depthMask===depthMask)&&(this.insideViewFrustum(scene.activeCamera))) {
582 			resourceCenter.setCurrentShader(this.shader,scene);
583 			this.assignUniforms(resourceCenter.gl);
584 			this.render(resourceCenter,screenSize,lodContext);
585 		} else if(scene.uniformsAssigned===false) {
586 			resourceCenter.setCurrentShader(this.shader,scene);
587 		}
588 	}
589 	for(var i=0;i<this.subnodes.length;i++) {
590 		this.subnodes[i].cascadeRender(this,resourceCenter,scene,screenSize,lodContext,depthMask);
591 	}

849 /**
850  * Creates a new SceneCamera.
851  * @class SceneCamera A camera that is used to draw a scene. Can follow one of
852  * the camera objects in the resource center, adapting its parameters to the
853  * ones of that camera in a given time.
854  * @extends Camera
855  * @param aspect The starting aspect ration of the camera.
856  * @param fov The starting field of view value of the camera.
857  * @param adaptationTime The time the camera will take when adapting its parameters
858  * to a new followed camera in milliseconds.
859  * @param followedCamera Initial camera object to follow.
860  * */
861 function SceneCamera(aspect,fov,adaptationTime,followedCamera) {
862     Camera.call(this,aspect,fov,true,true);
863     this.adaptationTime=adaptationTime;
864     this.followCamera(followedCamera);
865 }
866 
867 SceneCamera.prototype = new Camera();
868 SceneCamera.prototype.constructor = SceneCamera;
869 
870 SceneCamera.prototype.followCamera = function(camera) {
871     this.followedCamera=camera;
872     this.adaptationStartTime=new Date().getTime();
873     this.adaptationStartPosition=this.position;
874     this.adaptationStartOrientation=this.orientation;
875     this.adaptationStartFOV=this.fov;
876     this.adaptationTimeLeft=this.adaptationTime;
877 };
878 
879 SceneCamera.prototype.update = function() {
880     if(this.followedCamera!==undefined) {
881         if(this.adaptationTimeLeft>0) {
882             var currentTime=new Date().getTime();
883             var adaptationRate=Math.min(1.0,(currentTime-this.adaptationStartTime)/this.adaptationTime);
884             this.adaptationTimeLeft=this.adaptationTime-(currentTime-this.adaptationStartTime);
885             var trans = translationMatrix(
886                     (this.followedCamera.position[12]-this.adaptationStartPosition[12])*adaptationRate,
887                     (this.followedCamera.position[13]-this.adaptationStartPosition[13])*adaptationRate,
888                     (this.followedCamera.position[14]-this.adaptationStartPosition[14])*adaptationRate
889                     );
890             var newPosition=translate(this.adaptationStartPosition,trans);
891             var velocityMatrix = mul(translationMatrix(
892                 newPosition[12]-this.position[12],
893                 newPosition[13]-this.position[13],
894                 newPosition[14]-this.position[14]),this.orientation);
895             this.velocity = [velocityMatrix[12],velocityMatrix[13],velocityMatrix[14]];
896             this.position=newPosition;
897             this.orientation=correctOrthogonalMatrix(addMatrices4(
898                 mulMatrix4Scalar(this.adaptationStartOrientation,1.0-adaptationRate),
899                 mulMatrix4Scalar(this.followedCamera.orientation,adaptationRate)));
900             this.setFOV(this.adaptationStartFOV+(this.followedCamera.fov-this.adaptationStartFOV)*adaptationRate);
901         } else {
902             this.position=this.followedCamera.position;
903             this.orientation=this.followedCamera.orientation;
904             this.matrix=this.followedCamera.matrix;
905             this.perspective=this.followedCamera.perspective;
906         }
907     }
908 };

1163 ResourceCenter.prototype.setupBuffers = function(shader,loadLines) {
1164 	//alert("setting up vertex buffers for shader: "+shader.name);
1165 	this.gl.useProgram(shader.id);
1166 	var sumVertices=0;
1167 	for(var i=0;i<this.models.length;i++) {
1168 		sumVertices=sumVertices+
1169 			(loadLines?this.models[i].lines.length*2:0)+
1170 			this.models[i].triangles.length*3;
1171 	}
1172 	
1173 	// creating float32arrays of the appropriate size for the buffers
1174 	shader.vertexBuffers=new Array();
1175 	for(var i=0;i<shader.attributes.length;i++) {
1176 		shader.vertexBuffers.push(new VertexBuffer(-1,new Float32Array(sumVertices*shader.attributes[i].size),null,shader.attributes[i].size));
1177 	}
1178 	
1179 	// filling the buffer data arrays from model data
1180 	var bufferSize=0;
1181 	var objectBufferData=null;
1182 	for(var i=0;i<this.models.length;i++) {
1183 		var linesToLoad=loadLines;
1184 		var trianglesToLoad=true;
1185 		while(linesToLoad||trianglesToLoad) {
1186 			objectBufferData=this.models[i].getBuffers(linesToLoad);
1187 			linesToLoad?
1188 				(this.models[i].bufferStartLines=bufferSize):
1189 				(this.models[i].bufferStart=bufferSize);
1190 			for(var j=0;j<shader.attributes.length;j++) {
1191 				shader.vertexBuffers[j].data.set(objectBufferData[shader.attributes[j].role],bufferSize*shader.attributes[j].size);
1192 			}
1193 			bufferSize+=Math.round(objectBufferData[shader.attributes[0].role].length/shader.attributes[0].size);
1194 			linesToLoad?
1195 				(linesToLoad=false):
1196 				(trianglesToLoad=false);
1197 		}
1198 	}
1199 	
1200 	// provide the data to webGL
1201 	for(var i=0;i<shader.attributes.length;i++) {
1202 		shader.vertexBuffers[i].location = this.gl.getAttribLocation(shader.id, shader.attributes[i].name);
1203 		shader.vertexBuffers[i].id = this.gl.createBuffer();
1204 		this.gl.bindBuffer(this.gl.ARRAY_BUFFER, shader.vertexBuffers[i].id);
1205 		this.gl.bufferData(
1206 			this.gl.ARRAY_BUFFER, 
1207 			shader.vertexBuffers[i].data,
1208 			this.gl.STATIC_DRAW);
1209 		this.gl.enableVertexAttribArray(shader.vertexBuffers[i].location);
1210 		this.gl.vertexAttribPointer(shader.vertexBuffers[i].location, shader.attributes[i].size, this.gl.FLOAT, false, 0, 0);
1211 	}
1212 };
