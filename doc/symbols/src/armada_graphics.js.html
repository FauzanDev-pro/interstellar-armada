  9 /**
 10  * Creates a new Texture object.
 11  * @class Represents a webGL texture resource
 12  * @param {String} filename The name of file from which the texture resource 
 13  * is to be loaded. The constructor itself does not initiate the loading.
 14  */
 15 function Texture(filename) {
 16 	this.filename=filename;
 17 	this.id=-1;
 18 	this.image = new Image();
 19 }

 21 /**
 22  * Initiates the loading of the texture from file and sets a callback for the
 23  * finish of the loading to initiate loading the next texture from the resource
 24  * center if such exists, otherwise to execute the callback passed in parameter.
 25  * This is required to make sure all the textures are loaded by the time the
 26  * final callback is executed, that is proceeding with any further steps already
 27  * making use of the textures.
 28  * @param {ResourceCenter} resourceCenter The resource center object where the
 29  * subsequents texture resources to be loaded are stored.
 30  * @param {number} index The index of the current texture resource in the array
 31  * of textures.
 32  * @param {function} callback The function to be executed after all textures
 33  * have been loaded.
 34  */
 35 Texture.prototype.chainLoad = function(resourceCenter,index,callback) {
 36 	var self = resourceCenter.textures[index];
 37 	if(resourceCenter.textures.length-1===index) {
 38 		self.image.onload = callback;
 39 	} else {
 40 		self.image.onload = function() {self.chainLoad(resourceCenter,index+1,callback);};
 41 	}
 42 	self.image.src=self.filename;
 43 };

 45 /**
 46  * Sets up the webGL resource for the texture within the provided GL context.
 47  * Call only after the chainload has finished!
 48  * @param {object} gl The webGL context.
 49  */
 50 Texture.prototype.setup = function(gl) {
 51 	this.id = gl.createTexture();
 52 	gl.bindTexture(gl.TEXTURE_2D, this.id);
 53 	
 54 	// Set the parameters so we can render any size image.
 55 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
 56 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
 57 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
 58 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
 59 
 60 	// Upload the image into the texture.
 61 	gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.image);
 62 };

222 /**
223  * Initiates the loading of one of the face textures of a cubemapped texture 
224  * object from file and sets a callback for the finish of the loading to 
225  * initiate loading the next face, the next cubemapped texture object from the
226  * resource center if such exists, otherwise to execute the callback passed in 
227  * parameter.
228  * This is required to make sure all the pictures are loaded by the time the
229  * final callback is executed, that is proceeding with any further steps already
230  * making use of the textures.
231  * @param {ResourceCenter} resourceCenter The resource center object where the
232  * subsequents cubemapped texture resources to be loaded are stored.
233  * @param {number} index The index of the current cubemap resource in the array
234  * of cubemaps.
235  * @param {number} face The index of the current face to load (0-5)
236  * @param {function} callback The function to be executed after all textures
237  * have been loaded.
238  */
239 Cubemap.prototype.chainLoad = function(resourceCenter,index,face,callback) {
240 	var self = this;
241 	if(face<5) {
242 		this.images[face].onload = function() {self.chainLoad(resourceCenter,index,face+1,callback);};
243 	} else {
244 		if(resourceCenter.cubemaps.length-1===index) {
245 			this.images[face].onload = callback;
246 		} else {
247 			this.images[face].onload = function() {resourceCenter.cubemaps[index+1].chainLoad(resourceCenter,index+1,0,callback);};
248 		}
249 	}
250 	this.images[face].src=this.imageURLs[face];
251 };

305  */
306 var ShaderVariableTypes = Object.freeze(
307         {
308             none: 0,
309             float: 1, 
310             mat4: 2,

318 /**
319  * Determining the enumeration value of a shader variable type from the string
320  * containing the name of the variable type. 
321  * @param {String} type The name of the variable type 
322  */
323 function ShaderVariableTypeFromString(type) {
324     if (type==="float") return ShaderVariableTypes.float;
325     if (type==="mat4") return ShaderVariableTypes.mat4;
326     if (type==="mat3") return ShaderVariableTypes.mat3;
327     if (type==="vec3") return ShaderVariableTypes.vec3;
328     if (type==="vec4") return ShaderVariableTypes.vec4;
329     if (type==="sampler2D") return ShaderVariableTypes.sampler2D;
330     if (type==="samplerCube") return ShaderVariableTypes.samplerCube;
331     return ShaderVariableTypes.none;
332 }

362 /**
363  * Get the GL function that sets the uniform of the specified type at the
364  * specified location to the specified value.
365  * @param {object} gl The GL context
366  * @param {number} location The location of the uniform
367  * @param {ShaderVariableTypes member} type The type of the uniform
368  * @param {object} value The value to set the uniform to (can be of different
369  * types)
370  * @returns {function} A function that executes the webGL query of setting the
371  * uniform to the specified value.
372  */
373 ShaderUniform.prototype.getSetterFunction = function(gl,location,type,value) {
374     switch(type) {
375         case ShaderVariableTypes.float: return function() { gl.uniform1f(location,value); };
376             break;
377         case ShaderVariableTypes.mat4: return function() { gl.uniformMatrix4fv(location,false,value); };
378             break;
379         case ShaderVariableTypes.mat3: return function() { gl.uniformMatrix3fv(location,false,value); };
380             break;
381         case ShaderVariableTypes.vec3: return function() { gl.uniform3fv(location,value); };
382             break;
383         case ShaderVariableTypes.vec4: return function() { gl.uniform4fv(location,value); };
384             break;
385         case ShaderVariableTypes.sampler2D: return function() { gl.uniform1i(location,value); };
386             break;
387         case ShaderVariableTypes.samplerCube: return function() { gl.uniform1i(location,value); };
388             break;
389     }
390 };

446 /**
447  * Performs the full preparation of the webGL shader within the supplied GL
448  * context: dowloads the vertex and fragment shader files, compiles and links
449  * the code using webGL and saves the location IDs of the uniforms of the 
450  * shader.
451  * @param {object} gl The webGL context to use for webGL operations.
452  */
453 Shader.prototype.setup = function(gl) {
454 	var shaderSource = null;
455 	
456 	var request = new XMLHttpRequest();
457 	request.open('GET', this.vertexShaderFileName+"?1.0", false);
458 	request.send(null);
459 	shaderSource = request.responseText;
460 
461 	var vertexShader = gl.createShader(gl.VERTEX_SHADER);
462 	gl.shaderSource(vertexShader, shaderSource);
463 	gl.compileShader(vertexShader);
464 	
465 	request.open('GET', this.fragmentShaderFileName+"?1.0", false);
466 	request.send(null);
467 	shaderSource = request.responseText;
468 
469 	var fragmentShader = gl.createShader(gl.FRAGMENT_SHADER);
470 	gl.shaderSource(fragmentShader, shaderSource);
471 	gl.compileShader(fragmentShader);
472 
473 	this.id = gl.createProgram();
474 	gl.attachShader(this.id, vertexShader);
475         gl.attachShader(this.id, fragmentShader);
476         gl.linkProgram(this.id);
477 	
478 	for(var i=0;i<this.uniforms.length;i++) {
479 		this.uniforms[i].getAndSetLocation(gl,this.id);
480 	}
481 };

632 // if inside frustum, also sets visible width and height properties
633 /**
634  * Checks if the object is inside the viewing frustum of the passed camera,
635  * taking into account the parents of the object as well. Also sets the view
636  * width and height members of the object.
637  * @param {Camera} camera The camera the frustum of which is to be checked
638  * @returns {boolean} Whether the object is inside the frustum.
639  */
640 VisualObject.prototype.isInsideViewFrustum = function(camera) {
641 	var baseMatrix =
642 		mul(mul(
643 			this.getCascadeScalingMatrix(),
644 			translationMatrixv(getPositionVector4(
645 				mul(
646 					translationMatrixv(this.getCascadePositionVector()),
647 					camera.matrix
648 					)
649 				))
650 			),
651 			camera.perspectiveMatrix);
652 		
653 	var position = vector4Matrix4Product([0.0,0.0,0.0,1.0],baseMatrix);
654 	position = [position[0]/position[3],position[1]/position[3],position[2]/position[3]];
655 	var zOffsetPosition = vector4Matrix4Product([0.0,0.0,-this.getSize(),1.0],baseMatrix);
656 	var zOffset = (zOffsetPosition[2]/zOffsetPosition[3]);
657 		
658 	// frustum culling: back and front
659 	if (((zOffset>-1.0) && (zOffset<1.0)) || ((position[2]>-1.0) && (position[2]<1.0))) {
660 		var xyOffsetPosition = vector4Matrix4Product([this.getSize(),this.getSize(),0.0,1.0],baseMatrix);
661 		xyOffsetPosition = [xyOffsetPosition[0]/xyOffsetPosition[3],xyOffsetPosition[1]/xyOffsetPosition[3],xyOffsetPosition[2]/xyOffsetPosition[3]];
662 		var xOffset = xyOffsetPosition[0]-position[0];
663 		var yOffset = xyOffsetPosition[1]-position[1];
664 		
665 		// frustum culling: sides
666 		if ((position[0]+xOffset>-1)&&(position[0]-xOffset<1)&&
667 			(position[1]+yOffset>-1)&&(position[1]-yOffset<1)) {
668 			this.visibleWidth=xOffset;
669 			this.visibleHeight=yOffset;
670 			return true;
671 		} else {
672                         this.visibleWidth=0;
673 			this.visibleHeight=0;
674 			return false;
675 		}
676 	} else {
677                 this.visibleWidth=0;
678 		this.visibleHeight=0;
679 		return false;
680 	}
681 };

684  * Renders the object and all its subnodes.
685  * @param {VisualObject} parent The rendering parent to which the position,
686  * orientation and scaling of the object is relative.
687  * @param {ResourceCenter} resourceCenter The resource center that holds the
688  * necessary rendering resources .
689  * @param {Scene} scene The scene within which the object is located.
690  * @param {number} screenSize The size of the rendering viewport in pixels,
691  * to determine the actual drawn size of the object (for dynamic LOD)
692  * @param {boolean} depthMaskPhase Whether we are drawing in the depthmask
693  * enabled or disabled phase (renders only phase matches with the type of the
694  * shared the object has)
695  * */ 
696 VisualObject.prototype.cascadeRender = function(parent,resourceCenter,scene,screenSize,depthMaskPhase) {
697 	if(this.visible) {
698 		this.renderParent=parent;
699 		if((this.shader.depthMask===depthMaskPhase)&&(this.isInsideViewFrustum(scene.activeCamera))) {

703 		} else if(scene.uniformsAssigned===false) {
704 			resourceCenter.setCurrentShader(this.shader,scene);
705 		}
706 	}
707 	for(var i=0;i<this.subnodes.length;i++) {
708 		this.subnodes[i].cascadeRender(this,resourceCenter,scene,screenSize,depthMaskPhase);
709 	}

1606 			this.gl.STATIC_DRAW);

1608 		this.gl.vertexAttribPointer(shader.vertexBuffers[i].location, shader.attributes[i].size, this.gl.FLOAT, false, 0, 0);

1646  */

1648 	try {
1649 		// Try to grab the standard context. If it fails, fallback to experimental.
1650 		this.gl = canvas.getContext("webgl",{alpha: false}) || canvas.getContext("experimental-webgl",{alpha: false});
1651 	}
1652 	catch(e) {}
1653 
1654 	// If we don't have a GL context, give up now
1655 	if (!this.gl) {
1656 		alert("Unable to initialize WebGL. Your browser might not support it.");
1657 	}
1658 	
1659 	this.gl.clearDepth(1.0);
1660 	this.gl.colorMask(true, true, true, true);
1661 	this.gl.enable(this.gl.DEPTH_TEST);
1662 	this.gl.depthFunc(this.gl.LEQUAL); 
1663 	this.gl.enable(this.gl.CULL_FACE);
1664 	this.gl.cullFace(this.gl.BACK);
1665 	this.gl.frontFace(this.gl.CCW);
1666 	this.gl.blendFunc(this.gl.SRC_ALPHA, this.gl.ONE_MINUS_SRC_ALPHA);
1667 	this.gl.enable(this.gl.BLEND);
1668 	this.gl.enable(this.gl.SCISSOR_TEST);
1669 	this.gl.activeTexture(this.gl.TEXTURE0);
1670 	
1671 	this.boundTexture=null;
1672 	
1673 	this.setupShaders();
1674 	this.setupTextures();
1675 	this.setupCubemaps();
1676 	
1677 	for(var i=0;i<this.shaders.length;i++) {
1678 		this.setupBuffers(this.shaders[i],true);
1679 	}
1680 };
